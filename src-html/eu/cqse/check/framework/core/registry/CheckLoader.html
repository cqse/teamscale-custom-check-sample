001package eu.cqse.check.framework.core.registry;
002
003import java.lang.reflect.Field;
004import java.util.Arrays;
005import java.util.HashMap;
006import java.util.Map;
007
008import org.conqat.lib.commons.assertion.CCSMAssert;
009import org.conqat.lib.commons.collections.CollectionUtils;
010import org.conqat.lib.commons.logging.ILogger;
011import org.conqat.lib.commons.logging.SimpleLogger;
012import org.conqat.lib.commons.reflect.ReflectionUtils;
013
014import eu.cqse.check.framework.core.Check;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckImplementationBase;
017import eu.cqse.check.framework.core.CheckInfo;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.option.CheckOption;
020import eu.cqse.check.framework.core.option.CheckOptionWrapper;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.shallowparser.ShallowParserFactory;
023import eu.cqse.check.framework.typetracker.TypeTrackerFactory;
024import eu.cqse.check.util.clang.ClangUtils;
025
026/**
027 * Class for constructing a {@link CheckInfo} Object from a Custom Check Class
028 * ({@link CheckImplementationBase}).
029 */
030public class CheckLoader {
031
032        /**
033         * Dummy logger that is used while creating check implementation base instances.
034         */
035        private static final ILogger LOGGER = new SimpleLogger();
036
037        /** The class to load a check from. */
038        private final Class<?> checkClass;
039
040        /** Constructor. */
041        private CheckLoader(Class<?> checkClass) {
042                CCSMAssert.isNotNull(checkClass);
043                this.checkClass = checkClass;
044        }
045
046        /**
047         * Tries to load a check info from the check class that was set in the
048         * constructor. Returns null, if the given check class is not annotated with
049         * {@link Check}.
050         *
051         * @throws CheckException
052         *             if loading fails
053         */
054        private CheckInfo load() throws CheckException {
055                Check checkAnnotation = checkClass.getAnnotation(Check.class);
056                if (checkAnnotation == null) {
057                        return null;
058                }
059
060                CheckImplementationBase dummyInstance = CheckImplementationBase.createInstance(checkClass, LOGGER);
061
062                if (!languagesSupportParameters(checkAnnotation.languages(), checkAnnotation.parameters())) {
063                        throw new CheckException("Unsupported parameters!");
064                }
065
066                return new CheckInfo(checkClass, checkAnnotation.name(), checkAnnotation.description(),
067                                checkAnnotation.groupName(), checkAnnotation.categoryName(), checkAnnotation.defaultEnablement(),
068                                checkAnnotation.languages(), checkAnnotation.parameters(), checkAnnotation.phases(),
069                                loadOptions(dummyInstance));
070        }
071
072        /** Returns whether the given parameters support the given languages. */
073        private static boolean languagesSupportParameters(ELanguage[] languages, ECheckParameter[] parameters) {
074                for (ECheckParameter parameter : parameters) {
075                        if (parameter == ECheckParameter.CLANG) {
076                                if (CollectionUtils.intersectionSet(ClangUtils.CLANG_ENABLED_LANGUAGES, Arrays.asList(languages))
077                                                .isEmpty()) {
078                                        return false;
079                                }
080
081                                continue;
082                        }
083
084                        for (ELanguage language : languages) {
085                                if (!languageSupportParameter(language, parameter)) {
086                                        return false;
087                                }
088                        }
089                }
090                return true;
091        }
092
093        /** Returns whether the given parameter supports the given language. */
094        private static boolean languageSupportParameter(ELanguage language, ECheckParameter parameter) {
095                if (parameter == ECheckParameter.ABSTRACT_SYNTAX_TREE) {
096                        return ShallowParserFactory.supportsLanguage(language);
097                } else if (parameter == ECheckParameter.TYPE_RESOLUTION) {
098                        return TypeTrackerFactory.supportsLanguage(language);
099                }
100                return false;
101        }
102
103        /**
104         * Load options for a check. The given dummy instance is used to retrieve
105         * default values.
106         */
107        private Map<String, CheckOptionWrapper<?>> loadOptions(CheckImplementationBase dummyInstance)
108                        throws CheckException {
109                Map<String, CheckOptionWrapper<?>> options = new HashMap<>();
110                for (Field attribute : ReflectionUtils.getAllFields(checkClass)) {
111                        CheckOptionWrapper<?> option = getOptionFromField(attribute, dummyInstance);
112
113                        if (option == null) {
114                                continue;
115                        }
116
117                        if (options.put(option.getName(), option) != null) {
118                                throw new CheckException(
119                                                "Check " + checkClass.getName() + " has duplicated option: " + option.getName());
120                        }
121
122                }
123                return options;
124        }
125
126        /**
127         * Returns a CheckOption for the given field. If the given field is not
128         * annotated with ACheckOption null is returned.
129         */
130        @SuppressWarnings({ "rawtypes", "unchecked" })
131        private CheckOptionWrapper<?> getOptionFromField(Field field, CheckImplementationBase dummyInstance)
132                        throws CheckException {
133                field.setAccessible(true);
134                CheckOption annotation = field.getAnnotation(CheckOption.class);
135                if (annotation == null) {
136                        return null;
137                }
138
139                try {
140                        Object defaultValue = field.get(dummyInstance);
141                        if (defaultValue == null) {
142                                throw new CheckException("Check " + checkClass.getName() + " must have default values for options.");
143                        }
144
145                        Class<?> type = field.getType();
146                        if (type.isPrimitive()) {
147                                // If the type is primitive we wrap it into a the
148                                // corresponding wrapper class, as the whole custom check
149                                // option framework works with non-primitive types only
150                                // Writing values back to the attribute works due to
151                                // auto-boxing.
152                                type = ReflectionUtils.resolvePrimitiveClass(type);
153                        }
154
155                        return new CheckOptionWrapper(annotation, field, defaultValue, type);
156                } catch (IllegalAccessException e) {
157                        CCSMAssert.fail("Could not access default value: " + e.getMessage());
158                }
159                return null;
160        }
161
162        /**
163         * Tries to load a check info from the given check class. Returns null, if the
164         * given check class is not annotated with {@link Check}.
165         *
166         * @throws CheckException
167         *             if loading fails
168         */
169        public static CheckInfo loadFromClass(Class<?> checkClass) throws CheckException {
170                return new CheckLoader(checkClass).load();
171        }
172}
