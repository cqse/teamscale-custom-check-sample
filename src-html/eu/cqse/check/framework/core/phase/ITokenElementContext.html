006package eu.cqse.check.framework.core.phase;
007
008import java.util.List;
009import java.util.Optional;
010
011import org.conqat.lib.commons.logging.ILogger;
012
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.phase.ECodeViewOption.ETextViewOption;
016import eu.cqse.check.framework.preprocessor.IPreprocessor;
017import eu.cqse.check.framework.preprocessor.PreprocessorFactory;
018import eu.cqse.check.framework.scanner.ELanguage;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021import eu.cqse.check.framework.typetracker.ITypeResolution;
022import eu.cqse.check.util.clang.ClangTranslationUnitWrapper;
023
024/**
025 * Context interface for accessing different views of a token element.
026 */
027public interface ITokenElementContext {
028
029        /**
030         * Returns the language of the analyzed token element.
031         */
032        ELanguage getLanguage();
033
034        /**
035         * Returns the uniform path of the analyzed token element.
036         */
037        String getUniformPath();
038
039        /**
040         * Returns the tokens in the given view. This method can be used, if the
041         * executing check requests the {@link ECheckParameter#ABSTRACT_SYNTAX_TREE}
042         * parameter.
043         */
044        List<IToken> getTokens(ECodeViewOption view) throws CheckException;
045
046        /**
047         * Returns the text content in the given view. For most checks the preferred
048         * form of content will be {@link ETextViewOption#FILTERED_CONTENT}, as usually
049         * no findings shall be generated for filtered code.
050         */
051        String getTextContent(ETextViewOption view) throws CheckException;
052
053        /**
054         * Returns the synthetic root entity of the abstract syntax tree based on the
055         * given view.
056         * <p>
057         * This method can be used, if the executing check requests the
058         * {@link ECheckParameter#ABSTRACT_SYNTAX_TREE} parameter.
059         */
060        ShallowEntity getRootEntity(ECodeViewOption view) throws CheckException;
061
062        /**
063         * Returns the abstract syntax tree of the analyzed token element as list of
064         * shallow entities based on the given view.
065         * <p>
066         * This method can be used, if the executing check requests the
067         * {@link ECheckParameter#ABSTRACT_SYNTAX_TREE} parameter.
068         */
069        List<ShallowEntity> getAbstractSyntaxTree(ECodeViewOption view) throws CheckException;
070
071        /**
072         * Returns a logger, which may be used to indicate problems in situations where
073         * the check may continue. In more severe cases, a {@link CheckException} should
074         * be thrown instead.
075         */
076        ILogger getLogger();
077
078        /**
079         * Returns a type resolution, which contains information about known variables
080         * and their types. This method can be used, if the executing check request the
081         * {@link ECheckParameter#TYPE_RESOLUTION} parameter. Based on the given
082         * {@link ECodeViewOption}.
083         */
084        ITypeResolution getTypeResolution(ECodeViewOption view) throws CheckException;
085
086        /**
087         * Calculates the pre-processed tokens (no global include resolution).
088         */
089        default List<IToken> calculateLocalPreprocessedTokens(List<IToken> tokens) {
090                Optional<IPreprocessor> preprocessor = PreprocessorFactory.createLocalPreprocessor(getLanguage());
091
092                if (preprocessor.isPresent()) {
093                        return preprocessor.get().preprocess(getUniformPath(), tokens);
094                }
095
096                return tokens;
097        }
098
099        /**
100         * Returns a {@link ClangTranslationUnitWrapper} containing the initialized JNI
101         * interface to Clang for the current file. To access this, specify the
102         * {@link ECheckParameter#CLANG} option in the custom check.
103         */
104        ClangTranslationUnitWrapper getClangTranslationUnitWrapper();
105}
