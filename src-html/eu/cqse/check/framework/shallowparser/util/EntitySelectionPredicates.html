043/**
044 * Defines valid primitive expressions that can be used in the
045 * {@link EntitySelectionExpressionParser}. Each expression is defined by a
046 * public static method of same name. Methods found here must be parameterless
047 * or accept a single string parameter and must return an {@link Predicate} over
048 * {@link ShallowEntity}. To avoid clashes with Java keywords, an optional
049 * prefix "select" may be used.
050 * 
051 * Note that the comments for the public methods describe not exactly the
052 * function, but rather the returned predicate.
053 */
054public class EntitySelectionPredicates {
055
056        /** The visibility modifiers in C++. */
057        private static final EnumSet<ETokenType> CPP_VISIBILITY_MODIFIERS = EnumSet.of(ETokenType.PUBLIC, PRIVATE,
058                        ETokenType.PROTECTED);
059
060        /**
061         * Selects entities in files with the given language. If the given language name
062         * is invalid, no entities are matched.
063         */
064        public static Predicate<ShallowEntity> language(String languageName) {
065                ELanguage language = EnumUtils.valueOfIgnoreCase(ELanguage.class, languageName);
066                if (language == null) {
067                        return entity -> false;
068                }
069                return entity -> getLanguage(entity) == language;
070        }

072        /** Selects all modules/namespaces. */
073        public static Predicate<ShallowEntity> module() {
074                return typePredicate(EShallowEntityType.MODULE);
075        }
076
077        /** Selects all types/classes. */
078        public static Predicate<ShallowEntity> type() {
079                return typePredicate(EShallowEntityType.TYPE).and(entity -> !entity.getSubtype().equals("typedef"));
080        }
081
082        /** Selects all methods/functions. */
083        public static Predicate<ShallowEntity> method() {
084                return entity -> entity.getType() == EShallowEntityType.METHOD
085                                && (entity.getParent() == null || entity.getParent().getType() != EShallowEntityType.ATTRIBUTE);
086        }
087
088        /** Selects all shallow entities whose subtype is a declaration. */
089        public static Predicate<ShallowEntity> declaration() {
090                return entity -> {
091                        if (entity.getSubtype().contains(SubTypeNames.DECLARATION)) {
092                                return true;
093                        }
094
095                        // also consider inline methods in classes
096                        if (ELanguage.CPP.equals(getLanguage(entity))) {
097                                ShallowEntity parent = entity.getParent();
098                                return parent != null && parent.getType() == EShallowEntityType.TYPE;
099                        }
100                        return false;
101                };
102        }
103
104        /** Selects all shallow entities that are internal (C# keyword). */
105        public static Predicate<ShallowEntity> internal() {
106                return entity -> {
107                        if (TokenStreamUtils.containsAny(entity.ownStartTokens(), ETokenType.INTERNAL)) {
108                                return true;
109                        }
110                        ShallowEntity parent = entity.getParent();
111                        return parent != null && TokenStreamUtils.containsAny(parent.ownStartTokens(), ETokenType.INTERNAL);
112                };
113        }

115        /** Selects all attributes. */
116        public static Predicate<ShallowEntity> attribute() {
117                return typePredicate(EShallowEntityType.ATTRIBUTE);
118        }

120        /**
121         * Selects all properties, defined attributes that have only method children.
122         * More specific: ATTRIBUTE entities that have children and each child is a
123         * METHOD entity. E.g., C# properties.
124         */
125        public static Predicate<ShallowEntity> property() {
126                return entity -> {
127                        if (entity.getType() != EShallowEntityType.ATTRIBUTE) {
128                                return false;
129                        }
130
131                        for (ShallowEntity child : entity.getChildren()) {
132                                if (child.getType() != EShallowEntityType.METHOD) {
133                                        return false;
134                                }
135                        }
136                        return !entity.getChildren().isEmpty();
137                };
138        }

140        /** Selects all statements. */
141        public static Predicate<ShallowEntity> statement() {
142                return typePredicate(EShallowEntityType.STATEMENT);
143        }

145        /** Selects all meta information (defines, annotations, etc.). */
146        public static Predicate<ShallowEntity> meta() {
147                return typePredicate(EShallowEntityType.META);
148        }

150        /** Creates a predicate for checking the type of an entity. */
151        private static Predicate<ShallowEntity> typePredicate(EShallowEntityType type) {
152                return entity -> entity.getType() == type;
153        }

155        /**
156         * Matches all public entities as well as methods inside interfaces, which are
157         * considered public in most languages.
158         */
159        public static Predicate<ShallowEntity> selectPublic() {
160                return (modifierPredicate(PUBLIC).or(EntitySelectionPredicates::isInterfaceMethod)
161                                .or(EntitySelectionPredicates::isJavaScriptInterface))
162                                                .and(EntitySelectionPredicates::isNotTypeScriptConstructor);
163        }
164
165        /**
166         * Returns <code>true</code> if the given entity is is a JavaScript interface.
167         * JavaScript has no modifiers, so all interfaces are public.
168         */
169        private static boolean isJavaScriptInterface(ShallowEntity entity) {
170                return getLanguage(entity) == ELanguage.JAVASCRIPT && entity.getType() == EShallowEntityType.TYPE
171                                && modifierPredicate(INTERFACE).test(entity);
172        }
173
174        /**
175         * Returns <code>true</code> if the given entity is a method in an interface.
176         */
177        private static boolean isInterfaceMethod(ShallowEntity entity) {
178                ShallowEntity parent = entity.getParent();
179                return entity.getType() == EShallowEntityType.METHOD && parent != null && parent.getType() == TYPE
180                                && modifierPredicate(INTERFACE).test(parent);
181        }
182
183        /** Matches all protected entities. */
184        public static Predicate<ShallowEntity> selectProtected() {
185                return modifierPredicate(ETokenType.PROTECTED).and(EntitySelectionPredicates::isNotTypeScriptConstructor);
186        }

188        /** Matches all entities marked with override. */
189        public static Predicate<ShallowEntity> selectOverride() {
190                return modifierPredicate(ETokenType.OVERRIDE);
191        }

193        /**
194         * Matches all private entities (except the TypeScript constructor parameters).
195         */
196        public static Predicate<ShallowEntity> selectPrivate() {
197                return modifierPredicate(PRIVATE).and(EntitySelectionPredicates::isNotTypeScriptConstructor);
198        }

200        /**
201         * Checks that the entity is not a TypeScript constructor.
202         */
203        private static boolean isNotTypeScriptConstructor(ShallowEntity entity) {
204                if (getLanguage(entity) == ELanguage.JAVASCRIPT) {
205                        return !SubTypeNames.CONSTRUCTOR.equals(entity.getSubtype());
206                }
207                return true;
208        }
209
210        /** Matches all entities with default visibility. */
211        public static Predicate<ShallowEntity> selectDefault() {
212                return element -> {
213                        if (getLanguage(element) == ELanguage.JAVASCRIPT) {
214                                return isTypeScriptDefault(element);
215                        }
216                        return hasNoModifiers(element);
217                };
218        }
219
220        /**
221         * Checks that the given entity has no modifiers. The method checks for the
222         * following modifiers: PUBLIC, PRIVATE and PROTECTED.
223         */
224        private static boolean hasNoModifiers(ShallowEntity element) {
225                List<IToken> tokens = element.ownStartTokens();
226                return !TokenStreamUtils.containsAny(tokens, 0, tokens.size(), ETokenType.PUBLIC, ETokenType.PROTECTED,
227                                PRIVATE);
228        }
229
230        /**
231         * Returns <code>true</code> if the given entity is a TypeScript attribute or
232         * method that could be selected as 'DEFAULT' and has no explicit modifier (i.e
233         * default visibility). These include:
234         * <ol>
235         * <li>Types</li>
236         * <li>Constructors</li>
237         * <li>All methods except lambda methods (since they will already be covered by
238         * the check on their parent attributes)</li>
239         * <li>Attributes</li>
240         * </ol>
241         */
242        private static boolean isTypeScriptDefault(ShallowEntity entity) {
243                if (getLanguage(entity) != ELanguage.JAVASCRIPT) {
244                        return false;
245                }
246
247                // Special handling of constructors needed to support public attributes
248                // / members, e.g.
249                // constructor(public pos: Vector, lookAt: Vector) { ... }
250                boolean isConstructor = entity.getType() == EShallowEntityType.METHOD
251                                && SubTypeNames.CONSTRUCTOR.equals(entity.getSubtype());
252                if (isConstructor) {
253                        UnmodifiableList<IToken> tokens = entity.ownStartTokens();
254                        if (!tokens.isEmpty() && tokens.get(0).getText().equals(SubTypeNames.CONSTRUCTOR)) {
255                                return true;
256                        }
257                }
258
259                if (!hasNoModifiers(entity)) {
260                        return false;
261                }
262
263                // ignore lambdas
264                if (entity.getType() == EShallowEntityType.METHOD && SubTypeNames.LAMBDA.equals(entity.getSubtype())) {
265                        return false;
266                }
267
268                return entity.getType() == EShallowEntityType.TYPE || entity.getType() == EShallowEntityType.METHOD
269                                || entity.getType() == EShallowEntityType.ATTRIBUTE;
270        }

287        /** Returns a predicate that checks for existence of the given modifier. */
288        private static Predicate<ShallowEntity> modifierPredicate(final ETokenType modifier) {
289                return element -> {
290                        if (TokenStreamUtils.firstTokenOfType(element.ownStartTokens(), modifier) != TokenStreamUtils.NOT_FOUND) {
291                                return true;
292                        }
293
294                        if (ELanguage.CPP == getLanguage(element) && CPP_VISIBILITY_MODIFIERS.contains(modifier)) {
295                                return hasCppVisibilityModifer(element, modifier);
296                        }
297                        return false;
298                };
299        }

309        /**
310         * Determines the origin, i.e. the uniformPath, of the given entity or
311         * <code>null</code> if it cannot be determined.
312         */
313        private static String getUniformPath(ShallowEntity entity) {
314                UnmodifiableList<IToken> tokens = entity.includedTokens();
315                if (tokens.isEmpty()) {
316                        return null;
317                }
318                return CollectionUtils.getAny(tokens).getOriginId();
319        }

321        /**
322         * Checks whether the given shallow entity is affected by the specified C++
323         * visibility modifier by backwards searching the sibling shallow entities. The
324         * given modifier must be one of public, protected, or private.
325         */
326        private static boolean hasCppVisibilityModifer(ShallowEntity entity, ETokenType modifier) {
327                ShallowEntity parent = entity.getParent();
328                if (parent == null) {
329                        return false;
330                }
331
332                UnmodifiableList<ShallowEntity> siblings = parent.getChildren();
333                for (int i = siblings.indexOf(entity) - 1; i >= 0; i--) {
334                        ShallowEntity sibling = siblings.get(i);
335
336                        if (isCppVisibilityModifier(sibling)) {
337                                return TokenStreamUtils.containsAny(sibling.ownStartTokens(), modifier);
338                        }
339                }
340
341                // no explicit modifier found -> assume default
342                if (parent.getType() == EShallowEntityType.TYPE) {
343                        if (SubTypeNames.STRUCT.equals(parent.getSubtype())) {
344                                return ETokenType.PUBLIC.equals(modifier);
345                        }
346                        return PRIVATE.equals(modifier);
347                }
348
349                return false;
350        }
351
352        /** Returns whether the given entity is a C++ visibility modifier. */
353        private static boolean isCppVisibilityModifier(ShallowEntity entity) {
354                return entity.getType() == EShallowEntityType.META
355                                && TokenStreamUtils.containsAny(entity.ownStartTokens(), CPP_VISIBILITY_MODIFIERS);
356        }

368        /**
369         * Matches entities by their parent type's subtype (e.g. to select all members
370         * of interfaces).
371         */
372        public static Predicate<ShallowEntity> typeSubtype(String subtype) {
373                return entity -> {
374                        ShallowEntity parent = entity.getParent();
375                        return parent != null && type().test(parent) && parent.getSubtype().equals(subtype);
376                };
377        }

383                return entity -> {
384                        ShallowEntity parent = entity.getParent();
385                        return parent != null && type().test(parent) && export().test(parent);

394        /** Matches entities by their name using regular expression. */
395        public static Predicate<ShallowEntity> nameRegex(String regex) {
396                Pattern pattern = Pattern.compile(regex);
397                // name is null for e.g. lambdas and anonymous types
398                return entity -> entity.getName() != null && pattern.matcher(entity.getName()).matches();
399        }
400
401        /** Matches entities by their file name */
402        public static Predicate<ShallowEntity> fileRegex(String regex) {
403                Pattern pattern = Pattern.compile(regex);
404                return entity -> {
405                        String uniformPath = getUniformPath(entity);
406                        if (uniformPath == null) {
407                                return false;
408                        }
409                        return pattern.matcher(uniformPath).matches();
410                };
411        }

413        /**
414         * Matches all entities that are annotated with an annotation of given name
415         * (excluding the '@' sign).
416         */
417        public static Predicate<ShallowEntity> annotated(String annotationName) {
418                final String normalizedAnnotationName = StringUtils.stripPrefix(annotationName, "@");
419                return entity -> {
420                        ShallowEntity parent = entity.getParent();
421                        if (parent == null) {
422                                return false;
423                        }
424
425                        // Special handling needed for JavaScript, since annotations are
426                        // detected as decorators.
427                        if (getLanguage(entity) == ELanguage.JAVASCRIPT) {
428                                return hasAnnotation(annotationName, entity, parent.getChildren(), SubTypeNames.DECORATOR);
429                        }
430
431                        return hasAnnotation(normalizedAnnotationName, entity, parent.getChildren(), SubTypeNames.ANNOTATION);
432                };
433        }

435        /**
436         * Checks if the entity is annotated with the specified annotation. Traverses
437         * the siblings before the entity in the children list, as more than one
438         * annotation might exist.
439         */
440        private static boolean hasAnnotation(final String annotationName, ShallowEntity entity,
441                        List<ShallowEntity> children, String subTypeNameToCheck) {
442
443                int index = children.indexOf(entity);
444                index -= 1;
445                while (index >= 0 && children.get(index).getType() == EShallowEntityType.META) {
446                        ShallowEntity metaEntity = children.get(index);
447                        if (subTypeNameToCheck.equals(metaEntity.getSubtype())
448                                        && annotationName.equalsIgnoreCase(metaEntity.getName())) {
449                                return true;
450                        }
451                        index -= 1;
452                }
453                return false;
454        }

456        /**
457         * Matches simple getters, i.e. methods starting with "get" or "is" and with at
458         * most one contained statement.
459         */
460        public static Predicate<ShallowEntity> simpleGetter() {
461                return (simpleMethod("get").or(simpleMethod("is"))).and(modifierPredicate(ETokenType.ABSTRACT).negate());
462        }

473         * Matches simple setters, i.e. methods starting with "set" (case insensitive)

489        /**
490         * Matches getter in CSharp generated by the automatic property generation
491         * feature. These are recognized by the entity subtype '(empty) get'
492         */
493        public static Predicate<ShallowEntity> simpleCSharpGetter() {
494                return entity -> {
495                        boolean isSimpleMethod = entity.getType() == EShallowEntityType.METHOD && getStatementCount(entity) <= 1;
496                        return isSimpleMethod && StringUtils.equalsOneOf(entity.getSubtype().toLowerCase(), SubTypeNames.GET,
497                                        SubTypeNames.EMPTY_GET);
498                };
499        }

503         * feature. These are recognized by the entity subtype '(empty) set'

505        public static Predicate<ShallowEntity> simpleCSharpSetter() {
506                return entity -> {
507                        boolean isSimpleMethod = entity.getType() == EShallowEntityType.METHOD && getStatementCount(entity) <= 1;
508                        return isSimpleMethod && StringUtils.equalsOneOf(entity.getSubtype().toLowerCase(), SubTypeNames.SET,
509                                        SubTypeNames.EMPTY_SET);

513        /**
514         * Matches simple methods with the given prefix. A method is simple, if it
515         * contains as most one statement.
516         */
517        public static Predicate<ShallowEntity> simpleMethod(String namePrefix) {
518                return entity -> {
519                        boolean isSimpleMethod = entity.getType() == EShallowEntityType.METHOD && getStatementCount(entity) <= 1;
520                        return isSimpleMethod && (entity.getName() != null) && entity.getName().startsWith(namePrefix);
521                };
522        }
523
524        /**
525         * Matches simple methods with the given prefix (case insensitive). A method is
526         * simple, if it contains as most one statement.
527         */
528        public static Predicate<ShallowEntity> simpleMethodCaseInsensitive(String namePrefix) {
529                return entity -> {
530                        boolean isSimpleMethod = entity.getType() == EShallowEntityType.METHOD && getStatementCount(entity) <= 1;
531                        return isSimpleMethod && (entity.getName() != null)
532                                        && entity.getName().toLowerCase().startsWith(namePrefix.toLowerCase());
533                };
534        }
535
536        /**
537         * Returns the number of statements contained as children of the entity.
538         */
539        private static int getStatementCount(ShallowEntity entity) {
540                return ShallowEntityTraversalUtils.listEntitiesOfType(entity.getChildren(), EShallowEntityType.STATEMENT)
541                                .size();
542        }
