020import java.util.EnumSet;

029/**
030 * A node used for recognizing a part of the system. The parsers constructed
031 * work by trying a list of recognizers and "using" the first that matches.
032 * Recognizers can be chained to match more complicated patterns and allow
033 * branching.
034 * 
035 * @author $Author: kinnen $
036 * @version $Rev: 53381 $
037 * @ConQAT.Rating GREEN Hash: 3BEDC8D58F2BC17E93B5F07E1BD0992D
038 */
039public abstract class RecognizerBase<STATE extends Enum<STATE>> {
040
041        /** Constant used to indicate that no match occurred. */
042        public static final int NO_MATCH = -1;
043
044        /**
045         * The recognizers that are to be executed after this one. If there are more
046         * than one, they are tried in order.
047         */
048        protected final List<RecognizerBase<STATE>> tailRecognizers = new ArrayList<RecognizerBase<STATE>>();
049
050        /**
051         * Creates a recognizer for a certain sequence and moves the offset to the
052         * end of the match.
053         * 
054         * @param matchTerms
055         *            a list of match terms which must match in order. These may be
056         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
057         *            of them.
058         */
059        public RecognizerBase<STATE> sequence(Object... matchTerms) {
060                return appendRecognizer(new SequenceRecognizer<STATE>(matchTerms, false));
061        }
062
063        /**
064         * Creates a recognizer for a certain sequence and moves the offset to the
065         * beginning of the match (in contrast to {@link #sequence(Object...)},
066         * which moves the offset to the end of the match). This is useful to
067         * implement branches in the parser.
068         * 
069         * @param matchTerms
070         *            a list of match terms which must match in order. These may be
071         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
072         *            of them.
073         */
074        public RecognizerBase<STATE> sequenceBefore(Object... matchTerms) {
075                return appendRecognizer(new SequenceRecognizer<STATE>(matchTerms, true));
076        }
077
078        /**
079         * Creates a recognizer that matches a certain sequence repeatedly
080         * (including no matches).
081         * 
082         * @param matchTerms
083         *            a list of match terms which must match in order. These may be
084         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
085         *            of them.
086         */
087        public RecognizerBase<STATE> repeated(Object... matchTerms) {
088                return appendRecognizer(new RepeatedRecognizer<STATE>(matchTerms, 0,
089                                Integer.MAX_VALUE));
090        }
091
092        /**
093         * Creates a recognizer that matches the given sub-recognizer the given
094         * number of times. Note that you cannot use this method with a recognizer
095         * that does not advance the parser state. Use
096         * {@link #preCondition(RecognizerBase)} instead.
097         */
098        public RecognizerBase<STATE> subRecognizer(RecognizerBase<STATE> sub,
099                        int minRepetitions, int maxRepetitions) {
100                return appendRecognizer(new SubRecognizer<STATE>(sub, minRepetitions,
101                                maxRepetitions));
102        }
103
104        /**
105         * Creates a recognizer that matches the given sub-recognizer once but does
106         * not advance the parser state. This can be used to check if the given
107         * recognizer would apply at the current position and then perform other
108         * actions first, e.g. close the current node.
109         */
110        public RecognizerBase<STATE> preCondition(RecognizerBase<STATE> sub) {
111                return appendRecognizer(new PreconditionRecognizer<STATE>(sub));
112        }
113
114        /**
115         * Creates a recognizer that matches a certain sequence optionally.
116         * 
117         * @param matchTerms
118         *            a list of match terms which must match in order. These may be
119         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
120         *            of them.
121         */
122        public RecognizerBase<STATE> optional(Object... matchTerms) {
123                return appendRecognizer(new RepeatedRecognizer<STATE>(matchTerms, 0, 1));
124        }
125
126        /**
127         * Creates a recognizer that skips until the given match terms are found.
128         * The new position is set behind the found match.
129         * 
130         * @param matchTerms
131         *            a list of match terms which must match in order. These may be
132         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
133         *            of them.
134         */
135        public RecognizerBase<STATE> skipTo(Object... matchTerms) {
136                return appendRecognizer(new SkipToRecognizer<STATE>(matchTerms, false));
137        }
138
139        /**
140         * Creates a recognizer that skips until the given match terms are found.
141         * The new position is set directly before the found match.
142         * 
143         * @param matchTerms
144         *            a list of match terms which must match in order. These may be
145         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
146         *            of them.
147         */
148        public RecognizerBase<STATE> skipBefore(Object... matchTerms) {
149                return appendRecognizer(new SkipToRecognizer<STATE>(matchTerms, true));
150        }
151
152        /**
153         * Skips a region consisting of nested elements if it exists (i.e., the
154         * skipping is optional).
155         */
156        public RecognizerBase<STATE> skipNested(ETokenType open, ETokenType close) {
157                return skipNested(open, close, null);
158        }
159
160        /**
161         * Skips a region consisting of nested elements if it exists (i.e., the
162         * skipping is optional). The skipped region will be processed by the given
163         * subRecognizer (may be null), which can optionally create further
164         * entities.
165         */
166        public RecognizerBase<STATE> skipNested(ETokenType open, ETokenType close,
167                        RecognizerBase<STATE> subRecognizer) {
168                // we can not simply use the SkipToRecognizer here, as it implements
169                // greedy skipping, i.e. would also skip multiple nested structures
170                // most once if possible at all.
171                return appendRecognizer(new OptionalNestedRecognizer<STATE>(open,
172                                close, subRecognizer));
173        }
174
175        /** Skips all tokens that are of one of the given types (if any). */
176        public RecognizerBase<STATE> skipAny(final EnumSet<ETokenType> types) {
177                return appendRecognizer(new RecognizerBase<STATE>() {
178                        @Override
179                        protected int matchesLocally(ParserState<STATE> parserState,
180                                        List<IToken> tokens, int startOffset) {
181                                while (startOffset < tokens.size()
182                                                && types.contains(tokens.get(startOffset).getType())) {
183                                        startOffset += 1;
184                                }
185                                return startOffset;
186                        }
187                });
188        }
189
190        /**
191         * Skips all tokens up to the given match term. If the match term occurs
192         * within a nested structure, it is ignored.
193         */
194        public RecognizerBase<STATE> skipToWithNesting(Object matchTerm,
195                        ETokenType open, ETokenType close) {
196                return skipToWithNesting(matchTerm, open, close, null);
197        }
198
199        /**
200         * Skips all tokens up to the given match term. If the match term occurs
201         * within a nested structure, it is ignored. The skipped region will be
202         * processed by the given subRecognizer (may be null), which can optionally
203         * create further entities.
204         */
205        public RecognizerBase<STATE> skipToWithNesting(Object matchTerm,
206                        ETokenType open, ETokenType close,
207                        RecognizerBase<STATE> subRecognizer) {
208                SkipToRecognizer<STATE> skipper = new SkipToRecognizer<STATE>(
209                                new Object[] { matchTerm }, false);
210                skipper.addNestingTokens(open, close);
211                skipper.setSubRecognizer(subRecognizer);
212                return appendRecognizer(skipper);
213        }
214
215        /**
216         * Skips all tokens up to the given match term. If the match term occurs
217         * within a nested structure, it is ignored. The new position is set
218         * directly before the found match.
219         */
220        public RecognizerBase<STATE> skipBeforeWithNesting(Object matchTerm,
221                        ETokenType open, ETokenType close) {
222                SkipToRecognizer<STATE> skipper = new SkipToRecognizer<STATE>(
223                                new Object[] { matchTerm }, true);
224                skipper.addNestingTokens(open, close);
225                return appendRecognizer(skipper);
226        }
227
228        /**
229         * Skips all tokens up to the given match term. If the match term occurs
230         * within a nested structure, it is ignored. The new position is set
231         * directly before the found match.
232         */
233        public RecognizerBase<STATE> skipBeforeWithNesting(Object matchTerm,
234                        ETokenType open1, ETokenType close1, ETokenType open2,
235                        ETokenType close2) {
236                return skipBeforeWithNesting(matchTerm, open1, close1, open2, close2,
237                                null);
238        }
239
240        /**
241         * Skips all tokens up to the given match term. If the match term occurs
242         * within a nested structure, it is ignored. The new position is set
243         * directly before the found match. The skipped region will be processed by
244         * the given subRecognizer (may be null), which can optionally create
245         * further entities.
246         */
247        public RecognizerBase<STATE> skipBeforeWithNesting(Object matchTerm,
248                        ETokenType open1, ETokenType close1, ETokenType open2,
249                        ETokenType close2, RecognizerBase<STATE> subRecognizer) {
250                SkipToRecognizer<STATE> skipper = new SkipToRecognizer<STATE>(
251                                new Object[] { matchTerm }, true);
252                skipper.addNestingTokens(open1, close1);
253                skipper.addNestingTokens(open2, close2);
254                skipper.setSubRecognizer(subRecognizer);
255                return appendRecognizer(skipper);
256        }

258        /**
259         * Marks a different matching start position, which can be used to reference
260         * tokens during node creation.
261         */
262        public RecognizerBase<STATE> markStart() {
263                return appendRecognizer(new RecognizerBase<STATE>() {
264                        @Override
265                        public int matchesLocally(ParserState<STATE> parserState,
266                                        List<IToken> tokens, int startOffset) {
267                                parserState.markReferencePosition(startOffset);
268                                return startOffset;
269                        }
270                });
271        }

273        /**
274         * Creates a shallow parser entity node without a name.
275         * 
276         * @param subtype
277         *            this may be either a string which is used directly, or an
278         *            integer which is interpreted as an offset into the tokens
279         *            matched so far. The index may also be negative, which means
280         *            that the matching occurs from the end. If an array is
281         *            provided, the entries of the array are treated as described
282         *            and concatenated with spaces. If a {@link Region} is given,
283         *            the start and end values define a range of indexes.
284         */
285        public RecognizerBase<STATE> createNode(EShallowEntityType type,
286                        Object subtype) {
287                return createNode(type, subtype, null);
288        }

290        /**
291         * Creates a shallow parser entity node.
292         * 
293         * @param subtype
294         *            this may be either a string which is used directly, or an
295         *            integer which is interpreted as an offset into the tokens
296         *            matched so far. The index may also be negative, which means
297         *            that the matching occurs from the end. If an array is
298         *            provided, the entries of the array are treated as described
299         *            and concatenated with spaces. If a {@link Region} is given,
300         *            the start and end values define a range of indexes.
301         * @param name
302         *            this may be either null, a string which is used directly, or
303         *            an integer which is interpreted as an offset into the tokens
304         *            matched so far. The index may also be negative, which means
305         *            that the matching occurs from the end. If an array is
306         *            provided, the entries of the array are treated as described
307         *            and concatenated with spaces. If a {@link Region} is given,
308         *            the start and end values define a range of indexes.
309         */
310        public RecognizerBase<STATE> createNode(final EShallowEntityType type,
311                        final Object subtype, final Object name) {
312                return appendRecognizer(new CreateNodeRecognizer<STATE>(type, subtype,
313                                name, 0));
314        }

317         * Creates a shallow parser entity node, offsetting the node start by the

320         * @param subtype
321         *            this may be either a string which is used directly, or an

323         *            matched so far. The index may also be negative, which means
324         *            that the matching occurs from the end. If an array is

328         * @param name
329         *            this may be either null, a string which is used directly, or
330         *            an integer which is interpreted as an offset into the tokens
331         *            matched so far. The index may also be negative, which means
332         *            that the matching occurs from the end. If an array is
333         *            provided, the entries of the array are treated as described
334         *            and concatenated with spaces. If a {@link Region} is given,
335         *            the start and end values define a range of indexes.
336         * @param offset
337         *            an offset in the token stream. Moves the node start that many

340        public RecognizerBase<STATE> createNode(EShallowEntityType type,

342                return appendRecognizer(new CreateNodeRecognizer<STATE>(type, subtype,

346        /** Completes node construction. */
347        public void endNode() {
348                appendRecognizer(new RecognizerBase<STATE>() {
349                        @Override
350                        public int matches(ParserState<STATE> parserState,
351                                        List<IToken> tokens, int startOffset) {
352                                parserState.endNode(false);
353                                return startOffset;
354                        }
355                });
356        }
357
358        /**
359         * Completes node construction and overrides the name of the current node.
360         * This can be used for long constructs where the name is found at the end.
361         */
362        public void endNodeWithName(final Object name) {
363                appendRecognizer(new RecognizerBase<STATE>() {
364                        @Override
365                        public int matches(ParserState<STATE> parserState,
366                                        List<IToken> tokens, int startOffset) {
367                                parserState.setNodeName(RecognizerUtils.resolveName(tokens,
368                                                parserState, startOffset, name, CreateNodeRecognizer
369                                                                .shouldExtractStringContentFromName(parserState
370                                                                                .getCurrentEntityType())));
371                                parserState.endNode(false);
372                                return startOffset;
373                        }
374                });
375        }
376
377        /**
378         * Completes node construction and signals that this parsed node is only a
379         * partial language construct, i.e. parsing should in any case continue at
380         * the same level.
381         */
382        public void endNodeWithContinuation() {
383                appendRecognizer(new RecognizerBase<STATE>() {
384                        @Override
385                        public int matches(ParserState<STATE> parserState,
386                                        List<IToken> tokens, int startOffset) {
387                                parserState.endNode(true);
388                                return startOffset;
389                        }
390                });
391        }

406        /** Attempts to parse exactly once in the given state. */
407        public RecognizerBase<STATE> parseOnce(final STATE state) {
408                return appendRecognizer(new RecognizerBase<STATE>() {
409                        @Override
410                        public int matchesLocally(ParserState<STATE> parserState,
411                                        List<IToken> tokens, int startOffset) {
412                                return parserState.parse(state, tokens, startOffset);
413                        }
414                });
415        }

417        /** Matches if the parser is currently parsing on the top-level. */
418        public RecognizerBase<STATE> ensureTopLevel() {
419                return appendRecognizer(new RecognizerBase<STATE>() {
420                        @Override
421                        protected int matchesLocally(ParserState<STATE> parserState,
422                                        List<IToken> tokens, int startOffset) {
423                                if (parserState.getEntityStackSize() == 0) {
424                                        return startOffset;
425                                }
426                                return NO_MATCH;
427                        }
428                });
429        }
430
431        /** Template method that is called to append a recognizer to this node. */
432        private RecognizerBase<STATE> appendRecognizer(
433                        RecognizerBase<STATE> recognizer) {
434                tailRecognizers.add(recognizer);
435                return recognizer;
436        }
437
438        /**
439         * Attempts to match the recognizer including any tail recognizers.
440         * 
441         * @param tokens
442         *            the list of tokens.
443         * @param startOffset
444         *            the current offset into the tokens from where to start
445         *            matching.
446         * @return the new offset (position right behind the match) or
447         *         {@link #NO_MATCH} if matching was not possible at this position.
448         */
449        public int matches(ParserState<STATE> parserState, List<IToken> tokens,
450                        int startOffset) {
451                int localOffset = matchesLocally(parserState, tokens, startOffset);
452                if (localOffset == NO_MATCH) {
453                        return NO_MATCH;
454                }
455
456                if (tailRecognizers.isEmpty()) {
457                        return localOffset;
458                }
459
460                for (RecognizerBase<STATE> recognizer : tailRecognizers) {
461                        int match = recognizer.matches(parserState, tokens, localOffset);
462                        if (match != NO_MATCH) {
463                                return match;
464                        }
465                }
466
467                return NO_MATCH;
468        }
469
470        /**
471         * Attempts to match the recognizer locally (without respecting tail
472         * recognizers).
473         * 
474         * @param parserState
475         *            the current state of the underlying parser.
476         * @param tokens
477         *            the list of tokens.
478         * @param startOffset
479         *            the current offset into the tokens from where to start
480         *            matching.
481         * @return the new offset (position right behind the match) or
482         *         {@link #NO_MATCH} if matching was not possible at this position.
483         */
484        protected int matchesLocally(ParserState<STATE> parserState,
485                        List<IToken> tokens, int startOffset) {
486                return startOffset;
487        }
