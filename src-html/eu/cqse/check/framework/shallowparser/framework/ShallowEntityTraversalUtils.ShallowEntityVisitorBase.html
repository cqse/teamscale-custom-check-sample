035        /** Lists all entities. */
036        public static List<ShallowEntity> listAllEntities(Collection<ShallowEntity> entities) {

048        /**
049         * Lists all entities of the given types, includes matching child entities
050         * recursively.
051         */
052        public static List<ShallowEntity> listEntitiesOfTypes(Collection<ShallowEntity> entities,
053                        Set<EShallowEntityType> types) {
054                return new CollectingVisitorBase() {
055                        @Override
056                        protected boolean collect(ShallowEntity entity) {
057                                return types.contains(entity.getType());
058                        }
059                }.apply(entities);
060        }

062        /** Lists all entities that are selected by the given predicate. */
063        public static List<ShallowEntity> selectEntities(Collection<ShallowEntity> entities,
064                        Predicate<ShallowEntity> predicate) {
065                return new CollectingVisitorBase() {
066                        @Override
067                        protected boolean collect(ShallowEntity entity) {
068                                return predicate.test(entity);
069                        }
070                }.apply(entities);
071        }

073        /**
074         * Returns the first incomplete entity found (or null). Unclosed entities
075         * correspond to parsing errors.
076         */
077        private static ShallowEntity findIncompleteEntity(ShallowEntity entity) {
078                if (!entity.isCompleted()) {
079                        return entity;
080                }
081                return findIncompleteEntity(entity.getChildren());
082        }
083
084        /**
085         * Returns the first incomplete entity found (or null). Unclosed entities
086         * correspond to parsing errors.
087         */
088        public static ShallowEntity findIncompleteEntity(List<ShallowEntity> entities) {
089                for (ShallowEntity entity : entities) {
090                        ShallowEntity incomplete = findIncompleteEntity(entity);
091                        if (incomplete != null) {
092                                return incomplete;
093                        }
094                }
095                return null;
096        }

098        /**
099         * Traverses the given collection of entities and returns a flat list of
100         * entities.
101         */
102        public static List<ShallowEntity> getAllEntities(Collection<ShallowEntity> entities) {
103                return new CollectingVisitorBase() {
104                        @Override
105                        protected boolean collect(ShallowEntity entity) {
106                                return true;
107                        }
108                }.apply(entities);
109        }
110
111        /**
112         * Lists all entities of the given type, but does not traverse into the
113         * entities, i.e. entities contained in the returned ones are not returned.
114         */
115        public static List<ShallowEntity> listEntitiesOfTypeNonRecursive(Collection<ShallowEntity> entities,
116                        EShallowEntityType type) {
117                List<ShallowEntity> methods = new ArrayList<>();
118                ShallowEntity.traverse(entities, new ShallowEntityVisitorBase() {
119                        /** {@inheritDoc} */
120                        @Override
121                        public boolean visit(ShallowEntity entity) {
122                                if (entity.getType() == type) {
123                                        methods.add(entity);
124                                        // do not recurse into methods
125                                        return false;
126                                }
127                                return true;
128                        }
129                });
130                return methods;
131        }

133        /**
134         * Returns all methods found in the given entities, but does not traverse into
135         * the entities, i.e. methods contained in other methods are not returned.
136         */
137        public static List<ShallowEntity> listMethodsNonRecursive(List<ShallowEntity> entities) {
138                return listEntitiesOfTypeNonRecursive(entities, EShallowEntityType.METHOD);
139        }

141        /** Empty default implementation of {@link IShallowEntityVisitor}. */
142        public static abstract class ShallowEntityVisitorBase implements IShallowEntityVisitor {
143                /** {@inheritDoc} */
144                @Override
145                public boolean visit(ShallowEntity entity) {
146                        return true;
147                }
148
149                /** {@inheritDoc} */
150                @Override
151                public void endVisit(ShallowEntity entity) {
152                        // nothing
153                }
154        }
155
156        /**
157         * Base class for visitors that collect shallow entities, includes matching
158         * child entities recursively.
159         */
160        public static abstract class CollectingVisitorBase extends ShallowEntityVisitorBase {
161
162                /** The collected entities. */
163                private final List<ShallowEntity> entities = new ArrayList<>();
164
165                /** {@inheritDoc} */
166                @Override
167                public boolean visit(ShallowEntity entity) {
168                        if (collect(entity)) {
169                                entities.add(entity);
170                        }
171                        return true;
172                }
173
174                /**
175                 * Template method that returns true if the entity should be collected.
176                 */
177                protected abstract boolean collect(ShallowEntity entity);
178
179                /**
180                 * Applies this collecting visitor depth-first to the given entities and returns
181                 * the collected result.
182                 */
183                public List<ShallowEntity> apply(Collection<ShallowEntity> entities) {
184                        ShallowEntity.traverse(entities, this);
185                        return this.entities;
186                }
187        }

189        /** Utility method for traversing multiple entities. */
190        public static void traverseWithKey(List<ShallowEntity> entities, IShallowEntityKeyVisitor visitor) {
191                traverseWithKey(entities, visitor, new int[0]);
192        }
193
194        /**
195         * Utility method for traversing multiple entities.
196         * 
197         * @param parentKey
198         *            Specifies the unique key of the entity lists parent.
199         */
200        private static void traverseWithKey(List<ShallowEntity> entities, IShallowEntityKeyVisitor visitor,
201                        int[] parentKey) {
202                int length = parentKey.length;
203                for (int i = 0; i < entities.size(); i++) {
204                        ShallowEntity entity = entities.get(i);
205                        int[] key = Arrays.copyOf(parentKey, length + 1);
206                        key[length] = i;
207                        if (visitor.visit(key, entity)) {
208                                traverseWithKey(entity.getChildren(), visitor, key);
209                        }
210                }
211        }

213        /** Empty default implementation of {@link IShallowEntityVisitor}. */
214        public interface IShallowEntityKeyVisitor {
215
216                /**
217                 * Denotes that visiting the entity begins.
218                 * 
219                 * @param key
220                 *            Key, which uniquely identifies the entity for the traversed list
221                 *            of {@link ShallowEntity}s.
222                 * 
223                 * @return true if the children of this entity are to be visited as well.
224                 */
225                public boolean visit(int[] key, ShallowEntity entity);
226        }
227
228        /**
229         * Returns a {@link ShallowEntity} from the given list of entities, which is
230         * identified by the given key.
231         * 
232         * @see ShallowEntityTraversalUtils#traverseWithKey(List,
233         *      IShallowEntityKeyVisitor)
234         */
235        public static ShallowEntity getEntityFromListWithKey(List<ShallowEntity> entities, int[] key) {
236                if (key.length == 0 || key[0] >= entities.size()) {
237                        return null;
238                }
239                ShallowEntity entity = entities.get(key[0]);
240                if (key.length == 1) {
241                        return entity;
242                }
243                return getEntityFromListWithKey(entity.getChildren(), Arrays.copyOfRange(key, 1, key.length));
244        }

246        /**
247         * Returns the next {@link ShallowEntity} after the given entity. Returns
248         * <code>null</code> if there is no subsequent entity.
249         */
250        public static ShallowEntity getSubsequentEntity(ShallowEntity entity) {
251                boolean found = false;
252                while (entity != null && entity.getParent() != null) {
253                        List<ShallowEntity> neighboringEntities = entity.getParent().getChildren();
254                        for (ShallowEntity neighboringEntity : neighboringEntities) {
255                                if (found) {
256                                        return neighboringEntity;
257                                } else if (neighboringEntity.equals(entity)) {
258                                        found = true;
259                                }
260                        }
261                        entity = entity.getParent();
262                        found = false;
263                }
264                return null;
265        }
266
267        /**
268         * Returns the previous {@link ShallowEntity} before the given entity. Returns
269         * <code>null</code> if there is no previous entity.
270         */
271        public static ShallowEntity getPreviousEntity(ShallowEntity entity) {
272                ShallowEntity previousEntity = null;
273                while (entity != null && entity.getParent() != null) {
274                        List<ShallowEntity> neighboringEntities = entity.getParent().getChildren();
275                        for (ShallowEntity neighboringEntity : neighboringEntities) {
276                                if (neighboringEntity.equals(entity)) {
277                                        return previousEntity;
278                                }
279                                previousEntity = neighboringEntity;
280                        }
281                        entity = entity.getParent();
282                }
283                return null;
284        }

300        /** Returns the entity containing the given (one-based) line number. */
301        private static Optional<ShallowEntity> findEntityForLine(int line, boolean recursive,
302                        List<ShallowEntity> entities) {
303                // we know that entities are sorted according to line number, so we can
304                // binary search them
305                @SuppressWarnings("serial")
306                ShallowEntity comparisonEntity = new ShallowEntity(null, null, null, null, 0) {
307                        @Override
308                        public int getStartLine() {
309                                return line;
310                        }
311                };
312                int index = Collections.binarySearch(entities, comparisonEntity,
313                                Comparator.comparingInt(ShallowEntity::getStartLine));
314                if (index < 0) {
315                        // we want insertion point +1 in this case
316                        index = -index - 2;
317                }
318                if (index >= 0 && index < entities.size() && entities.get(index).getStartLine() <= line
319                                && line <= entities.get(index).getEndLine()) {
320                        ShallowEntity match = entities.get(index);
321                        if (recursive && match.hasChildren()) {
322                                Optional<ShallowEntity> betterMatch = findEntityForLine(line, match.getChildren());
323                                if (betterMatch.isPresent()) {
324                                        return betterMatch;
325                                }
326                        }
327                        return Optional.of(match);
328                }
329
330                return Optional.empty();
331        }

333        /**
334         * Traverses the parent hierarchy of the given startEntity upwards. Returns the
335         * first parent that matches the given predicate or Optional.empty if no entity
336         * matches the predicate.
337         */
338        public static Optional<ShallowEntity> findParentEntity(ShallowEntity startEntity,
339                        Predicate<ShallowEntity> shallowEntityPredicate) {
340                ShallowEntity current = startEntity;
341                while (current.getParent() != null) {
342                        current = current.getParent();
343                        if (shallowEntityPredicate.test(current)) {
344                                return Optional.of(current);
345                        }
346                }
347                return Optional.empty();
348        }
349
350        /**
351         * Traverses the parent hierarchy of the given startEntity upwards. Returns all
352         * the shallow entities that match the given predicate, or an empty list if no
353         * entity matches the predicate.
354         */
355        public static List<ShallowEntity> findMatchingParentEntities(ShallowEntity startEntity,
356                        Predicate<ShallowEntity> shallowEntityPredicate) {
357                List<ShallowEntity> parentEntities = new ArrayList<>();
358                ShallowEntity current = startEntity;
359                while (current.getParent() != null) {
360                        current = current.getParent();
361                        if (shallowEntityPredicate.test(current)) {
362                                parentEntities.add(current);
363                        }
364                }
365                return parentEntities;
366        }
