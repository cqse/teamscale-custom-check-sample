080public class TsqlShallowParser
081                extends ShallowParserBase<TsqlShallowParser.ETsqlParserStates> {
082
083        /** The states used in this parser. */
084        public static enum ETsqlParserStates {
085
086                /** Top-level parsing state. */
087                TOP_LEVEL,
088
089                IN_METHOD, IN_BLOCK,
090
091        }
092
093        private static final EnumSet<ETokenType> SELECT_STATEMENT_KEYWORDS = EnumSet
094                        .of(SELECT, WITH, ORDER, FOR, OPTION, UNION, ALL, EXCEPT, INTERSECT,
095                                        DISTINCT, TOP, PERCENT, INTO, FROM, WHERE, GROUP, BY,
096                                        HAVING);
097
098        private static final EnumSet<ETokenType> NO_SELECT_STATEMENT_KEYWORDS = EnumSet
099                        .copyOf(ETokenType.KEYWORDS);
100
101        {
102                NO_SELECT_STATEMENT_KEYWORDS.removeAll(SELECT_STATEMENT_KEYWORDS);
103        }
104
105        /**
106         * @param stateClass
107         * @param initialState
108         */
109        public TsqlShallowParser() {
110                super(ETsqlParserStates.class, ETsqlParserStates.TOP_LEVEL);
111
112                createMethodRules();
113                createStatementRules();
114        }
115
116        /** Creates parser rules for statements. */
117        private void createStatementRules() {
118
119                createBlockStatementRule(IF);
120                createBlockStatementRule(ELSE);
121                createBlockStatementRule(WHILE);
122
123                inAnyState().sequence(BEGIN).createNode(STATEMENT, 0, new Region(1, -1))
124                                .parseUntil(IN_BLOCK).sequence(END).endNode();
125
126                inAnyState().sequence(SELECT).createNode(STATEMENT, 0)
127                                .skipBefore(NO_SELECT_STATEMENT_KEYWORDS).endNode();
128
129                inAnyState().markStart().sequence(KEYWORD).skipBefore(KEYWORD)
130                                .createNode(STATEMENT, 0).endNode();
131
132                // TODO (DP): EOF is not recognized, so that the node is always
133                // incomplete. Should be STATEMENT, not META
134                inAnyState().sequence(KEYWORD).createNode(META, 0).skipBefore(EOF)
135                                .endNode();
136
137        }
138
139        /**
140         * @param blockStatementType
141         */
142        private void createBlockStatementRule(ETokenType blockStatementType) {
143                inAnyState().sequence(blockStatementType).skipNested(LPAREN, RPAREN)
144                                .skipBefore(KEYWORD).sequence(BEGIN).createNode(STATEMENT, 0)
145                                .parseUntil(IN_BLOCK).sequence(END).endNode();
146        }
147
148        /** Creates parser rules for functions and procedures. */
149        private void createMethodRules() {
150                // TODO (DP): We might need to create an Identifier set, which copes
151                // with [FUNCTION] or [My cool table name]
152
153                RecognizerBase<ETsqlParserStates> methodStart = inState(TOP_LEVEL)
154                                .sequence(CREATE).markStart().sequence(FUNCTION)
155                                .createNode(EShallowEntityType.METHOD, 0).skipTo(RETURNS);
156
157                // T-SQL Scalar Function Syntax
158                methodStart.skipTo(BEGIN).parseUntil(IN_METHOD).sequence(END).endNode();
159
160                // T-SQL Inline Table-Valued Function Syntax
161                methodStart.sequence(TABLE).skipTo(RETURN).parseUntil(IN_METHOD)
162                                .sequence(END).endNode();
163        }
164
165        /** {@inheritDoc} */
166        @Override
167        protected boolean isFilteredToken(IToken token, IToken previousToken) {
168                return super.isFilteredToken(token, previousToken)
169                                || token.getType() == EOL || token.getType() == SEMICOLON;
170        }
