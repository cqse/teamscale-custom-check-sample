093/**
094 * Shallow parser for JavaScript. The parser also supports various newer
095 * language features from EcmaScript as well as TypeScript. Additionally,
096 * library specific constructs from multiple libraries (e.g., Google Closure,
097 * Base.js, AngularJS) are handled explicitly.
098 */
099public class JavaScriptShallowParser extends ShallowParserBase<EJavaScriptParserStates> {
100
101        /** The states used in this parser. */
102        public static enum EJavaScriptParserStates {
103
104                /** Top-level parsing state. */
105                TOP_LEVEL,
106
107                /** Generic state, as many constructs can occur at any place. */
108                ANY,
109
110                /** Parsing within a type, such as a TypeScript class. */
111                IN_TYPE,
112
113                /** Parses a lambda (arrow notation) construct. */
114                IN_LAMBDA,
115
116                /**
117                 * Special state that is only valid in the constructor of a google
118                 * closure type
119                 */
120                CLOSURE_CONSTRUCTOR,
121
122                /** Special state used inside of the base.js extend method */
123                BASE_JS_EXTEND;
124        }
125
126        /**
127         * All possible identifiers, including new keywords, which can be used as
128         * identifiers in older JavaScript files.
129         */
130        private static final EnumSet<ETokenType> ALL_IDENTIFIERS = EnumSet.of(IDENTIFIER, CLASS, IMPLEMENTS, DECLARE, ENUM,
131                        EXPORT, EXTENDS, IMPLEMENTS, IMPORT, INSTANCEOF, INTERFACE, MODULE, NAMESPACE, PRIVATE, PROTECTED, PUBLIC,
132                        STATIC, TYPE, SET, GET);
133
134        /** Valid modifiers in TypeScript. */
135        private static final EnumSet<ETokenType> TYPESCRIPT_MODIFIERS = EnumSet.of(STATIC, PUBLIC, PRIVATE, PROTECTED,
136                        READONLY, GET, SET);

154        /** Creates the rules for handling lambda expressions (arrow notation). */
155        private void createLambdaRules() {
156                completeLambda(inState(IN_LAMBDA, ANY).sequence(IDENTIFIER));
157                completeLambda(inState(IN_LAMBDA, ANY).sequence(LPAREN).skipToWithNesting(RPAREN, LPAREN, RPAREN));
158
159                // additional rule for parsing lambda expressions (without braces). see
160                // completeLambda() for details
161                // the node start is moved one token to the right so the shallow
162                // entities produced by this rule don't include the double arrow
163                // (instead it will be included in the parent entity)
164                inState(IN_LAMBDA).sequence(DOUBLE_ARROW)
165                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.LAMBDA_EXPRESSION, null, 1)
166                                .skipBeforeWithNesting(EnumSet.of(RPAREN, SEMICOLON, RBRACE, RBRACK, COMMA), LPAREN, RPAREN).endNode();
167        }

169        /**
170         * Completes a rule for parsing lambda expressions.
171         * 
172         * Lambdas may optionally specify the type of the return value, e.g:
173         * <code>() : any => {}</code>
174         */
175        private static void completeLambda(RecognizerBase<EJavaScriptParserStates> ruleStart) {
176
177                RecognizerBase<EJavaScriptParserStates> lambdaAlternativeWithReturnType = ruleStart.sequence(COLON)
178                                // we don't want to arbitrarily consume tokens until we find a
179                                // double arrow, since we are not yet sure that we are actually
180                                // parsing a lambda. So we skip to either a double arrow (in
181                                // which case this is a lambda) or a token that we know cannot
182                                // occur before the double arrow
183                                .skipBefore(EnumSet.of(DOUBLE_ARROW, LBRACE, COMMA, SEMICOLON)).sequenceBefore(DOUBLE_ARROW)
184                                .createNode(EShallowEntityType.METHOD, "lambda");
185                lambdaAlternativeWithReturnType.sequence(DOUBLE_ARROW, LBRACE).parseUntil(ANY).sequence(RBRACE).endNode();
186
187                // we start parsing before the double arrow (=>) as this allows our
188                // special statement rule to capture this case. This is required, as
189                // this kind of expression is not terminated by a semicolon.
190                lambdaAlternativeWithReturnType.parseOnce(IN_LAMBDA).endNode();
191
192                RecognizerBase<EJavaScriptParserStates> lambdaAlternative = ruleStart.sequenceBefore(DOUBLE_ARROW)
193                                .createNode(EShallowEntityType.METHOD, "lambda");
194                lambdaAlternative.sequence(DOUBLE_ARROW, LBRACE).parseUntil(ANY).sequence(RBRACE).endNode();
195
196                // we start parsing before the double arrow (=>) as this allows our
197                // special statement rule to capture this case. This is required, as
198                // this kind of expression is not terminated by a semicolon.
199                lambdaAlternative.parseOnce(IN_LAMBDA).endNode();
200        }

202        /** Creates parsing rules for meta elements. */
203        private void createMetaRules() {
204                // Closure special commands (require, provide, inherits)
205                inState(TOP_LEVEL).sequence(identifiers("goog"), DOT, identifiers("provide", "require", "inherits"), LPAREN)
206                                .skipTo(RPAREN).createNode(EShallowEntityType.META, new Region(0, 2), new Region(4, -2))
207                                .optional(SEMICOLON).endNode();
208
209                // Support for namespace.js
210                inState(TOP_LEVEL).sequence(NAMESPACE, LPAREN).createNode(EShallowEntityType.META, SubTypeNames.NAMESPACE, 2)
211                                .skipTo(RPAREN).optional(SEMICOLON).endNode();
212        }

219        /**
220         * Creates type rules specific to the Base.JS Framework
221         * http://dean.edwards.name/weblog/2006/03/base/
222         */
223        private void createBaseJSRules() {
224                RecognizerBase<EJavaScriptParserStates> baseJSTypeRecognizer = inState(TOP_LEVEL).repeated(ALL_IDENTIFIERS, DOT)
225                                .markStart().sequence(new UpperCaseStartIdentifierMatcher(), EQ)
226                                .repeated(new AllLowercaseIdentifierMatcher(), DOT)
227                                .sequence(new UpperCaseStartIdentifierMatcher(), DOT, new ExactIdentifierMatcher("extend"), LPAREN,
228                                                LBRACE)
229                                .createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS, 0).parseUntil(BASE_JS_EXTEND).sequence(RBRACE);
230
231                baseJSTypeRecognizer.sequence(RPAREN).optional(SEMICOLON).endNode();
232
233                baseJSTypeRecognizer.sequence(COMMA, LBRACE).parseUntil(BASE_JS_EXTEND).sequence(RBRACE, RPAREN)
234                                .optional(SEMICOLON).endNode();
235        }

237        /** Creates recognizers for the google closure framework */
238        private void createClosureRecognizers() {
239                // Google closure constructor, displayed as method here, as that is what
240                // it technically is
241                inState(TOP_LEVEL).repeated(ALL_IDENTIFIERS, DOT).markStart()
242                                .sequence(new UpperCaseStartIdentifierMatcher(), EQ, FUNCTION)
243                                .createNode(EShallowEntityType.METHOD, SubTypeNames.CONSTRUCTOR, new Region(0, -3)).skipTo(LBRACE)
244                                .parseUntil(CLOSURE_CONSTRUCTOR).sequence(RBRACE).optional(SEMICOLON).endNode();
245
246                RecognizerBase<EJavaScriptParserStates> attributeBeginning = inState(CLOSURE_CONSTRUCTOR)
247                                .sequence(ETokenType.THIS, DOT, ALL_IDENTIFIERS)
248                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.ATTRIBUTE, new Region(0, 2));
249                attributeBeginning.sequence(SEMICOLON).endNode();
250
251                RecognizerBase<EJavaScriptParserStates> attributeAssignment = attributeBeginning.sequence(EQ);
252                attributeAssignment.sequenceBefore(LBRACE).skipToWithNesting(SEMICOLON, LBRACE, RBRACE).endNode();
253
254                attributeAssignment.skipToWithNesting(SEMICOLON, LPAREN, RPAREN).endNode();
255
256        }

258        /** Creates parsing rules specifically for AngularJS. */
259        private void createAngularRules() {
260                // angular modules
261                inState(TOP_LEVEL).optional(VAR, ALL_IDENTIFIERS, EQ).sequence(identifiers("angular"), DOT, MODULE, LPAREN)
262                                .createNode(EShallowEntityType.META, "AngularJS module declaration")
263                                .skipToWithNesting(RPAREN, LPAREN, ETokenType.RPAREN).optional(SEMICOLON).endNode();
264
265                // angular style for unnamed and named functions
266                inState(TOP_LEVEL).optional(ALL_IDENTIFIERS).markStart().sequence(DOT, identifiers("config"), LPAREN, FUNCTION)
267                                .createNode(EShallowEntityType.METHOD, new Object[] { "AngularJS", 1 }).skipTo(LBRACE).parseUntil(ANY)
268                                .sequence(RBRACE, RPAREN).optional(SEMICOLON).endNode();
269                inState(TOP_LEVEL).optional(ALL_IDENTIFIERS).markStart()
270                                .sequence(DOT, identifiers("controller", "filter", "factory", "directive", "constant"), LPAREN,
271                                                STRING_LITERAL, COMMA)
272                                .optional(LBRACK).sequence(FUNCTION)
273                                .createNode(EShallowEntityType.METHOD, new Object[] { "AngularJS", 1 }, 3).skipTo(LBRACE)
274                                .parseUntil(ANY).sequence(RBRACE).optional(RBRACK).sequence(RPAREN).optional(SEMICOLON).endNode();
275        }

277        /**
278         * Creates rules for parsing namespace/module constructs. These are used to
279         * handle those constructs provided by well-known libraries, as JavaScript
280         * itself does not provide a direct namespace concept.
281         */
282        private void createModuleRules() {
283                // YUI library, see http://yuilibrary.com/yui/docs/yui/
284                completeModuleRecognizer(inAnyState().ensureTopLevel().subRecognizer(new YuiCallRecognizer("add"), 1, 1)
285                                .skipTo(STRING_LITERAL).createNode(EShallowEntityType.MODULE, "YUI module", -1));
286                completeModuleRecognizer(inAnyState().ensureTopLevel().subRecognizer(new YuiCallRecognizer("use"), 1, 1)
287                                .createNode(EShallowEntityType.MODULE, "YUI code", -1));
288
289                // TypeScript namespace
290                inState(TOP_LEVEL).repeated(EnumSet.of(DECLARE, EXPORT)).markStart().sequence(EnumSet.of(NAMESPACE, MODULE))
291                                .sequence(IDENTIFIER).repeated(DOT, IDENTIFIER).sequence(LBRACE)
292                                .createNode(EShallowEntityType.MODULE, 0, new Region(1, -2)).parseUntil(TOP_LEVEL).sequence(RBRACE)
293                                .endNode();
294        }

296        /** Completes a recognizer for modules. */
297        private static void completeModuleRecognizer(RecognizerBase<EJavaScriptParserStates> recognizer) {
298                recognizer.skipTo(FUNCTION).skipTo(LBRACE).parseUntil(ANY).sequence(RBRACE)
299                                .skipToWithNesting(RPAREN, LPAREN, RPAREN).optional(SEMICOLON).endNode();
300        }

302        /** Creates rules for TypeScript types. */
303        private void createTypeRules() {
304                RecognizerBase<EJavaScriptParserStates> typeRecognizer = inAnyState().repeated(EnumSet.of(DECLARE, EXPORT))
305                                .markStart();
306                typeRecognizer.sequence(TYPE, IDENTIFIER, EQ).skipTo(SEMICOLON).createNode(EShallowEntityType.TYPE, 0, 1)
307                                .endNode();
308                typeRecognizer.sequence(EnumSet.of(CLASS, INTERFACE), IDENTIFIER).createNode(EShallowEntityType.TYPE, 0, 1)
309                                .skipTo(LBRACE).parseUntil(IN_TYPE).sequence(RBRACE).endNode();
310
311                // enums
312                inAnyState().repeated(EnumSet.of(DECLARE, EXPORT, CONST)).markStart().sequence(ENUM, IDENTIFIER, LBRACE)
313                                .createNode(EShallowEntityType.TYPE, 0, 1).skipToWithNesting(RBRACE, ETokenType.LBRACE, RBRACE)
314                                .endNode();
315        }

317        /** Creates rules for TypeScript type members. */
318        private void createTypeMemberRules() {
319                // constructor
320                inState(IN_TYPE).repeated(TYPESCRIPT_MODIFIERS).markStart().sequence(CONSTRUCTOR)
321                                .createNode(EShallowEntityType.METHOD, SubTypeNames.CONSTRUCTOR).skipNested(ETokenType.LPAREN, RPAREN)
322                                .sequence(LBRACE).parseUntil(ANY).sequence(RBRACE).endNode();
323
324                // "normal" methods
325                RecognizerBase<EJavaScriptParserStates> methodAlternative = inState(IN_TYPE).repeated(TYPESCRIPT_MODIFIERS)
326                                .markStart().repeatedAtLeastOnce(IDENTIFIER).sequence(LPAREN).skipTo(RPAREN)
327                                .skipBefore(EnumSet.of(LBRACE, SEMICOLON));
328                methodAlternative.sequence(SEMICOLON).createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD_DECLARATION, 0)
329                                .endNode();
330                methodAlternative.sequence(LBRACE).createNode(EShallowEntityType.METHOD, "method", 0).parseUntil(ANY)
331                                .sequence(RBRACE).endNode();
332
333                // attributes
334                // the semicolon at the end of attribute declarations is optional, so we
335                // have to use a simple statement recognizer to find the correct end of
336                // the current statement
337                inState(IN_TYPE).repeated(TYPESCRIPT_MODIFIERS).markStart()
338                                .sequenceBefore(IDENTIFIER, EnumSet.of(EQ, COLON, QUESTION))
339                                .subRecognizer(new JavaScriptSimpleStatementRecognizer(EShallowEntityType.ATTRIBUTE,
340                                                SubTypeNames.ATTRIBUTE, true), 1, 1)
341                                .endNode();
342        }

394        /** Creates parsing rules for statements. */
395        private void createStatementRules() {
396
397                // empty statement
398                inAnyState().sequence(SEMICOLON).createNode(EShallowEntityType.STATEMENT, SubTypeNames.EMPTY_STATEMENT)
399                                .endNode();
400
401                // Attributes
402                RecognizerBase<EJavaScriptParserStates> baseJsAttribute = inState(BASE_JS_EXTEND)
403                                .sequence(ALL_IDENTIFIERS, COLON).createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.ATTRIBUTE, 0);
404                baseJsAttribute.skipToWithNesting(COMMA, LBRACE, RBRACE).endNode();
405                // Might be the last one in the constructor
406                baseJsAttribute.skipBeforeWithNesting(RBRACE, LBRACE, RBRACE).endNode();
407
408                // filter out labels as meta as they do not increase statement count
409                inAnyState().sequence(ALL_IDENTIFIERS, COLON).createNode(EShallowEntityType.META, SubTypeNames.LABEL, 0)
410                                .endNode();
411
412                createElseIfRule();
413                createSimpleBlockRules();
414                createSwitchCaseRules();
415                createDoWhileRules();
416
417                // anonymous block
418                inAnyState().sequence(LBRACE).createNode(EShallowEntityType.STATEMENT, "anonymous block").parseUntil(ANY)
419                                .sequence(RBRACE).endNode();
420
421                // variables
422                inAnyState().sequence(EXPORT, EnumSet.of(VAR, CONST, LET))
423                                .subRecognizer(new JavaScriptSimpleStatementRecognizer(EShallowEntityType.ATTRIBUTE,
424                                                SubTypeNames.EXPORTED_VARIABLE, false), 1, 1)
425                                .endNode();
426
427                // simple statement
428                inAnyState().subRecognizer(new JavaScriptSimpleStatementRecognizer(EShallowEntityType.STATEMENT,
429                                SubTypeNames.SIMPLE_STATEMENT, false), 1, 1).endNode();
430        }

432        /** Creates the rule for "else if" construct. */
433        private void createElseIfRule() {
434                RecognizerBase<EJavaScriptParserStates> elseIfAlternative = inAnyState().sequence(ELSE, IF)
435                                .skipNested(LPAREN, RPAREN, new JavaScriptFunctionRecognizer())
436                                .createNode(EShallowEntityType.STATEMENT, new int[] { 0, 1 });
437                endWithPossibleContinuation(elseIfAlternative.sequence(LBRACE).parseUntil(ANY).sequence(RBRACE),
438                                EnumSet.of(ELSE));
439                endWithPossibleContinuation(elseIfAlternative.parseOnce(ANY), EnumSet.of(ELSE));
440        }

442        /** Creates simple block rules. */
443        private void createSimpleBlockRules() {
444                createBlockRuleWithContinuation(EnumSet.of(WHILE, FOR, SWITCH, WITH), null, true);
445                createBlockRuleWithContinuation(EnumSet.of(ELSE, FINALLY), null, false);
446                createBlockRuleWithContinuation(EnumSet.of(IF), EnumSet.of(ELSE), true);
447                createBlockRuleWithContinuation(EnumSet.of(TRY, CATCH), EnumSet.of(CATCH, FINALLY), true);
448        }

450        /** Creates rules for do/while. */
451        private void createDoWhileRules() {
452                RecognizerBase<EJavaScriptParserStates> doWhileAlternative = inState(ANY).sequence(DO)
453                                .createNode(EShallowEntityType.STATEMENT, 0);
454                doWhileAlternative.sequence(LBRACE).parseUntil(ANY).sequence(RBRACE, WHILE)
455                                .skipNested(LPAREN, RPAREN, new JavaScriptFunctionRecognizer()).endNode();
456                doWhileAlternative.parseOnce(ANY).sequence(WHILE).skipNested(LPAREN, RPAREN, new JavaScriptFunctionRecognizer())
457                                .endNode();
458        }

460        /** Creates rules for switch/case. */
461        private void createSwitchCaseRules() {
462                HashSet<ETokenType> literalsAndIdentifiers = new HashSet<ETokenType>(LITERALS);
463                literalsAndIdentifiers.addAll(IDENTIFIERS);
464                inAnyState().sequence(CASE, literalsAndIdentifiers, COLON)
465                                .createNode(EShallowEntityType.META, new int[] { 0, 1 }).endNode();
466                inAnyState().sequence(CASE).skipToWithNesting(COLON, LPAREN, RPAREN).createNode(EShallowEntityType.META, 0)
467                                .endNode();
468
469                inAnyState().sequence(DEFAULT, COLON).createNode(EShallowEntityType.META, 0).endNode();
470        }

472        /**
473         * Creates a rule for recognizing a statement starting with a single
474         * keyword, optionally followed by an expression in parentheses, and
475         * followed by a block or a single statement.
476         *
477         * @param continuationTokens
478         *            list of tokens that indicate a continued statement if
479         *            encountered after the block. May be null.
480         */
481        private void createBlockRuleWithContinuation(EnumSet<ETokenType> startTokens,
482                        EnumSet<ETokenType> continuationTokens, boolean canBeFollowedByParentheses) {
483                RecognizerBase<EJavaScriptParserStates> alternative = inAnyState().sequence(startTokens)
484                                .createNode(EShallowEntityType.STATEMENT, 0);
485                if (canBeFollowedByParentheses) {
486                        alternative = alternative.skipNested(LPAREN, RPAREN, new JavaScriptFunctionRecognizer());
487                }
488
489                endWithPossibleContinuation(alternative.sequence(LBRACE).parseUntil(ANY).sequence(RBRACE), continuationTokens);
490                endWithPossibleContinuation(alternative.parseOnce(ANY), continuationTokens);
491        }
