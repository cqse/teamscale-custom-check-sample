064/**
065 * Language feature parser for Cs.
066 * 
067 * @author $Author: heinemann $
068 * @version $Rev: 55643 $
069 * @ConQAT.Rating GREEN Hash: 2D36CDBA2CD5AEA6F9EE04BBF7523677

071public class CsLanguageFeatureParser extends CLikeLanguageFeatureParserBase {
072
073        /** All token types that can be used to specify a type. */
074        public static final EnumSet<ETokenType> TYPE_TOKENS = EnumSet.of(IDENTIFIER,
075                        VAR, BOOL, CHAR, BYTE, SBYTE, SHORT, INT, LONG, USHORT, UINT, ULONG,
076                        FLOAT, DOUBLE, DECIMAL, STRING, OBJECT);
077
078        /** All token types that are modifiers for variables or attributes. */
079        public static final EnumSet<ETokenType> VARIABLE_DECLARATION_MODIFIERS = EnumSet
080                        .of(PUBLIC, PRIVATE, INTERNAL, PROTECTED, CONST, READONLY, STATIC,
081                                        VOLATILE, NEW);
082
083        /** Suffix for CS-EventHandlers. */
084        public static final String EVENT_ARGS_SUFFIX = "EventArgs";
085
086        /** Constructor. */
087        public CsLanguageFeatureParser() {
088                super(CsShallowParser.VALID_IDENTIFIERS, TYPE_TOKENS);
089        }
090
091        /**
092         * Returns the imported namespace name from a given "using" entity. If the
093         * "using" clause is an aliasing clause, null is returned.
094         */
095        @Override
096        public String getImportName(ShallowEntity entity) {
097                CCSMAssert.isTrue(isImport(entity),
098                                "entity.getType() must be equal to EShallowEntityType.META and "
099                                                + "entity.getSubtype() must be equal to SubTypeNames.USING");
100
101                List<IToken> tokens = entity.ownStartTokens();
102
103                if (TokenStreamUtils.containsAll(tokens, EQ)) {
104                        // if this using directive is aliasing a namespace (thus it contains
105                        // a "="), we don't extract the namespace
106                        return null;
107                }
108
109                int lastIdentifierIndex = TokenStreamUtils.findLast(tokens,
110                                CsShallowParser.VALID_IDENTIFIERS);
111                if (lastIdentifierIndex == TokenStreamUtils.NOT_FOUND) {
112                        return null;
113                }
114
115                return TokenStreamTextUtils
116                                .concatTokenTexts(tokens.subList(1, lastIdentifierIndex + 1));
117        }
118
119        /** Returns whether the given entity is a using directive. */
120        @Override
121        public boolean isImport(ShallowEntity entity) {
122                return entity.getType().equals(EShallowEntityType.META)
123                                && entity.getSubtype().equals(SubTypeNames.USING);
124        }
125
126        /** Checks whether the given method is a CS-EventHandler. */
127        public boolean isEventHandler(ShallowEntity method) {
128                CCSMAssert.isTrue(method.getType() == EShallowEntityType.METHOD,
129                                "method.getType() must be \"METHOD\"");
130                if (!hasVoidReturnType(method)) {
131                        return false;
132                }
133
134                List<List<IToken>> parameterTokens = getParameterTokens(method);
135                if (parameterTokens.size() != 2) {
136                        return false;
137                }
138
139                List<IToken> firstParameter = parameterTokens.get(0);
140                List<IToken> secondParameter = parameterTokens.get(1);
141
142                if (!TokenStreamUtils.hasTypes(firstParameter, ETokenType.OBJECT,
143                                ETokenType.IDENTIFIER)) {
144                        return false;
145                }
146                if (TokenStreamUtils.hasTypes(secondParameter, ETokenType.IDENTIFIER,
147                                ETokenType.IDENTIFIER)) {
148                        return secondParameter.get(0).getText().endsWith(EVENT_ARGS_SUFFIX);
149                }
150                return false;
151        }
152
153        /** Returns whether the given entity is partial. */
154        public boolean isPartial(ShallowEntity entity) {
155                return TokenStreamUtils.find(entity.ownStartTokens(),
156                                ETokenType.PARTIAL) != TokenStreamUtils.NOT_FOUND;
157        }

159        // Currently the following methods are only supported for CS, but the future
160        // they should be moved to CLikeLanguageUtils and adapted in a way that they
161        // support Java too.
162
163        /** Returns a list of all class names the given entity inherits from. */
164        public List<String> getParentNames(ShallowEntity type) {
165                CCSMAssert.isTrue(type.getType() == EShallowEntityType.TYPE,
166                                "type.getType() must be \"TYPE\"");
167                List<IToken> inheritanceTokens = TokenStreamUtils.tokensBetween(
168                                type.ownStartTokens(), ETokenType.COLON, ETokenType.LBRACE);
169                if (inheritanceTokens.isEmpty()) {
170                        return CollectionUtils.emptyList();
171                }
172
173                List<List<IToken>> splitInheritanceTokens = TokenStreamUtils
174                                .splitWithNesting(inheritanceTokens, ETokenType.COMMA,
175                                                ETokenType.LT, ETokenType.GT);
176                return TokenStreamTextUtils.concatAllTokenTexts(splitInheritanceTokens);
177        }
178
179        /**
180         * Returns all variable names from a variable declaration within a for
181         * statement.
182         */
183        public List<IToken> getVariableNamesFromForLikeTokens(
184                        List<IToken> forLikeTokens, ETokenType endToken) {
185                List<IToken> variableDeclarationTokens = getVariableTokensFromForLikeTokens(
186                                forLikeTokens, endToken);
187                return getVariableDeclarationNamesFromTokens(variableDeclarationTokens);
188        }
189
190        /**
191         * Returns all variable names from the given variable declaration tokens.
192         * Names are only returned if they are newly introduced, that means, that e.
193         * g. the variable "t" from "using(t) {" will not be returned. This method
194         * does not support generic types at the moment.
195         */
196        public List<IToken> getVariableDeclarationNamesFromTokens(
197                        List<IToken> variableDeclarationTokens) {
198                List<IToken> variableNames = new ArrayList<>();
199
200                int offset = 0;
201                while (variableDeclarationTokens.size() > 0
202                                && VARIABLE_DECLARATION_MODIFIERS.contains(
203                                                variableDeclarationTokens.get(offset).getType())) {
204                        offset += 1;
205                }
206
207                if (variableDeclarationTokens.size() >= 2 + offset) {
208                        IToken token1 = variableDeclarationTokens.get(offset);
209                        IToken token2 = variableDeclarationTokens.get(1 + offset);
210                        if (isVariableDeclaration(token1, token2)) {
211                                variableNames.add(token2);
212                                addAdditionalVariableNameTokens(variableDeclarationTokens,
213                                                offset, variableNames);
214                        }
215
216                }
217
218                return variableNames;
219        }
220
221        /** Adds additional variable name tokens. */
222        private static void addAdditionalVariableNameTokens(
223                        List<IToken> variableDeclarationTokens, int offset,
224                        List<IToken> variableNames) {
225                int parenthesisNesting = 0;
226                for (int i = 2 + offset; i < variableDeclarationTokens.size(); ++i) {
227                        IToken token = variableDeclarationTokens.get(i);
228                        ETokenType tokenType = token.getType();
229                        if (tokenType == ETokenType.LPAREN) {
230                                parenthesisNesting += 1;
231                        } else if (tokenType == ETokenType.RPAREN) {
232                                parenthesisNesting -= 1;
233                        } else if (tokenType == ETokenType.IDENTIFIER
234                                        && parenthesisNesting == 0 && variableDeclarationTokens
235                                                        .get(i - 1).getType() == ETokenType.COMMA) {
236                                variableNames.add(token);
237                        }
238                }
239        }
240
241        /** Returns whether the two given tokens indicate a variable declaration. */
242        private boolean isVariableDeclaration(IToken token1, IToken token2) {
243                return (typeTokens.contains(token1.getType())
244                                && validIdentifiers.contains(token2.getType()));
245        }
246
247        /** {@inheritDoc} */
248        @Override
249        public ELanguage getLanguage() {
250                return ELanguage.CS;
251        }
