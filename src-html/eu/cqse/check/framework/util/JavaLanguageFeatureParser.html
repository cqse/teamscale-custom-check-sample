026import static eu.cqse.check.framework.scanner.ETokenType.INT;
027import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
028import static eu.cqse.check.framework.scanner.ETokenType.LONG;
029import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
030import static eu.cqse.check.framework.scanner.ETokenType.SHORT;

047import eu.cqse.check.framework.shallowparser.TokenStreamUtils;

049import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;

051/**
052 * Language feature parser for Java.
053 * 
054 * @author $Author: heinemann $
055 * @version $Rev: 55643 $
056 * @ConQAT.Rating GREEN Hash: B8C39CD0EBEC27232D278751191BE9F3
057 */
058public class JavaLanguageFeatureParser extends CLikeLanguageFeatureParserBase {
059
060        /** All token types that can be used to specify a type. */
061        public static final EnumSet<ETokenType> TYPE_TOKENS = EnumSet.of(IDENTIFIER,
062                        BOOLEAN, CHAR, BYTE, SHORT, INT, LONG, FLOAT, DOUBLE);
063
064        /** All token types that can be used as identifier. */
065        public static final EnumSet<ETokenType> VALID_IDENTIFIERS = EnumSet
066                        .of(IDENTIFIER);
067
068        /** Constructor. */
069        public JavaLanguageFeatureParser() {
070                super(VALID_IDENTIFIERS, TYPE_TOKENS);
071        }
072
073        /**
074         * Returns the imported package name from a given "import" entity. Static
075         * imports are ignored.
076         */
077        @Override
078        public String getImportName(ShallowEntity entity) {
079                CCSMAssert.isTrue(isImport(entity),
080                                "entity.getType() must be equal to EShallowEntityType.META and "
081                                                + "entity.getSubtype() must be equal to SubTypeNames.IMPORT");
082
083                List<IToken> tokens = entity.ownStartTokens();
084                if (TokenStreamUtils.find(tokens,
085                                STATIC) != TokenStreamUtils.NOT_FOUND) {
086                        return null;
087                }
088
089                String importName = entity.getName();
090                return StringUtils.removeLastPart(importName, '.');
091        }
092
093        /** Returns whether the given entity is an import. */
094        @Override
095        public boolean isImport(ShallowEntity entity) {
096                return entity.getType().equals(EShallowEntityType.META)
097                                && entity.getSubtype().equals(SubTypeNames.IMPORT);
098        }
099
100        /** Returns whether the given entity is a for each loop. */
101        public boolean isForEachLoop(ShallowEntity entity) {
102                if (entity.getType().equals(EShallowEntityType.STATEMENT)
103                                && entity.getSubtype().equals(SubTypeNames.FOR)) {
104                        return TokenStreamUtils.findFirstTopLevel(entity.ownStartTokens(),
105                                        COLON, Arrays.asList(LBRACE),
106                                        Arrays.asList(RBRACE)) != TokenStreamUtils.NOT_FOUND;
107                }
108                return false;
109        }
110
111        /** Returns whether the given entity is an annotation. */
112        public boolean isAnnotation(ShallowEntity entity) {
113                return entity.getType() == EShallowEntityType.META
114                                && SubTypeNames.ANNOTATION.equals(entity.getSubtype());
115        }
116
117        /** Returns all annotation entities that annotate the given entity. */
118        public List<ShallowEntity> getAnnotations(ShallowEntity entity) {
119                ShallowEntity parent = entity.getParent();
120                if (parent == null) {
121                        return CollectionUtils.emptyList();
122                }
123
124                List<ShallowEntity> annotations = new ArrayList<>();
125
126                UnmodifiableList<ShallowEntity> entities = parent.getChildren();
127                for (int i = entities.indexOf(entity) - 1; i >= 0; i--) {
128                        ShallowEntity previousEntity = entities.get(i);
129                        if (!isAnnotation(previousEntity)) {
130                                break;
131                        }
132                        annotations.add(previousEntity);
133                }
134                return annotations;
135        }
136
137        /** {@inheritDoc} */
138        @Override
139        public ELanguage getLanguage() {
140                return ELanguage.JAVA;
141        }
142}
