017package eu.cqse.check.framework.util;
018
019import static eu.cqse.check.framework.scanner.ETokenType.AND;
020import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
021import static eu.cqse.check.framework.scanner.ETokenType.MULT;
022import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
023
024import java.util.ArrayList;
025import java.util.EnumSet;
026import java.util.List;
027
028import org.conqat.lib.commons.string.StringUtils;
029
030import eu.cqse.check.framework.scanner.ETokenType;
031import eu.cqse.check.framework.scanner.IToken;
032
033/**
034 * A collection that manages tokens of a type name. It distinguishes between
035 * primitive and non-primitive types and formats the type name accordingly.
036 */
037public class TypeNameTokenCollection {
038
039        /**
040         * Token types that need no separator when they are formatted as typename.
041         */
042        public static final EnumSet<ETokenType> NON_SEPARATED_TOKENS = EnumSet.of(MULT, AND, LBRACK, RBRACK);
043
044        /** The tokens that belong to the type name. */
045        private final List<IToken> typeNameTokens = new ArrayList<>();
046
047        /** The string that separates tokens in the resulting type name. */
048        private String separator = StringUtils.EMPTY_STRING;
049
050        /** Marks the type as a primitive one. */
051        public void markPrimitive() {
052                this.separator = StringUtils.SPACE;
053        }
054
055        /**
056         * Adds a token that belongs to the type name. Tokens are added in reverse
057         * order.
058         */
059        public void addToken(IToken typeNameToken) {
060                typeNameTokens.add(typeNameToken);
061        }
062
063        /** Formats the type name. */
064        public String formatTypeName() {
065                StringBuilder builder = new StringBuilder();
066                for (int i = typeNameTokens.size() - 1; i >= 0; i--) {
067                        IToken token = typeNameTokens.get(i);
068                        if (i < typeNameTokens.size() - 1 && isSeparatedToken(token)) {
069                                builder.append(separator);
070                        }
071                        builder.append(token.getText());
072                }
073                return builder.toString();
074        }
075
076        /**
077         * Returns whether a separator should be prepended to the given token in the
078         * formatted string.
079         */
080        private static boolean isSeparatedToken(IToken token) {
081                return !NON_SEPARATED_TOKENS.contains(token.getType());
082        }
083}
