017package eu.cqse.check.framework.scanner;
018
019/**
020 * Base class for implementations of the token interface. This class is
021 * immutable.
022 * 
023 * @author $Author: hummelb $
024 * @version $Revision: 52865 $
025 * @ConQAT.Rating GREEN Hash: B13955FB29C3A07F1C7F6D28BB7F960B
026 */
027public abstract class Token implements IToken {
028
029        /** Version for serialization. */
030        private static final long serialVersionUID = 1L;
031
032        /** Token type. */
033        protected final ETokenType type;
034
035        /** Number of characters before token in the resource it originates from */
036        protected final int offset;
037
038        /** Line number token was found at. */
039        protected final int lineNumber;
040
041        /** Verbatim copy of the text found. */
042        protected final String text;
043
044        /** Identifier of the resource this token originates from */
045        protected final String originId;
046
047        /** Constructor. */
048        protected Token(ETokenType type, int offset, int lineNumber, String text,
049                        String originId) {
050                this.type = type;
051                this.offset = offset;
052                this.lineNumber = lineNumber;
053                this.text = text;
054                this.originId = originId;
055        }
056
057        /**
058         * Obtain string representation of this token. This is meant for debugging
059         * purposes.
060         */
061        @Override
062        public String toString() {
063                String result = "TOKEN (" + type + ") Text: >>" + text + "<<";
064
065                result += " Origin: '" + originId + "' line#: " + lineNumber;
066                return result;
067        }
068
069        /** {@inheritDoc} */
070        @Override
071        public String getText() {
072                return text;
073        }
074
075        /**
076         * {@inheritDoc}
077         */
078        @Override
079        public int getOffset() {
080                return offset;
081        }
082
083        /** {@inheritDoc} */
084        @Override
085        public int getEndOffset() {
086                return offset + text.length() - 1;
087        }
088
089        /** {@inheritDoc} */
090        @Override
091        public int getLineNumber() {
092                return lineNumber;
093        }
094
095        /** {@inheritDoc} */
096        @Override
097        public String getOriginId() {
098                return originId;
099        }
100
101        /** {@inheritDoc} */
102        @Override
103        public ETokenType getType() {
104                return type;
105        }
106
107}
