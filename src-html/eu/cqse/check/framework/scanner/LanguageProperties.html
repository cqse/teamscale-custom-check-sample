083/**

085 * by the scanner framework. These are intentionally stored separately from
086 * {@link eu.cqse.check.framework.scanner.ELanguage} as ELanguage does not have
087 * access to the language's token classes.
088 */
089public class LanguageProperties {

142        private static void register(ELanguage language, ITokenFactory tokenFactory, Set<ETokenType> statementDelimiters,
143                        String commentLineTrimRegex, boolean cLike) {
144                LANGUAGE_PROPERTIES_MAPPING.put(language, new LanguageProperties(tokenFactory,
145                                new StatementOracle(statementDelimiters), commentLineTrimRegex, cLike));
146        }
147
148        private static void register(ELanguage language, ITokenFactory tokenFactory, IStatementOracle oracle,
149                        String commentLineTrimRegex, boolean cLike) {
150                LANGUAGE_PROPERTIES_MAPPING.put(language,
151                                new LanguageProperties(tokenFactory, oracle, commentLineTrimRegex, cLike));
152        }
153
154        /**
155         * @return The {@link LanguageProperties} associated with the given language.
156         */
157        public static LanguageProperties of(ELanguage language) {
158                return LANGUAGE_PROPERTIES_MAPPING.get(language);
159        }
160
161        /** The statement oracle for this language. */
162        private final IStatementOracle statementOracle;
163
164        /**
165         * Pattern describing the parts of a comment line that should be trimmed to
166         * reveal the text.
167         */
168        private final Pattern commentLineTrimPattern;
169
170        /** Whether the language is C-like, i.e. based on curly braces, etc. */
171        private final boolean cLike;
172
173        private final ITokenFactory tokenFactory;
174
175        /** Create language properties. */
176        private LanguageProperties(ITokenFactory tokenFactory, IStatementOracle oracle, String commentLineTrimRegex,
177                        boolean cLike) {
178                this.statementOracle = oracle;
179                this.commentLineTrimPattern = Pattern.compile(commentLineTrimRegex);
180                this.cLike = cLike;
181                this.tokenFactory = tokenFactory;
182        }
183
184        /** Get statement oracle for this language. */
185        public IStatementOracle getStatementOracle() {
186                return statementOracle;
187        }
188
189        /**
190         * Get the content of a comment, i.e. with the comment delimiters removed.
191         */
192        public String getCommentContent(String commentText) {
193                StringBuilder content = new StringBuilder();
194                for (String line : StringUtils.splitLinesAsList(commentText)) {
195                        if (content.length() > 0) {
196                                content.append(StringUtils.LINE_SEPARATOR);
197                        }
198                        content.append(commentLineTrimPattern.matcher(line).replaceAll(StringUtils.EMPTY_STRING).trim());
199                }
200                return content.toString();
201        }
202
203        /** @see #cLike */
204        public boolean isCLike() {
205                return cLike;
206        }
207
208        /** Creates a new token for the language. */
209        public IToken createToken(ETokenType type, int offset, int lineNumber, String text, String originId) {
210                return tokenFactory.create(type, offset, lineNumber, text, originId);
211        }
212
213        /** Factory interface for token creation. */
214        @FunctionalInterface
215        private interface ITokenFactory {
216
217                /** Creates a token. */
218                IToken create(ETokenType type, int offset, int lineNumber, String text, String originId);
219        }
