017package eu.cqse.check.framework.scanner;
018
019import java.io.IOException;
020import java.util.ArrayList;
021import java.util.List;
022
023/**
024 * This class offers utility methods for scanners.
025 * 
026 * @author Florian Deissenboeck
027 * @author Elmar Juergens
028 * @author $Author: hummelb $
029 * @version $Rev: 52656 $
030 * @ConQAT.Rating GREEN Hash: 7653C46FCA092A85783E4D6D1B975488
031 */
032public class ScannerUtils {
033
034        /**
035         * Read tokens from a strict scanner and store them in a list. This expects
036         * the scanner to return a token with type {@link ETokenType#EOF} at the end
037         * of the file. This last token will not be added to the list of tokens.
038         * 
039         * @param scanner
040         *            the scanner to read from
041         * @param tokens
042         *            the list to store tokens
043         * @param exceptions
044         *            another list to store possible exceptions that occurred during
045         *            scanning.
046         * @throws IOException
047         *             thrown if scanner throws an IO exception
048         */
049        public static void readTokens(IScanner scanner, List<IToken> tokens,
050                        List<ScannerException> exceptions) throws IOException {
051
052                IToken token = null;
053
054                do {
055                        try {
056                                token = scanner.getNextToken();
057                                if (token.getType() != ETokenType.EOF) {
058                                        tokens.add(token);
059                                }
060                        } catch (ScannerException e) {
061                                exceptions.add(e);
062                                continue;
063                        }
064                } while (token != null && token.getType() != ETokenType.EOF);
065
066        }
067
068        /**
069         * Read tokens from a lenient scanner and store them in a list. This expects
070         * the scanner to return a token with type {@link ETokenType#EOF} at the end
071         * of the file. This last token will not be added to the list of tokens.
072         * 
073         * @param scanner
074         *            the scanner to read from
075         * 
076         * @throws IOException
077         *             thrown if scanner throws an IO exception
078         */
079        public static List<IToken> readTokens(ILenientScanner scanner)
080                        throws IOException {
081
082                List<IToken> tokens = new ArrayList<IToken>();
083
084                IToken token = null;
085                while ((token = scanner.getNextToken()).getType() != ETokenType.EOF) {
086                        tokens.add(token);
087                }
088
089                return tokens;
090
091        }
092
093        /**
094         * Returns the results of scanning a string. The origin of the tokens is set
095         * to some arbitrary value.
096         */
097        public static List<IToken> getTokens(String content, ELanguage language) {
098                return getTokens(content, language, "origin");
099        }
100
101        /**
102         * Returns the results of scanning a string. The origin of the tokens is set
103         * to some arbitrary value.
104         */
105        public static List<IToken> getTokens(String content, ELanguage language,
106                        String origin) {
107                ILenientScanner scanner = ScannerFactory.newLenientScanner(language,
108                                content, origin);
109                try {
110                        return readTokens(scanner);
111                } catch (IOException e) {
112                        throw new AssertionError(
113                                        "can not happen as the scanner is lenient and we are working from memory");
114                }
115        }
116}
