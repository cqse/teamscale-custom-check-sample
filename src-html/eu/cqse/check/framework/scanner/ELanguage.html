203        /** List of languages that do not have methods. */
204        private static final EnumSet<ELanguage> LANGUAGES_WITHOUT_METHODS = EnumSet.of(ABAP_DDIC, LINE, NL_REQUIREMENTS,
205                        NL_TESTS, TEXT, XML, HANA_VIEW);
206
207        /** This maps from extensions to languages. */
208        private static final ListMap<String, ELanguage> EXTENSION_2_LANGUAGE_MAP = new ListMap<>();
209
210        /** Initialize {@link #EXTENSION_2_LANGUAGE_MAP}. */
211        static {
212                for (ELanguage language : values()) {
213                        for (String extension : language.extensions) {
214                                EXTENSION_2_LANGUAGE_MAP.add(extension.toLowerCase(), language);
215                        }
216                }
217        }
218
219        /** The readable name of the language, to be used, e.g., in a UI. */
220        private final String readableName;
221
222        /** Whether the language is case sensitive. */
223        private final boolean caseSensitive;
224
225        /** File extensions commonly used for this language. */
226        private final String[] extensions;
227
228        /** Create language. */
229        ELanguage(String readableName, boolean caseSensitive, String... extensions) {
230                this.readableName = readableName;
231                this.caseSensitive = caseSensitive;
232                this.extensions = extensions;
233        }
234
235        /** Returns {@link #readableName}. */
236        public String getReadableName() {
237                return readableName;
238        }
239
240        /** Return whether the language is case sensitive. */
241        public boolean isCaseSensitive() {
242                return caseSensitive;
243        }
244
245        /** Get the file extensions commonly used for this language. */
246        public String[] getFileExtensions() {
247                return CollectionUtils.copyArray(extensions);
248        }
249
250        /**
251         * Gets the {@link ELanguage} value corresponding to the file extension of the
252         * path. Returns null if no extension was found. If there are multiple possible
253         * languages, the first one is returned.
254         */
255        public static ELanguage fromPath(String path) {
256                return fromFile(new File(path));
257        }
258
259        /**
260         * Gets the {@link ELanguage} value corresponding to the file extension of the
261         * file. Returns null if no extension was found. If there are multiple possible
262         * languages, the first one is returned.
263         */
264        public static ELanguage fromFile(File file) {
265                return fromFileExtension(FileSystemUtils.getFileExtension(file));
266        }
267
268        /**
269         * Gets the {@link ELanguage} value corresponding to the given file extension
270         * (without a dot). Returns {@link ELanguage#LINE} as a fallback if no language
271         * was registered for the extension. If there are multiple possible languages
272         * registered for the extension, however, the first one is returned.
273         */
274        public static ELanguage fromFileExtension(String extension) {
275                if (extension == null) {
276                        return ELanguage.LINE;
277                }
278
279                List<ELanguage> result = EXTENSION_2_LANGUAGE_MAP.getCollection(extension.toLowerCase());
280
281                if (result == null || result.isEmpty()) {
282                        return ELanguage.LINE;
283                }
284                return result.get(0);
285        }
286
287        /** Returns all languages for the given file extension */
288        public static Set<ELanguage> getAllLanguagesForExtension(String extension) {
289                return new HashSet<>(EXTENSION_2_LANGUAGE_MAP.getCollectionOrEmpty(extension));
290        }
291
292        /**
293         * Returns all {@link ELanguage}s matching the file extension of the given file
294         * path.
295         */
296        public static Set<ELanguage> getAllLanguagesForPath(String path) {
297                return getAllLanguagesForExtension(FileSystemUtils.getFileExtension(path));
298        }
299
300        /** @return Whether the given language has the concept of methods. */
301        public static boolean languageHasMethods(ELanguage language) {
302                return !LANGUAGES_WITHOUT_METHODS.contains(language);
303        }
