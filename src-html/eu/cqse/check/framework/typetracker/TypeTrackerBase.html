017package eu.cqse.check.framework.typetracker;
018
019import java.util.EnumSet;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023import java.util.Optional;
024import java.util.Set;
025
026import eu.cqse.check.framework.scanner.ELanguage;
027import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
028import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
029
030/**
031 * Abstract base class for type trackers.
032 * <p>
033 * While this base class itself does not keep any state and is thread-safe,
034 * implementing classes may introduce such behavior.
035 *
036 * @see ITypeTracker
037 */
038public abstract class TypeTrackerBase implements ITypeTracker {
039
040        /**
041         * A set of all entity types that provide an ordered introduction of type
042         * information, i.e. entities inside which variables are only visible after they
043         * have been declared. This is typically the case inside methods and also inside
044         * control-flow blocks.
045         *
046         * @see #needsOrderedScope(ShallowEntity)
047         */
048        private static final EnumSet<EShallowEntityType> ORDERED_ENTITY_TYPES = EnumSet.of(EShallowEntityType.METHOD,
049                        EShallowEntityType.STATEMENT);
050
051        /** Language of the type tracker. */
052        private final ELanguage language;
053
054        /** Constructor. */
055        public TypeTrackerBase(ELanguage language) {
056                this.language = language;
057        }
058
059        /** {@inheritDoc} */
060        @Override
061        public ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities) {
062                ScopedTypeLookup variableLookup = new ScopedTypeLookup(language.isCaseSensitive());
063                return createTypeResolution(rootEntities, variableLookup);
064        }
065
066        /** {@inheritDoc} */
067        @Override
068        public ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities, ScopedTypeLookup variableLookup) {
069                initialize(rootEntities);
070
071                ITypeInfoExtractor typeInfoExtractor = TypeInfoExtractorFactory.getTypeInfoExtractor(language, rootEntities);
072
073                addDefaultImports(variableLookup);
074
075                return new PrecomputedTypeResolution(
076                                createTypeInformation(rootEntities, variableLookup, false, typeInfoExtractor));
077        }
078
079        /**
080         * Recursively creates type information for the given root entities.
081         *
082         * @param rootEntities
083         *            The root entities to create type information for.
084         * @param variableLookup
085         *            Contains type information for all known variables.
086         * @param ordered
087         *            Specifies whether an ordered scope should be used.
088         * @param typeInfoExtractor
089         *            The extractor to create type information.
090         */
091        private Map<ShallowEntity, ScopedTypeLookup> createTypeInformation(List<ShallowEntity> rootEntities,
092                        ScopedTypeLookup variableLookup, boolean ordered, ITypeInfoExtractor typeInfoExtractor) {
093                Map<ShallowEntity, ScopedTypeLookup> typeLookups = new HashMap<>();
094                createTypeInformation(rootEntities, variableLookup, ordered, typeLookups, typeInfoExtractor);
095                return typeLookups;
096        }
097
098        /**
099         * Recursive method that is called to traverse all given entities. Scopes are
100         * created as needed. Child entities always have a new child scope. Sibling
101         * entities by default share the same scope (e.g. members of a class) unless
102         * {@link #needsSeparateScope(ShallowEntity)} is {@code true} or the siblings
103         * are ordered. In these cases, a new scope is created whenever a new variable
104         * is declared.
105         *
106         * @param entities
107         *            The entities to extract information from. May be a file's root
108         *            entities or the children of any entity down the AST.
109         * @param variableLookup
110         *            Contains type information for all known variables.
111         * @param ordered
112         *            Specifies whether an ordered scope should be used.
113         * @param typeLookups
114         *            Mapping from shallow entities to type lookup maps.
115         * @param typeInfoExtractor
116         *            The extractor to create type information.
117         */
118        private void createTypeInformation(List<ShallowEntity> entities, ScopedTypeLookup variableLookup, boolean ordered,
119                        Map<ShallowEntity, ScopedTypeLookup> typeLookups, ITypeInfoExtractor typeInfoExtractor) {
120                for (ShallowEntity entity : entities) {
121                        processImport(entity, variableLookup);
122
123                        List<TypedVariable> typeInfo = typeInfoExtractor.extract(entity);
124
125                        boolean newTypeInfo = !typeInfo.isEmpty();
126
127                        if (ordered && newTypeInfo) {
128                                variableLookup = variableLookup.createChildScope();
129                        }
130
131                        ScopedTypeLookup currentLookup = variableLookup;
132                        if (newTypeInfo) {
133                                if (needsSeparateScope(entity)) {
134                                        currentLookup = currentLookup.createChildScope();
135                                }
136                                for (TypedVariable info : typeInfo) {
137                                        currentLookup.putTypeInfo(info.getVariableName(), info);
138                                }
139                        }
140                        typeLookups.put(entity, currentLookup);
141
142                        // Add lookup from declaring entity if present
143                        Optional<ShallowEntity> additionalParent = getAdditionalParent(entity);
144                        if (additionalParent.isPresent()) {
145                                currentLookup.addAddtitionalParent(typeLookups.get(additionalParent.get()));
146                        }
147
148                        List<ShallowEntity> children = entity.getChildren();
149                        if (!children.isEmpty()) {
150                                createTypeInformation(children, currentLookup.createChildScope(), needsOrderedScope(entity),
151                                                typeLookups, typeInfoExtractor);
152                        }
153                }
154        }
155
156        /**
157         * Performs initialization steps for the given set of root entities.
158         *
159         * This is intended to be overwritten by subclasses. The default implementation
160         * does nothing.
161         */
162        protected void initialize(@SuppressWarnings("unused") List<ShallowEntity> rootEntities) {
163                // do nothing by default
164        }
165
166        /**
167         * Returns an additional shallowEntity whose type information scope should be
168         * visible for the given entity.
169         *
170         * This is intended to be overwritten by subclasses. The default implementation
171         * an empty optional.
172         */
173        protected Optional<ShallowEntity> getAdditionalParent(@SuppressWarnings("unused") ShallowEntity entity) {
174                return Optional.empty();
175        }
176
177        /**
178         * Adds default namespaces to the initial variable lookup.
179         *
180         * This is intended to be overwritten by subclasses. The default implementation
181         * does nothing.
182         */
183        protected void addDefaultImports(@SuppressWarnings("unused") ScopedTypeLookup variableLookup) {
184                // do nothing by default
185        }
186
187        /**
188         * Checks whether the given entity is an import of a namespace and adds the
189         * namespace to the variable lookup.
190         *
191         * This is intended to be overwritten by subclasses. The default implementation
192         * does nothing.
193         */
194        protected void processImport(@SuppressWarnings("unused") ShallowEntity entity,
195                        @SuppressWarnings("unused") ScopedTypeLookup variableLookup) {
196                // do nothing by default
197        }
198
199        /**
200         * Returns whether the given entity needs an ordered scope. This is used e.g.
201         * for methods or statements, where variables are only accessible if they have
202         * been introduced in an entity earlier in the code. Other entities like types
203         * don't need an ordered scope, because introduced attributes are visible
204         * everywhere within the class.
205         *
206         * The default implementation returns {@code true} if the given entity's type is
207         * contained in {@link #ORDERED_ENTITY_TYPES}. Subclasses may override this in
208         * order to provide language-specific behavior.
209         */
210        protected boolean needsOrderedScope(ShallowEntity entity) {
211                return ORDERED_ENTITY_TYPES.contains(entity.getType());
212        }
213
214        /**
215         * Returns whether this entity needs a separate variable scope. This is the case
216         * for methods and some statements that are determined by
217         * {@link #getSeparateScopeSubtypes()}, as otherwise those would introduce type
218         * information into the wrong scope.
219         */
220        private boolean needsSeparateScope(ShallowEntity entity) {
221                return entity.getType().equals(EShallowEntityType.METHOD)
222                                || getSeparateScopeSubtypes().contains(entity.getSubtype());
223        }
224
225        /** Returns all statement sub-types that need a separate scope. */
226        protected abstract Set<String> getSeparateScopeSubtypes();
227
228}
