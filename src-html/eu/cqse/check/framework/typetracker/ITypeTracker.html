017package eu.cqse.check.framework.typetracker;
018
019import java.util.List;
020
021import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
022
023/**
024 * A type tracker attaches type information about known variables to shallow
025 * entities. It creates a {@link ITypeResolution} that provides a mapping from
026 * shallow entities to type lookups.
027 * <p>
028 * This interface does not guarantee thread-safety or reusability across
029 * different input files. Implementing classes may hold state about the
030 * currently processed file, so that you should always obtain a new type tracker
031 * instance for every input.
032 */
033public interface ITypeTracker {
034
035        /**
036         * Creates type information for all given shallow entities and their children. A
037         * type resolution is returned.
038         */
039        ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities);
040
041        /**
042         * Creates type information for all given shallow entities and their children
043         * and uses the given {@link ScopedTypeLookup} as root lookup. A type resolution
044         * is returned.
045         */
046        ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities, ScopedTypeLookup variableLookup);
047}
