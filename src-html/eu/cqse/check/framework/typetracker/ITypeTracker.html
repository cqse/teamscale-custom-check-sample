017package eu.cqse.check.framework.typetracker;
018
019import java.util.List;
020
021import org.conqat.lib.commons.logging.ILogger;
022
023import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
024
025/**
026 * A type tracker attaches type information about known variables to shallow
027 * entities. It creates a {@link ITypeResolution} that provides a mapping from
028 * shallow entities to type lookups.
029 * <p>
030 * This interface does not guarantee thread-safety or reusability across
031 * different input files. Implementing classes may hold state about the
032 * currently processed file, so that you should always obtain a new type tracker
033 * instance for every input.
034 */
035public interface ITypeTracker {
036
037        /**
038         * Creates type information for all given shallow entities and their
039         * children. A type resolution is returned.
040         */
041        public ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities, ILogger logger);
042
043        /**
044         * Creates type information for all given shallow entities and their
045         * children and uses the given {@link ScopedTypeLookup} as root lookup. A
046         * type resolution is returned.
047         */
048        public ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities, ILogger logger,
049                        ScopedTypeLookup variableLookup);
050}
