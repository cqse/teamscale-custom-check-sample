032/**
033 * Main class of the C/C++ preprocessor. It iterates over the tokens of a given
034 * {@link org.conqat.engine.index.shared.BasicTokenElementInfo} and generates
035 * {@link PreprocessorTokenReplacement} objects for each token sequence that is
036 * replaced by a different token sequence.
037 * <p>
038 * Besides the whole-file iteration, this class handles simple tasks such as
039 * #define directives. Complex tasks such as macro expansion or #if directives
040 * are delegated (e.g., to {@link NewMacroInvocationPreprocessor} and
041 * {@link IfDirectivePreprocessor}.
042 */
043public class NewCPreprocessor implements IPreprocessor {
044
045        /**
046         * Patterns used for parsing include statements. This also matches #import
047         * directives (Microsoft and also gcc). There is actually a small difference
048         * between them which we ignore here: #import directives include each file at
049         * most once (#include is actually unlimited).
050         * <p>
051         * Match group 1 contains the included name enclosed in "" or <> (might also
052         * need a macro expansion first).
053         */
054        private static final Pattern INCLUDE_PATTERN = Pattern.compile("#\\s*(?:include|import)(.*)");
055        /**
056         * Patterns used for parsing error directives.
057         */
058        private static final Pattern ERROR_PATTERN = Pattern.compile("#\\s*error\\s*(.*)");
059        /**
060         * Patterns used for parsing warning directives.
061         */
062        private static final Pattern WARNING_PATTERN = Pattern.compile("#\\s*warning\\s*(.*)");
063        /**
064         * Patterns used for parsing line directives.
065         */
066        private static final Pattern LINE_PATTERN = Pattern.compile("#\\s*line\\s*(.*)");
067        /**
068         * Patterns used for parsing pragma directives. After pragma, we can have any
069         * chars (including newlines if preceded by \).
070         */
071        private static final Pattern PRAGMA_PATTERN = Pattern.compile("#\\s*pragma(.|(\\\n))*");
072        /**
073         * How deep #include statements can be nested. The C/C++ standard states that
074         * the limit is implementation defined. The Microsoft C++ compiler limits to 10.
075         * GCC seems to implement no limit.
076         */
077        private static final int INCLUDE_NESTING_DEPTH_LIMIT = 20;

079        /**
080         * This is relevant only if {@link #dropSourceCodeFromIncludedFiles} is true. In
081         * that case (normal Teamscale mode), we will ignore any tokens in included
082         * headers except for these token types.
083         */
084        private static final EnumSet<ETokenType> INTERESTING_TOKEN_TYPES_IN_HEADERS = EnumSet
085                        .of(ETokenType.PREPROCESSOR_DIRECTIVE, ETokenType.PREPROCESSOR_INCLUDE);

093        /**
094         * Whether to ignore any "normal" C code in files included via #include
095         * directives. In normal Teamscale operation we would set this to true since we
096         * want to find and report findings no findings on header code. (We want to
097         * report those in the header file.)
098         * <p>
099         * But a real compiler would include any code (including code in headers) in the
100         * compilation unit (would set this to false). We need this switch to properly
101         * test with the GCC test suite.
102         */
103        private final boolean dropSourceCodeFromIncludedFiles;
104
105        /**
106         * Whether we skip files that are included more than once. The inclusion might
107         * even be recursive (i.e., a.c contains #include a.c).
108         *
109         * In normal Teamscale operation this is false. However, our handling does not
110         * conform to the standard here. At least gcc allows to (even recursively)
111         * include the same header again. This can even make sense if the #defines have
112         * changed in the meantime (see TS-21506 for artificial example). However, this
113         * should be an extremely exotic case and just aborting should be fine for our
114         * needs.
115         *
116         * In the standard include-guard use case, the same header is actually included
117         * many times. However in each but the first inclusion, the guard macro will be
118         * defined, so no actual code will be processed.
119         */
120        private final boolean disallowRepeatedIncludes;

151        /**
152         * Computes a list of {@link PreprocessorTokenReplacement}s that represent the
153         * result of preprocessing the given tokens as if they were the entire content
154         * of a C/C++ file
155         * <p>
156         * This method assumes that the file that spawns a compilation unit (e.g., a
157         * xy.c file). This should not be called if the file is processed because of an
158         * include statement (#include xy.c) because we then would lose the
159         * include-hierarchy info.
160         *
161         * @return A list of token replacements. Each element's
162         *         {@link PreprocessorTokenReplacement#originalTokensStartIndex} refers
163         *         to the given fileTokens list. This list is sorted by
164         *         {@link PreprocessorTokenReplacement#originalTokensStartIndex} and the
165         *         replaced regions are non-overlapping.
166         */
167        public List<PreprocessorTokenReplacement> computeTopLevelReplacements(String uniformPath, List<IToken> fileTokens) {
168                IncludedFilesContext includedFiles = new IncludedFilesContext(uniformPath);
169                return computeTopLevelReplacements(uniformPath, fileTokens, includedFiles);
170        }

172        /**
173         * Computes a list of {@link PreprocessorTokenReplacement}s that represent the
174         * result of preprocessing the given tokens as if they were the entire content
175         * of a C/C++ file.
176         * <p>
177         * This uses and extends the given includedFiles (if the file contains #include
178         * statements).
179         *
180         * @return A list of token replacements. Each element's
181         *         {@link PreprocessorTokenReplacement#originalTokensStartIndex} refers
182         *         to the given fileTokens list. This list is sorted by
183         *         {@link PreprocessorTokenReplacement#originalTokensStartIndex} and the
184         *         replaced regions are non-overlapping.
185         */
186        private List<PreprocessorTokenReplacement> computeTopLevelReplacements(String uniformPath, List<IToken> fileTokens,
187                        IncludedFilesContext includedFiles) {
188                NewMacroInvocationPreprocessor macroPreprocessor = new NewMacroInvocationPreprocessor(expansionStepsLogger);
189                List<PreprocessorTokenReplacement> replacements = new ArrayList<>();
190                boolean isCpp = CppLanguageFeatureParser.isCpp(uniformPath, fileTokens);
191                IfDirectivePreprocessor.ConditionalRegionStack conditionalRegionStack = new IfDirectivePreprocessor.ConditionalRegionStack();
192                FilePreprocessorContext fileContext = new FilePreprocessorContext(macroProvider, uniformPath, isCpp, 0);
193                for (int originalTokenIndex = 0; originalTokenIndex < fileTokens.size(); originalTokenIndex++) {
194                        PreprocessorTokenReplacement tokenReplacement;
195                        try {
196                                if (!replacements.isEmpty()
197                                                && CollectionUtils.getLast(replacements).replacesOriginalTokenIndex(originalTokenIndex)) {
198                                        continue;
199                                }
200                                IToken token = fileTokens.get(originalTokenIndex);
201                                if (dropSourceCodeFromIncludedFiles && includedFiles.includeHierarchy.size() > 1
202                                                && !INTERESTING_TOKEN_TYPES_IN_HEADERS.contains(token.getType())) {
203                                        tokenReplacement = createReplacementUntilNextPreprocessorDirective(fileTokens, originalTokenIndex);
204                                } else {
205                                        fileContext.setLineNumber(token.getLineNumber());
206                                        tokenReplacement = computeReplacementForToken(originalTokenIndex, fileTokens, macroPreprocessor,
207                                                        fileContext, conditionalRegionStack, includedFiles);
208                                }
209                        } catch (AssertionError | StackOverflowError | PreprocessorException e) {
210                                /*
211                                 * We catch some rather hard exceptions here and might think about adding more
212                                 * (e.g. NPE). Two reasons: First, if an exception comes up here, it is probably
213                                 * triggered by very special, complex macros being expanded. We need to know
214                                 * which macro in which context to reproduce this. Second, there is a good
215                                 * chance that we can preprocess the rest of the code. So we should not
216                                 * terminate the ContentIndexSynchronizer because of a bug that is triggered by
217                                 * a special macro.
218                                 */
219                                LOGGER.error("Exception while preprocessing " + fileTokens.get(originalTokenIndex).getOriginId() + ":"
220                                                + fileTokens.get(originalTokenIndex).getLineNumber() + "\ninclude hierarchy: "
221                                                + includedFiles.getCurrentIncludeHierarchy(), e);
222                                // leave the token in. If this is a function macro, it will probably look like a
223                                // function call and removing it would make parsing difficult. We add a 0-length
224                                // replacement to attach the error message.
225                                tokenReplacement = new PreprocessorErrorTokenReplacement(e.getMessage(),
226                                                includedFiles.getCurrentIncludeHierarchy(), Collections.emptyList(), originalTokenIndex,
227                                                originalTokenIndex, new PreprocessorUsageInformation());
228                        }
229                        if (tokenReplacement != null) {
230                                replacements.add(tokenReplacement);
231                        }
232                }
233                replacements.sort(Comparator.comparingInt(replacement -> replacement.originalTokensStartIndex));
234                return replacements;
235        }

237        /**
238         * Returns a new {@link PreprocessorTokenReplacement} that replaces all tokens
239         * from the current position to the next token with a type from
240         * {@link #INTERESTING_TOKEN_TYPES_IN_HEADERS} (or until the file end if no such
241         * token exists).
242         */
243        private static PreprocessorTokenReplacement createReplacementUntilNextPreprocessorDirective(List<IToken> fileTokens,
244                        int originalTokenIndex) {
245                int indexOfNextRelevantToken = TokenStreamUtils.firstTokenOfType(fileTokens, originalTokenIndex,
246                                fileTokens.size(), INTERESTING_TOKEN_TYPES_IN_HEADERS);
247                if (indexOfNextRelevantToken == TokenStreamUtils.NOT_FOUND) {
248                        indexOfNextRelevantToken = fileTokens.size();
249                }
250                // drop any non-preprocessor-directive tokens in headers
251                return new PreprocessorTokenReplacement(Collections.emptyList(), originalTokenIndex, indexOfNextRelevantToken,
252                                new PreprocessorUsageInformation());
253        }

255        /**
256         * Tries to preprocess the current token (and potentially following tokens that
257         * form a parameter list of the current token).
258         * <p>
259         * Returns a {@link PreprocessorTokenReplacement} if it anything needed to be
260         * preprocessed, null otherwise.
261         */
262        private PreprocessorTokenReplacement computeReplacementForToken(int originalTokenIndex, List<IToken> fileTokens,
263                        NewMacroInvocationPreprocessor macroPreprocessor, FilePreprocessorContext fileContext,
264                        IfDirectivePreprocessor.ConditionalRegionStack conditionalRegionStack, IncludedFilesContext includedFiles)
265                        throws PreprocessorException {
266                String uniformPath = fileContext.uniformPath;
267                IToken token = fileTokens.get(originalTokenIndex);
268                if (isUnknownDirectiveSituation(originalTokenIndex, fileTokens, token)) {
269                        // this fixes an issue that would be better handled in the scanner TS-21534
270                        int nextToken = getIndexOfNextLineToken(originalTokenIndex, fileTokens);
271                        return new PreprocessorTokenReplacement(CollectionUtils.emptyList(), originalTokenIndex, nextToken,
272                                        new PreprocessorUsageInformation());
273                }
274                PreprocessorTokenReplacement result = null;
275                // switch between the different token-types we can encounter. I intentionally
276                // don't use early return to have a proper place for a debug breakpoint.
277                Optional<MacroDefinition> macro = fileContext.findMacroDefinition(token, includedFiles);
278                if (macro.isPresent()) {
279                        result = macroPreprocessor.expandTopLevelMacroInvocation(originalTokenIndex, fileTokens, macro.get(),
280                                        fileContext, includedFiles);
281                        if (result != null) {
282                                result.setTokensOriginAndOffset(fileTokens.get(originalTokenIndex));
283                        }
284                } else if (token.getType() == ETokenType.PREPROCESSOR_INCLUDE
285                                || (token.getType() == ETokenType.PREPROCESSOR_DIRECTIVE
286                                                && INCLUDE_PATTERN.matcher(token.getText()).matches())) {
287                        PreprocessorUsageInformation preprocessorUsageInInclude = new PreprocessorUsageInformation();
288                        result = handleInclude(uniformPath, fileTokens, includedFiles, originalTokenIndex,
289                                        preprocessorUsageInInclude, fileContext, macroPreprocessor);
290                } else if (token.getType() == ETokenType.PREPROCESSOR_DIRECTIVE) {
291                        if (isErrorDirective(token) || isWarningDirective(token) || isLineDirective(token)) {
292                                // drop token, log if error
293                                result = handleSimpleDirective(includedFiles, originalTokenIndex, token, fileContext);
294                        } else if (IfDirectivePreprocessor.isConditionalDirective(token)) {
295                                result = IfDirectivePreprocessor.processIfDirective(fileTokens, originalTokenIndex,
296                                                conditionalRegionStack, fileContext, this, includedFiles);
297                        } else {
298                                handleIfMacroDefinitionDirective(token);
299                                // We drop unknown-directive tokens, pragma-directive tokens, assert-directive
300                                // tokens, single HASH token, and define/undef-directive tokens.
301                                result = PreprocessorTokenReplacement.createSingleTokenRemoval(originalTokenIndex);
302                        }
303                } else {
304                        // seems to be a normal (identifier?) token. To be correct, we would have to
305                        // store somewhere that the result depends on the fact that this is not defined
306                        // as a macro name.
307                }
308                return result;
309        }

311        /**
312         * Traverses the token sequence and returns the index of the next token that has
313         * a higher line number than the token at the current index. If there is no
314         * token with higher line number returns the size of the given list (i.e.,
315         * highest index + 1).
316         * <p>
317         * This does not handle \ newline escapes. Not sure whether we need that so far.
318         */
319        private static int getIndexOfNextLineToken(int startIndex, List<IToken> tokens) {
320                IToken startToken = tokens.get(startIndex);
321                int end = startIndex + 1;
322                while (end < tokens.size() && tokens.get(end).getLineNumber() == startToken.getLineNumber()) {
323                        end++;
324                }
325                return end;
326        }

328        /**
329         * Returns whether this is a ILLEGAL_CHARACTER token with text "#" that is the
330         * first token on its line (or the first token of a file).
331         * <p>
332         * Example for this: <code># 11 bar "any text"</code>. Since 11 is no name of a
333         * pp directive, this is ignored by real preprocessors.
334         * <p>
335         * Such tokens are scanned incorrectly by our scanner (TS-21534). Actually they
336         * should be scanned as a preprocessor directive (with maybe unknown name).
337         * However the # is scanned as ILLEGAL_CHARACTER and the rest as normal tokens.
338         */
339        private static boolean isUnknownDirectiveSituation(int originalTokenIndex, List<IToken> fileTokens, IToken token) {
340                return token.getType() == ETokenType.ILLEGAL_CHARACTER && token.getText().equals("#")
341                                && (originalTokenIndex == 0
342                                                || fileTokens.get(originalTokenIndex - 1).getLineNumber() < token.getLineNumber());
343        }

345        /**
346         * Handles all simple directives such as #error, #warning, #line. Logs an error
347         * if an error directive is hit, otherwise drops the token.
348         */
349        private static PreprocessorTokenReplacement handleSimpleDirective(IncludedFilesContext includedFiles,
350                        int originalTokenIndex, IToken token, FilePreprocessorContext context) {
351                if (!isErrorDirective(token)) {
352                        // just drop this token
353                        return PreprocessorTokenReplacement.createSingleTokenRemoval(originalTokenIndex);
354                }
355                PreprocessorTokenReplacement replacementFromCurrentStep;
356                String locationAndErrorMessage = token.getOriginId() + ":" + token.getLineNumber() + ": " + token.getText();
357                String preprocessorMessage = "hit error directive in " + locationAndErrorMessage;
358                if (context.storeNewLoggedErrorDirective(locationAndErrorMessage)) {
359                        LogManager.getLogger().warn(preprocessorMessage + "\n(only first hit is logged)");
360                }
361                replacementFromCurrentStep = createErrorSingleTokenReplacement(preprocessorMessage, originalTokenIndex,
362                                includedFiles);
363                return replacementFromCurrentStep;
364        }

378        /**
379         * Finds the correct header file for the current include-directive token and
380         * preprocesses that file. Returns a PreprocessorTokenReplacement that replaces
381         * the current token with
382         * <ul>
383         * <li>the source code of the header (transitive) if
384         * {@link #dropSourceCodeFromIncludedFiles} is false or</li>
385         * <li>the empty list if {@link #dropSourceCodeFromIncludedFiles} is true</li>
386         * </ul>
387         */
388        private PreprocessorTokenReplacement handleInclude(String uniformPath, List<IToken> fileTokens,
389                        IncludedFilesContext includedFiles, int originalTokenIndex,
390                        PreprocessorUsageInformation preprocessorUsageInformation, FilePreprocessorContext fileContext,
391                        NewMacroInvocationPreprocessor macroPreprocessor) throws PreprocessorException {
392                IToken token = fileTokens.get(originalTokenIndex);
393                if (includedFiles.getCurrentIncludeHierarchy().size() > INCLUDE_NESTING_DEPTH_LIMIT) {
394                        return createErrorSingleTokenReplacement("hit include level limit of " + INCLUDE_NESTING_DEPTH_LIMIT,
395                                        originalTokenIndex, includedFiles);
396                }
397
398                Matcher matcher = INCLUDE_PATTERN.matcher(token.getText());
399                if (!matcher.find()) {
400                        return createErrorSingleTokenReplacement("could not parse include directive", originalTokenIndex,
401                                        includedFiles);
402                }
403                Optional<String> includedName = extractIncludedName(matcher.group(1).trim(), includedFiles, fileContext,
404                                macroPreprocessor);
405                if (!includedName.isPresent()) {
406                        return createErrorSingleTokenReplacement("could not parse include directive", originalTokenIndex,
407                                        includedFiles);
408                }
409                IncludedTokens pathAndTokens = resolveIncludedTokens(uniformPath, includedName.get());
410                String resolvedIncludePath = pathAndTokens.getFirst();
411                List<IToken> includedTokens = pathAndTokens.getSecond();
412                if (resolvedIncludePath == null && includedTokens.isEmpty()) {
413                        preprocessorUsageInformation.addUnresolvedIncludePath(includedName.get());
414                        return createErrorSingleTokenReplacement("unresolved include", originalTokenIndex, includedFiles);
415                }
416                if (disallowRepeatedIncludes && resolvedIncludePath != null
417                                && includedFiles.isAlreadyIncludedBefore(resolvedIncludePath)) {
418                        // repeated include (not necessarily recursive)
419                        return createErrorSingleTokenReplacement("repeated include", originalTokenIndex, includedFiles);
420                }
421                return preprocessIncludedTokens(includedTokens, resolvedIncludePath, fileTokens.get(originalTokenIndex),
422                                includedFiles, preprocessorUsageInformation, originalTokenIndex);
423        }

425        /**
426         * Given the "name" part of an #include directive (everything behind the include
427         * keyword), returns the name of the included file. In the easiest case, this
428         * means just removing enclosing "" or <>, but the name can also be a macro
429         * invocation which needs to be expanded to determine the included name.
430         * <p>
431         * Returns the included file name or {@link Optional#empty()} if it can't be
432         * determined.
433         */
434        private static Optional<String> extractIncludedName(String includeDirectiveNamePart,
435                        IncludedFilesContext includedFiles, FilePreprocessorContext fileContext,
436                        NewMacroInvocationPreprocessor macroPreprocessor) throws PreprocessorException {
437                List<IToken> includedNameTokens = NewCPreprocessingUtils.scanMacroContent(includeDirectiveNamePart);
438                if (includedNameTokens.isEmpty()) {
439                        return Optional.empty();
440                }
441                Optional<String> result = extractIncludedNameWithoutMacroExpansion(includedNameTokens);
442                if (result.isPresent()) {
443                        return result;
444                }
445                List<IToken> expandedTokens = macroPreprocessor.expandMacrosInTokenList(includedNameTokens,
446                                Collections.emptyList(), fileContext, new NewMacroInvocationPreprocessor.TokenDisablingContext(),
447                                new PreprocessorUsageInformation(), includedFiles).getFirst();
448                result = extractIncludedNameWithoutMacroExpansion(expandedTokens);
449                if (result.isPresent()) {
450                        return Optional.of(result.get()
451                                        // Need to remove whitespace since we sometimes insert too much whitespace in
452                                        // stringification. Assuming filenames don't contain whitespace. This happens in
453                                        // test include3.c of gcc testsuite.
454                                        .replaceAll(" ", ""));
455                }
456                return Optional.empty();
457        }

459        /**
460         * Extracts the name of the included file from the given token stream assuming
461         * that this is the tokens after a #include or #import keyword. It can either be
462         * a single literal enclosed in \" or a list of tokens enclosed in
463         * {@link ETokenType#LT}, {@link ETokenType#GT}.
464         */
465        private static Optional<String> extractIncludedNameWithoutMacroExpansion(List<IToken> includedNameTokens) {
466                if (includedNameTokens.isEmpty()) {
467                        return Optional.empty();
468                }
469                IToken firstToken = includedNameTokens.get(0);
470                if (firstToken.getType() == ETokenType.STRING_LITERAL && firstToken.getText().length() > 2) {
471                        return Optional.of(firstToken.getText().substring(1, firstToken.getText().length() - 1));
472                } else if (includedNameTokens.size() > 2 && TokenStreamUtils.startsWith(includedNameTokens, ETokenType.LT)
473                                && TokenStreamUtils.endsWith(includedNameTokens, ETokenType.GT)) {
474                        return Optional.of(TokenStreamTextUtils
475                                        .concatTokenTexts(includedNameTokens.subList(1, includedNameTokens.size() - 1)));
476                }
477                return Optional.empty();
478        }

480        /**
481         * Preprocesses the given included tokens (tokens to be included because of an
482         * #include directive) and adds the preprocessor usage info gained during this
483         * preprocessing to the parameter preprocessorUsageInformation.
484         */
485        private PreprocessorTokenReplacement preprocessIncludedTokens(List<IToken> includedTokens,
486                        String resolvedIncludePath, IToken includeDirectiveToken, IncludedFilesContext includedFiles,
487                        PreprocessorUsageInformation preprocessorUsageInformation, int originalTokenIndex) {
488                preprocessorUsageInformation.addIncludedPath(resolvedIncludePath);
489                includedFiles.pushIncludedPath(resolvedIncludePath);
490                List<PreprocessorTokenReplacement> replacements = computeTopLevelReplacements(resolvedIncludePath,
491                                includedTokens, includedFiles);
492                includedFiles.popIncludedPath();
493                List<IToken> resultTokens = NewCPreprocessingUtils.applyReplacements(replacements, includedTokens);
494                for (int i = 0; i < resultTokens.size(); i++) {
495                        IToken resultToken = resultTokens.get(i);
496                        resultTokens.set(i,
497                                        resultToken.newToken(resultToken.getType(), includeDirectiveToken.getOffset(),
498                                                        includeDirectiveToken.getLineNumber(), resultToken.getText(),
499                                                        MacroHandlingCPreprocessorBase.MACRO_ORIGIN));
500                }
501                List<String> errors = new ArrayList<>();
502                for (PreprocessorTokenReplacement replacement : replacements) {
503                        preprocessorUsageInformation.addFrom(replacement.preprocessorUsageInformation);
504                        if (replacement instanceof PreprocessorErrorTokenReplacement) {
505                                errors.add(((PreprocessorErrorTokenReplacement) replacement).errorMessage);
506                        }
507                }
508                if (!errors.isEmpty()) {
509                        return new PreprocessorErrorTokenReplacement(StringUtils.concat(errors, "\n"),
510                                        includedFiles.getCurrentIncludeHierarchy(), resultTokens, originalTokenIndex,
511                                        originalTokenIndex + 1, preprocessorUsageInformation);
512                }
513                return new PreprocessorTokenReplacement(resultTokens, originalTokenIndex, originalTokenIndex + 1,
514                                preprocessorUsageInformation);
515        }

532        /**
533         * Handles any preprocessor directives affecting macro definitions.
534         */
535        private void handleIfMacroDefinitionDirective(IToken token) {
536                // strip leading '#'
537                String content = token.getText().trim().substring(1);
538                List<IToken> subTokens = NewCPreprocessingUtils.scanMacroContent(content);
539                if (subTokens.isEmpty()) {
540                        return;
541                }
542                switch (subTokens.get(0).getText()) {
543                case "define":
544                        if (subTokens.size() > 1) {
545                                macroProvider.define(StringUtils.stripPrefix(content.trim(), "define").trim(),
546                                                NewCPreprocessingUtils.locationFromToken(token));
547                        }
548                        return;
549                case "undef":
550                        if (subTokens.size() > 1) {
551                                macroProvider.undefine(subTokens.get(1).getText());
552                        }
553                        return;
554                default:
555                        // ignore other/unknown directives
556                        return;
557                }
558        }

587        /**
588         * Context of the preprocessing process of one file (i.e., of a c file or of an
589         * included header, not of the entire compilation unit).
590         */
591        /* package */ static class FilePreprocessorContext {
592                private final ParsedMacroProvider macroProvider;
593                /* package */ final boolean isCpp;
594                /* package */ final String uniformPath;
595                /**
596                 * The line number of the currently processed macro invocation. This is
597                 * primarily needed for the __LINE__ macro.
598                 * {@link CompilerDefinedMacroSupplier.CompilerDefinedMacroNames#LINE}
599                 * 
600                 * The line number must be mutable because we can't determine the last line of
601                 * the current macro invocation when seeing the macro-name token (and creating
602                 * this context).
603                 */
604                /* package */int lineNumber;
605
606                /**
607                 * Set of the #error directives that we encountered during preprocessing and
608                 * that were already logged (in worker log). We want to avoid logging them a
609                 * second time since this can easily make the worker log unusable.
610                 */
611                private final HashSet<String> loggedErrorDirectiveHits = new HashSet<>();
612
613                private FilePreprocessorContext(ParsedMacroProvider macroProvider, String uniformPath, boolean isCpp,
614                                int lineNumber) {
615                        this.macroProvider = macroProvider;
616                        this.uniformPath = uniformPath;
617                        this.lineNumber = lineNumber;
618                        this.isCpp = isCpp;
619                }
620
621                /**
622                 * Stores the fact that we encountered and logged an #error directive with the
623                 * given location and message. Returns true if this was the first time this
624                 * location and message was encountered. The format of the
625                 * locationAndErrorMessage string is defined by the callers of this method.
626                 */
627                private boolean storeNewLoggedErrorDirective(String locationAndErrorMessage) {
628                        return loggedErrorDirectiveHits.add(locationAndErrorMessage);
629                }
630
631                /* package */ void setLineNumber(int lineNumber) {
632                        this.lineNumber = lineNumber;
633                }
634
635                /**
636                 * Searches for the current {@link MacroDefinition} for the text of the given
637                 * token. Returns {@link Optional#empty()} if no such macro is known.
638                 * <p>
639                 * First checks in the user-defined macros, then in the
640                 * {@link CompilerDefinedMacroSupplier}. I.e., allows the user to overwrite
641                 * compiler-defined macros.
642                 */
643                /* package */
644                Optional<MacroDefinition> findMacroDefinition(IToken currentToken, IncludedFilesContext includedFiles) {
645                        String currentMacroName = currentToken.getText();
646                        Optional<MacroDefinition> macro = Optional.ofNullable(macroProvider.getDefinition(currentMacroName));
647                        if (!macro.isPresent()) {
648                                macro = CompilerDefinedMacroSupplier.getMacroForName(currentMacroName, includedFiles, this);
649                        }
650                        return macro;
651                }
652        }

655         * Collection of debugging information (not only for Teamscale debugging, but

659         * which include paths could not be found.

661         */

664                private Set<MacroDefinition> definitionDependencyMacros = null;
665                private Set<String> includedUniformPaths = null;
666                private Set<String> unresolvedIncludePaths = null;

668                private void initializeIfNecessary() {

670                                        || unresolvedIncludePaths == null) {

672                                definitionDependencyMacros = new HashSet<>();
673                                includedUniformPaths = new HashSet<>();
674                                unresolvedIncludePaths = new HashSet<>();
675                        }
676                }

678                /* package */ void addMacroContentDependencyOn(MacroDefinition macro) {
679                        initializeIfNecessary();
680                        contentDependencyMacros.add(macro);
681                }
682
683                /* package */ void addMacroDefinitionDependencyOn(MacroDefinition macro) {
684                        initializeIfNecessary();
685                        definitionDependencyMacros.add(macro);
686                }
687
688                public Set<MacroDefinition> getDefinitionDependencyMacros() {
689                        if (definitionDependencyMacros == null) {
690                                return Collections.emptySet();
691                        }
692                        return definitionDependencyMacros;
693                }
694
695                public Set<MacroDefinition> getContentDependencyMacros() {
696                        if (contentDependencyMacros == null) {
697                                return Collections.emptySet();
698                        }
699                        return contentDependencyMacros;

701
702                public Set<String> getIncludedUniformPaths() {
703                        if (includedUniformPaths == null) {
704                                return Collections.emptySet();
705                        }
706                        return includedUniformPaths;
707                }
708
709                private void addIncludedPath(String uniformPath) {
710                        initializeIfNecessary();
711                        includedUniformPaths.add(uniformPath);
712                }

714                /**
715                 * Adds all information from the given {@link PreprocessorUsageInformation}
716                 * object to this.
717                 */
718                public void addFrom(PreprocessorUsageInformation preprocessorUsageInformation) {
719                        initializeIfNecessary();
720                        contentDependencyMacros.addAll(preprocessorUsageInformation.getContentDependencyMacros());
721                        definitionDependencyMacros.addAll(preprocessorUsageInformation.getDefinitionDependencyMacros());
722                        addIncludeInfoFrom(preprocessorUsageInformation);

730                        initializeIfNecessary();
731                        includedUniformPaths.addAll(preprocessorUsageInformation.getIncludedUniformPaths());
732                        unresolvedIncludePaths.addAll(preprocessorUsageInformation.getUnresolvedIncludePaths());
733                }

742                }

749                                return Collections.emptySet();

751                        return unresolvedIncludePaths;

755        /**
756         * Stores which files were included (to resolve cycles) and the current include
757         * hierarchy (for use in the __INCLUDE_LEVEL__ macro and in error messages).
758         * <p>
759         * This is based on the entire file (not on a single top-level macro expansion).
760         */
761        /* package */ static class IncludedFilesContext {
762
763                /**
764                 * Log of all uniform paths that were included in the current preprocessing
765                 * (preprocessing of the main file/ compilation unit).
766                 * <p>
767                 * This always includes the initial source file (.c file) as first element.
768                 */
769                private final Set<String> allIncludedPaths = new HashSet<>();
770                /**
771                 * Stack of #include directives that are currently expanded. For each directive,
772                 * we store the uniform paths that is included. We push at the beginning of an
773                 * included file and pop at the end.
774                 * <p>
775                 * This always includes the initial source file (.c file) as first element.
776                 */
777                private final Deque<String> includeHierarchy = new ArrayDeque<>();
778
779                private IncludedFilesContext(String sourceFileUniformPath) {
780                        pushIncludedPath(sourceFileUniformPath);
781                }
782
783                /**
784                 * Pushes the given path on the include-hierarchy stack.
785                 */
786                private void pushIncludedPath(String uniformPath) {
787                        allIncludedPaths.add(uniformPath);
788                        includeHierarchy.push(uniformPath);
789                }
790
791                /**
792                 * Pops one uniform path from the include-hierarchy stack.
793                 */
794                private void popIncludedPath() {
795                        includeHierarchy.pop();
796                }
797
798                /**
799                 * Returns whether the given uniform path has already been included before
800                 * during preprocessing of this file (compilation unit). If this returns true,
801                 * that does not necessarily mean that the uniform path is in the current
802                 * {@link #includeHierarchy} (i.e., include recursion), its inclusion could also
803                 * have been completed.
804                 */
805                private boolean isAlreadyIncludedBefore(String uniformPath) {
806                        return allIncludedPaths.contains(uniformPath);
807                }
808
809                /**
810                 * @see #includeHierarchy
811                 */
812                public UnmodifiableList<String> getCurrentIncludeHierarchy() {
813                        return CollectionUtils.asUnmodifiable(new ArrayList<>(includeHierarchy));
814                }
815        }
