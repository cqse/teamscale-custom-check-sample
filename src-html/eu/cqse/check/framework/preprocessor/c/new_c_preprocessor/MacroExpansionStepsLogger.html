001package eu.cqse.check.framework.preprocessor.c.new_c_preprocessor;
002
003import java.io.IOException;
004import java.io.OutputStreamWriter;
005import java.io.PrintStream;
006import java.io.Writer;
007import java.nio.charset.StandardCharsets;
008import java.util.List;
009import java.util.function.Function;
010import java.util.function.Supplier;
011
012import org.apache.logging.log4j.LogManager;
013import org.conqat.lib.commons.collections.CollectionUtils;
014import org.conqat.lib.commons.string.StringUtils;
015
016import eu.cqse.check.framework.scanner.IToken;
017
018/**
019 * "Logger" for storing intermediate steps during preprocessor expansion. In
020 * production this will usually not be used (i.e., use the {@link #NO_OP} Logger
021 * which does nothing).
022 * <p>
023 * This needs to be in the production code (rather than test code) because it is
024 * used at least one debug service in production instances
025 * (PreprocessorExpansionDebugService).
026 */
027public class MacroExpansionStepsLogger {
028
029        /**
030         * The maximum length we assume for log messages (excluding the macro content).
031         * This is no hard requirement, longer messages will be shown. This is just used
032         * for alignment (readability improvement).
033         */
034        private static final int MAX_ASSUMED_MESSAGE_LENGTH = 30;
035
036        /**
037         * The default {@link MacroExpansionStepsLogger}, which does nothing and does
038         * not cost runtime.
039         */
040        public static final MacroExpansionStepsLogger NO_OP = new MacroExpansionStepsLogger() {
041                @Override
042                public void incrementNestingDepth() {
043                        // empty
044                }
045
046                @Override
047                public void decrementNestingDepth() {
048                        // empty
049                }
050
051                @Override
052                public void logStatusMessage(List<IToken> macroContent, String logMessage,
053                                NewMacroInvocationPreprocessor.TokenDisablingContext tokenDisablingContext) {
054                        // empty
055                }
056
057                @Override
058                public void logStatusMessage(List<IToken> macroContent, Supplier<String> logMessage,
059                                NewMacroInvocationPreprocessor.TokenDisablingContext tokenDisablingContext) {
060                        // empty
061                }
062
063                @Override
064                public void logStatusMessage(List<IToken> currentTokens1, List<IToken> currentTokens2, String logMessage,
065                                NewMacroInvocationPreprocessor.TokenDisablingContext tokenDisablingContext) {
066                        // empty
067                }
068
069                @Override
070                public void logStatusMessage(String messageTag, Supplier<String> messageContent) {
071                        // empty
072                }
073        };
074        /**
075         * The output writer to which we store the logged messages.
076         */
077        private final Writer outputWriter;
078
079        /**
080         * How many tabs to prepend before each message. This is used to direct the
081         * indentation/nesting depth of log messages.
082         */
083        private int nestingDepth = 0;
084
085        /**
086         * Creates a {@link MacroExpansionStepsLogger} that writes to a PrintStream with
087         * UTF-8 encoding. Call with <code>System.out</code> argument for debugging.
088         */
089        public MacroExpansionStepsLogger(PrintStream outputPrintStream) {
090                outputWriter = new OutputStreamWriter(outputPrintStream, StandardCharsets.UTF_8);
091        }
092
093        /**
094         * Creates a {@link MacroExpansionStepsLogger} that writes log messages to the
095         * given Writer.
096         */
097        public MacroExpansionStepsLogger(Writer outputPrintWriter) {
098                outputWriter = outputPrintWriter;
099        }
100
101        /**
102         * Private constructor that must be used by
103         * {@link MacroExpansionStepsLogger#NO_OP} only. This writes null to the
104         * {@link #outputWriter} field.
105         */
106        private MacroExpansionStepsLogger() {
107                outputWriter = null;
108        }
109
110        /**
111         * Increment the tab-nesting depth before future messages.
112         */
113        public void incrementNestingDepth() {
114                nestingDepth++;
115        }
116
117        /**
118         * Decrement the tab-nesting depth before future messages.
119         */
120        public void decrementNestingDepth() {
121                nestingDepth--;
122        }
123
124        /**
125         * Logs the given message with printing the current tokens (in contrast to
126         * {@link #logStatusMessage(String, String)}).
127         * <p>
128         * Log message will look like this:<br/>
129         * INDENTATION_SPACES logMessage MORE_SPACES tokens
130         * <p>
131         * The number of spaces in INDENTATION_SPACES corresponds to the current
132         * indentation level. The number of spaces in MORE_SPACES is computed such that
133         * tokens of all messages on the same indentation level start in the same column
134         * (improves readability) unless logMessage is too long.
135         * <p>
136         * Tokens that are "painted blue"/flagged as no_expand in the given
137         * tokenDisablingContext are marked with a °.
138         */
139        public void logStatusMessage(List<IToken> macroContent, String logMessage,
140                        NewMacroInvocationPreprocessor.TokenDisablingContext tokenDisablingContext) {
141                try {
142                        logMessage += NewCPreprocessingUtils.getFillerSpaces(logMessage, MAX_ASSUMED_MESSAGE_LENGTH);
143                        String currentContent = concatTokenTextsWithBluePaintMarkers(macroContent, tokenDisablingContext);
144                        String completeMessage = StringUtils.repeat("  ", nestingDepth + 1) + logMessage + currentContent;
145                        outputWriter.write(completeMessage + "\n");
146                        outputWriter.flush();
147                } catch (IOException e) {
148                        LogManager.getLogger().error("IOException while writing preprocessor expansion steps:", e);
149                }
150        }
151
152        /**
153         * Delegates to
154         * {@link #logStatusMessage(List, String, NewMacroInvocationPreprocessor.TokenDisablingContext)}
155         * using the given {@link Supplier} for the log message. Since this log call
156         * does nothing in production (when using {@link #NO_OP}), using the
157         * {@link Supplier} will be cheaper than building an unnecessary String.
158         */
159        public void logStatusMessage(List<IToken> macroContent, Supplier<String> logMessage,
160                        NewMacroInvocationPreprocessor.TokenDisablingContext tokenDisablingContext) {
161                logStatusMessage(macroContent, logMessage.get(), tokenDisablingContext);
162        }
163
164        /**
165         * Delegates to
166         * {@link #logStatusMessage(List, String, NewMacroInvocationPreprocessor.TokenDisablingContext)}
167         * but concatenates the two given lists first. Using this method (with a two
168         * lists) will be cheaper in production (using {@link #NO_OP} since
169         * concatenating the token lists might be expensive and the list would be
170         * dropped anyway (if using {@link #NO_OP}).
171         */
172        public void logStatusMessage(List<IToken> currentTokens1, List<IToken> currentTokens2, String logMessage,
173                        NewMacroInvocationPreprocessor.TokenDisablingContext tokenDisablingContext) {
174                logStatusMessage(NewCPreprocessingUtils.concat(currentTokens1, currentTokens2), logMessage,
175                                tokenDisablingContext);
176        }
177
178        /**
179         * Concatenates the given token's texts with " " separators. Appends a ° to each
180         * token that is painted blue in the given
181         * {@link NewMacroInvocationPreprocessor.TokenDisablingContext} (i.e., is
182         * flagged as "no-expand").
183         */
184        private static String concatTokenTextsWithBluePaintMarkers(List<IToken> tokens,
185                        NewMacroInvocationPreprocessor.TokenDisablingContext tokenDisablingContext) {
186                Function<IToken, String> computeTokenRepresentation = token -> {
187                        if (tokenDisablingContext.isPaintedBlue(token)) {
188                                return token.getText() + "°";
189                        }
190                        return token.getText();
191                };
192                return StringUtils.concat(CollectionUtils.map(tokens, computeTokenRepresentation), " ");
193        }
194
195        /**
196         * Logs the given message without printing the current tokens (in contrast to
197         * {@link #logStatusMessage(List, String, NewMacroInvocationPreprocessor.TokenDisablingContext)}).
198         * <p>
199         * Log message will look like this:<br/>
200         * INDENTATION_SPACES messageTag MORE_SPACES messageContent
201         * <p>
202         * The number of spaces in INDENTATION_SPACES corresponds to the current
203         * indentation level. The number of spaces in MORE_SPACES is computed such that
204         * messageContent of all messages on the same indentation level start in the
205         * same column (improves readability) unless messageTag is too long.
206         */
207        public void logStatusMessage(String messageTag, Supplier<String> messageContent) {
208                try {
209                        String fillerSpaces = NewCPreprocessingUtils.getFillerSpaces(messageTag, MAX_ASSUMED_MESSAGE_LENGTH);
210                        String completeMessage = StringUtils.repeat("  ", nestingDepth + 1) + messageTag + fillerSpaces
211                                        + messageContent.get();
212                        outputWriter.write(completeMessage + "\n");
213                        outputWriter.flush();
214                } catch (IOException e) {
215                        LogManager.getLogger().error("IOException while writing preprocessor expansion steps:", e);
216                }
217        }
218
219}
