030/**
031 * Representation of a C preprocessor macro. Contains all information from the
032 * define directive that defines the macro including the
033 * {@link TextRegionLocation} of the define.
034 */
035public class MacroDefinition implements Serializable {
036
037        private static final long serialVersionUID = 1L;
038
039        /**
040         * {@link Pattern} that matches on any char that can not be part of an
041         * identifier in C.
042         */
043        private static final Pattern NO_CPP_IDENTIFIER_CHAR_PATTERN = Pattern.compile("[^A-Za-z0-9_]");
044
045        /**
046         * The identifier used to invoke this macro. For the macro definition
047         * <code>#define CAT(x,y) x##y</code>, this is <code>"CAT"</code>.
048         */
049        public final String macroName;
050
051        /**
052         * The "content" of the macro. This is the string/token list that is expanded
053         * and then replaces the macro-invocation tokens.
054         * <p>
055         * For the macro definition <code>#define CAT(x,y) x##y</code>, this is
056         * <code>"x##y"</code>.
057         */
058        public final List<IToken> replacementList;
059
060        /**
061         * Whether this is a function macro (otherwise an object macro). Function macros
062         * accept parameters and require parentheses to be invoked.
063         */
064        public final boolean isFunctionMacro;
065
066        /**
067         * The names of the parameters of this function. Empty if
068         * {@link #isFunctionMacro} is false.
069         * <p>
070         * If the last parameter in the macro declaration is "..." (variadic parameter),
071         * its name is "__VA_ARGS__" in this list. If the last parameter in the macro
072         * declaration is "xy..." (named variadic parameter, common gcc extension), its
073         * name is "xy" in this list. In both cases we store the id with which the
074         * parameter value can be referred to in the replacement list.
075         */
076        public final UnmodifiableList<String> parameterNames;
077
078        /**
079         * Whether the last parameter in this macro definition is a variadic parameter.
080         * False if {@link #isFunctionMacro} is false.
081         */
082        public final boolean hasVariadicParameter;
083
084        /**
085         * Location where this macro was declared in the source code.
086         */
087        public final TextRegionLocation macroDeclarationLocation;
088
089        private MacroDefinition(String macroName, String replacementList, boolean isFunctionMacro,
090                        List<String> parameterNames, boolean hasVariadicParameter, TextRegionLocation macroDeclarationLocation) {
091                this.macroName = macroName;
092                if (replacementList.isEmpty()) {
093                        this.replacementList = Collections.emptyList();
094                } else {
095                        this.replacementList = new ArrayList<>();
096                        for (IToken token : ScannerUtils.getTokens(replacementList, ELanguage.CPP,
097                                        MacroHandlingCPreprocessorBase.MACRO_ORIGIN)) {
098                                // this is a bit ugly. We need to ensure that #IDENTIFIER is not scanned as
099                                // PREPROCESSOR_DIRECTIVE (which would contain all text until line end). This
100                                // would interfere with parameter stringification. This is really a scanner
101                                // issue and it should only be a problem if IDENTIFIER is in parameterNames.
102                                if (isStringificationToken(token)) {
103                                        this.replacementList.addAll(fixStringificationScanning(token));
104                                } else {
105                                        this.replacementList.add(token);
106                                }
107                        }
108                }
109                this.isFunctionMacro = isFunctionMacro;
110                this.parameterNames = CollectionUtils.asUnmodifiable(parameterNames);
111                this.hasVariadicParameter = hasVariadicParameter;
112                this.macroDeclarationLocation = macroDeclarationLocation;
113        }
114
115        /**
116         * Parses the given macro definition and generates a {@link MacroDefinition}
117         * object from it. The parameter should be the definition string except for the
118         * leading "#define ".
119         * <p>
120         * For example <code>parseMacroDefinition("FOO(x) x")</code>. The parameter must
121         * not be null. This never returns null.
122         */
123        public static MacroDefinition parseMacroDefinition(String macroDefinition, TextRegionLocation location) {
124                try {
125                        macroDefinition = macroDefinition.trim().replaceAll("\\\\\n|\t", " ");
126                        List<IToken> definitionTokens = NewCPreprocessingUtils.scanMacroContent(macroDefinition);
127                        List<IToken> definitionTokensWithoutComments = CollectionUtils.filter(definitionTokens,
128                                        PredicateUtils.not(TokenUtils::isCommentToken));
129
130                        IToken macroNameToken = definitionTokensWithoutComments.get(0);
131                        if (definitionTokensWithoutComments.size() == 1) {
132                                // simple object-macro define without replacement list
133                                return new MacroDefinition(macroNameToken.getText(), "", false, Collections.emptyList(), false,
134                                                location);
135                        }
136                        // for a function macro, '(' must be directly (without space) behind macro name
137                        boolean isFunctionMacro = definitionTokensWithoutComments.get(1).getType() == ETokenType.LPAREN
138                                        && definitionTokensWithoutComments.get(1).getOffset() == macroNameToken.getEndOffset() + 1;
139                        if (isFunctionMacro) {
140                                return parseFunctionMacroDefinition(definitionTokens, definitionTokensWithoutComments, location);
141                        }
142                        // this is an object macro (no parameters)
143                        return parseObjectMacro(definitionTokens, definitionTokensWithoutComments, location);
144                } catch (AssertionError e) {
145                        throw new AssertionError("Error while parsing macro: " + macroDefinition, e);
146                }
147        }
148
149        /**
150         * Parses an object macro definition. That means the char directly behind the
151         * macro name must not be a '('.
152         */
153        private static MacroDefinition parseObjectMacro(List<IToken> definitionTokens,
154                        List<IToken> definitionTokensWithoutComments, TextRegionLocation location) {
155                int replacementListStartIndex = 1; // first token after macroNameToken
156                List<IToken> replacementList = Collections.emptyList();
157                if (replacementListStartIndex < definitionTokensWithoutComments.size()) {
158                        replacementList = definitionTokens.subList(
159                                        definitionTokens.indexOf(definitionTokensWithoutComments.get(replacementListStartIndex)),
160                                        definitionTokens.size());
161                }
162                IToken macroNameToken = definitionTokensWithoutComments.get(0);
163                // this is an object macro (no parameters)
164                return new MacroDefinition(macroNameToken.getText(),
165                                TokenStreamTextUtils.concatTokenTexts(replacementList, " "), false, Collections.emptyList(), false,
166                                location);
167        }
168
169        /**
170         * Parses a function macro definition. That means the char directly (without
171         * space) behind the macro name must be a '(' starting a parameter list.
172         */
173        private static MacroDefinition parseFunctionMacroDefinition(List<IToken> definitionTokens,
174                        List<IToken> definitionTokensWithoutComments, TextRegionLocation location) {
175                int replacementListStartIndex = TokenStreamUtils.findMatchingClosingToken(definitionTokensWithoutComments, 2,
176                                ETokenType.LPAREN, ETokenType.RPAREN) + 1;
177                CCSMAssert.isTrue(replacementListStartIndex > 0,
178                                "Did not find closing parenthesis in function macro definition.");
179
180                List<IToken> replacementList = Collections.emptyList();
181                if (replacementListStartIndex < definitionTokensWithoutComments.size()) {
182                        replacementList = definitionTokens.subList(
183                                        definitionTokens.indexOf(definitionTokensWithoutComments.get(replacementListStartIndex)),
184                                        definitionTokens.size());
185                }
186
187                Pair<List<String>, Boolean> parameterInfo = parseParameterList(
188                                definitionTokensWithoutComments.subList(2, replacementListStartIndex - 1));
189                List<String> parameterNames = parameterInfo.getFirst();
190                boolean hasVariadicParameter = parameterInfo.getSecond();
191                IToken macroNameToken = definitionTokensWithoutComments.get(0);
192                return new MacroDefinition(macroNameToken.getText(),
193                                TokenStreamTextUtils.concatTokenTexts(replacementList, " "), true, parameterNames, hasVariadicParameter,
194                                location);
195        }
196
197        /**
198         * Scans the given function-macro parameter list and returns (1) the names of
199         * the parameters and (2) whether the last parameter is a variadic parameter.
200         */
201        private static Pair<List<String>, Boolean> parseParameterList(List<IToken> parameterList) {
202                boolean hasVariadicParameter = false;
203                List<String> parameterNames = CollectionUtils.map(TokenStreamUtils.splitWithNesting(parameterList,
204                                ETokenType.COMMA, ETokenType.LPAREN, ETokenType.RPAREN), TokenStreamTextUtils::concatTokenTexts);
205                parameterNames.removeIf(String::isEmpty);
206
207                if (!parameterNames.isEmpty() && CollectionUtils.getLast(parameterNames).endsWith("...")) {
208                        hasVariadicParameter = true;
209                        String variadicParameterName = CollectionUtils.getLast(parameterNames);
210                        if (variadicParameterName.equals("...")) {
211                                parameterNames.set(parameterNames.size() - 1, "__VA_ARGS__");
212                        } else {
213                                parameterNames.set(parameterNames.size() - 1,
214                                                variadicParameterName.substring(0, variadicParameterName.length() - 3));
215                        }
216                }
217                return new Pair<>(parameterNames, hasVariadicParameter);
218        }
219
220        /**
221         * Returns whether the given token's text starts with # and is not one of the
222         * "common" preprocessor directives. If this returns true, we assume that the
223         * token is a "stringification" token in a macro replacement list (#xy will be
224         * expanded with the value of the parameter xy).
225         */
226        public static boolean isStringificationToken(IToken token) {
227                String tokenText = token.getText();
228                return tokenText.startsWith("#") && tokenText.length() > 1 && !StringUtils.startsWithOneOf(tokenText, "##",
229                                "#define", "#undef", "#line", "#pragma", "#assert", "#error", "#if", "#endif", "#include");
230        }
231
232        /**
233         * Assumes that the given token has been classified as preprocessor-directive
234         * token but is actually a stringification token in a macro replacement list.
235         * <p>
236         * This splits the token and returns the correct components. For example, for
237         * the token with text "#xy foo CALL( bar)" returns tokens with texts "#xy",
238         * "foo", "CALL", "(", "bar", ")".
239         * 
240         * Special case: for the token with text "#foo #bar #baz", we fix all tokens (we
241         * rescan the result after fixing the first and then fix the resulting tail
242         * tokens too, we return "#foo", "#bar", "#baz").
243         */
244        private static List<? extends IToken> fixStringificationScanning(IToken token) {
245                List<IToken> result = new ArrayList<>();
246                Deque<IToken> tokensToFix = new ArrayDeque<>(Collections.singletonList(token));
247                while (!tokensToFix.isEmpty()) {
248                        IToken currentToken = tokensToFix.poll();
249                        if (!isStringificationToken(currentToken)) {
250                                result.add(currentToken);
251                                continue;
252                        }
253                        String text = currentToken.getText();
254                        // we know the token text starts with #
255                        int indexOfNonWhitespace = 1;
256                        while (indexOfNonWhitespace < text.length() && text.charAt(indexOfNonWhitespace) == ' ') {
257                                indexOfNonWhitespace++;
258                        }
259                        int indexOfNonIdentifierChar = StringUtils.indexOfMatch(text.substring(indexOfNonWhitespace),
260                                        NO_CPP_IDENTIFIER_CHAR_PATTERN) + indexOfNonWhitespace;
261                        if (indexOfNonIdentifierChar < indexOfNonWhitespace) {
262                                result.add(currentToken);
263                        } else {
264                                result.add(NewCPreprocessingUtils.scanMacroContent(text.substring(0, indexOfNonIdentifierChar)).get(0));
265                                NewCPreprocessingUtils.scanMacroContent(text.substring(indexOfNonIdentifierChar))
266                                                .forEach(tokensToFix::add);
267                        }
268                }
269                return result;
270        }
271
272        /**
273         * Returns all parameter names that need to be expanded recursively during
274         * expansion of this macro. If a parameter is only used in stringification, it
275         * is not contained in the result (does not need to be expanded).
276         */
277        public List<String> getParametersThatRequireExpansion() {
278                return replacementList.stream().map(IToken::getText).filter(parameterNames::contains).distinct()
279                                .collect(Collectors.toList());
280        }
281
282        @Override
283        public String toString() {
284                if (isFunctionMacro) {
285                        String parameterList = "";
286                        // string concatenation in loop is ok since this is always a very short loop
287                        // (there are never hundreds of parameters)
288                        for (int i = 0; i < parameterNames.size(); i++) {
289                                if (i != 0) {
290                                        parameterList += ", ";
291                                }
292                                String parameter = parameterNames.get(i);
293                                boolean isLastAndVariadicParameter = hasVariadicParameter && i == parameterNames.size() - 1;
294                                if (isLastAndVariadicParameter && parameter.equals("__VA_ARGS__")) {
295                                        parameterList += "...";
296                                } else if (isLastAndVariadicParameter) {
297                                        parameterList += parameter + "...";
298                                } else {
299                                        parameterList += parameter;
300                                }
301                        }
302                        return "#define " + macroName + "(" + parameterList + ") "
303                                        + TokenStreamTextUtils.concatTokenTexts(replacementList, " ");
304                }
305                return "#define " + macroName + " " + TokenStreamTextUtils.concatTokenTexts(replacementList, " ");
306        }
307
308        @Override
309        public boolean equals(Object o) {
310                if (this == o) {
311                        return true;
312                }
313                if (o == null || getClass() != o.getClass()) {
314                        return false;
315                }
316                MacroDefinition that = (MacroDefinition) o;
317                return isFunctionMacro == that.isFunctionMacro && hasVariadicParameter == that.hasVariadicParameter
318                                && macroName.equals(that.macroName) && replacementList.equals(that.replacementList)
319                                && parameterNames.equals(that.parameterNames)
320                                && macroDeclarationLocation.equals(that.macroDeclarationLocation);
321        }
322
323        @Override
324        public int hashCode() {
325                return Objects.hash(macroName, replacementList, isFunctionMacro, parameterNames, hasVariadicParameter,
326                                macroDeclarationLocation);
327        }
