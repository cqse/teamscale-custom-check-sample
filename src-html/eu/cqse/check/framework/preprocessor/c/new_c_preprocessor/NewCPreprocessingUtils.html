018/**
019 * Util methods for {@link NewCPreprocessor}.
020 */
021public class NewCPreprocessingUtils {
022
023        /**
024         * Scans the given content as {@link ELanguage#CPP} code. All generated tokens
025         * will have {@link MacroHandlingCPreprocessorBase#MACRO_ORIGIN} origin.
026         */
027        public static List<IToken> scanMacroContent(String content) {
028                // We could use caching as in the old MacroHandlingCPreprocessorBase. Probably
029                // not necessary since we store macros already as token stream.
030                return CollectionUtils.asUnmodifiable(
031                                ScannerUtils.getTokens(content, ELanguage.CPP, MacroHandlingCPreprocessorBase.MACRO_ORIGIN));
032        }
033
034        /**
035         * Returns a string containing of spaces. Used for log-message alignment. The
036         * number of spaces is such that if appended to the given logMessage, the result
037         * will have exactly maxAssumedMessageLength. Returns the empty string if called
038         * with a longer message.
039         */
040        @Nonnull
041        public static String getFillerSpaces(String logMessage, int maxAssumedMessageLength) {
042                int numFillerSpaces = (maxAssumedMessageLength + 1) - logMessage.length();
043                if (numFillerSpaces <= 0) {
044                        // message is longer than normal; we still display it, but formatting will be
045                        // more difficult to read.
046                        return "";
047                }
048                return StringUtils.repeat(" ", (maxAssumedMessageLength + 1) - logMessage.length());
049        }
050
051        /**
052         * Computes a new List of tokens which consists of the given tokens but applies
053         * the given replacements.
054         * 
055         * @param replacements
056         *            The replacements to be applied. This list must be sorted by
057         *            {@link PreprocessorTokenReplacement#originalTokensStartIndex} and
058         *            the replaced regions must be non-overlapping.
059         * @param tokens
060         *            the original tokens which the replacements refer to (i.e.
061         *            {@link PreprocessorTokenReplacement#originalTokensStartIndex}).
062         *            This list is not modified.
063         * @return A new list of tokens after applying the replacements.
064         */
065        public static List<IToken> applyReplacements(List<PreprocessorTokenReplacement> replacements, List<IToken> tokens) {
066                if (replacements.isEmpty()) {
067                        return tokens;
068                }
069                Iterator<PreprocessorTokenReplacement> replacementsIterator = replacements.iterator();
070                PreprocessorTokenReplacement currentReplacement = replacementsIterator.next();
071                List<IToken> result = new ArrayList<>();
072                for (int i = 0; i < tokens.size(); i++) {
073                        while (currentReplacement.countReplacedTokens() == 0 && replacementsIterator.hasNext()) {
074                                currentReplacement = replacementsIterator.next();
075                        }
076                        if (i >= currentReplacement.originalTokensEndIndex) {
077                                if (replacementsIterator.hasNext()) {
078                                        currentReplacement = replacementsIterator.next();
079                                }
080                        }
081                        if (i < currentReplacement.originalTokensStartIndex || i >= currentReplacement.originalTokensEndIndex) {
082                                result.add(tokens.get(i));
083                        } else if (i == currentReplacement.originalTokensStartIndex) {
084                                result.addAll(currentReplacement.replacementTokens);
085                                // (optimization) fast forward
086                                i = currentReplacement.originalTokensEndIndex - 1;
087                        }
088                }
089                return result;
090        }
091
092        /**
093         * Generates a {@link TextRegionLocation} from the given token.
094         */
095        /* package */static TextRegionLocation locationFromToken(IToken token) {
096                return new TextRegionLocation(token.getOriginId(), token.getOriginId(), token.getOffset(), token.getEndOffset(),
097                                token.getLineNumber(), token.getLineNumber() + StringUtils.countLines(token.getText()));
098        }
099
100        /**
101         * Returns a new list containing both given lists concatenated.
102         */
103        /* package */static List<IToken> concat(List<IToken> list1, List<IToken> list2) {
104                List<IToken> result = new ArrayList<>();
105                result.addAll(list1);
106                result.addAll(list2);
107                return result;
108        }
109}
