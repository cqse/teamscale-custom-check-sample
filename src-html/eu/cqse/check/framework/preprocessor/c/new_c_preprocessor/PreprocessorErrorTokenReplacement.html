001package eu.cqse.check.framework.preprocessor.c.new_c_preprocessor;
002
003import java.util.Collections;
004import java.util.List;
005
006import org.conqat.lib.commons.collections.UnmodifiableList;
007
008import eu.cqse.check.framework.scanner.IToken;
009
010/**
011 * {@link PreprocessorTokenReplacement} that also contains information about one
012 * or more errors encountered during preprocessing (e.g., too few parameters for
013 * a function macro).
014 */
015public class PreprocessorErrorTokenReplacement extends PreprocessorTokenReplacement {
016        /**
017         * A message that explains why we generate an error replacement.
018         */
019        public final String errorMessage;
020        /**
021         * The hierarchy of #include statements that were currently processed when this
022         * error occurred.
023         */
024        public final UnmodifiableList<String> includeHierarchy;
025
026        /* package */ PreprocessorErrorTokenReplacement(String errorMessage, UnmodifiableList<String> includeHierarchy,
027                        List<IToken> replacementTokens, int originalTokensStartIndex, int originalTokensEndIndex,
028                        NewCPreprocessor.PreprocessorUsageInformation preprocessorUsageInformation) {
029                super(replacementTokens, originalTokensStartIndex, originalTokensEndIndex, preprocessorUsageInformation);
030                this.errorMessage = errorMessage;
031                this.includeHierarchy = includeHierarchy;
032        }
033
034        /* package */ static PreprocessorErrorTokenReplacement createSingleTokenRemoval(String errorMessage,
035                        UnmodifiableList<String> includeHierarchy, int tokenIndex,
036                        NewCPreprocessor.PreprocessorUsageInformation preprocessorUsageInformation) {
037                return new PreprocessorErrorTokenReplacement(errorMessage, includeHierarchy, Collections.emptyList(),
038                                tokenIndex, tokenIndex + 1, preprocessorUsageInformation);
039        }
040}
