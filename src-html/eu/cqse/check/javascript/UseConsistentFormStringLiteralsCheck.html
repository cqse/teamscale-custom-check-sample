017package eu.cqse.check.javascript;
018
019import java.util.EnumSet;
020import java.util.List;
021
022import eu.cqse.check.base.EntityCheckBase;
023import eu.cqse.check.framework.core.ACheck;
024import eu.cqse.check.framework.core.CheckException;
025import eu.cqse.check.framework.core.CheckGroupNames;
026import eu.cqse.check.framework.core.ECheckParameter;
027import eu.cqse.check.framework.core.EFindingEnablement;
028import eu.cqse.check.framework.core.option.ACheckOption;
029import eu.cqse.check.framework.scanner.ELanguage;
030import eu.cqse.check.framework.scanner.ETokenType;
031import eu.cqse.check.framework.scanner.IToken;
032import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
033import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
034
035/**
036 * This work is a step to encourage developers to maintain a consistent use of
037 * either single or double-quoted string literals in Javascript. The check is
038 * configurable so it can be turned on or off.
039 */
040@ACheck(name = "Use consistent form for String literals", description = "Use consistent form for String literals, i.e. only one of '' or \"\"", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
041                ELanguage.JAVASCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
042public class UseConsistentFormStringLiteralsCheck extends EntityCheckBase {
043
044        /** {@ConQAT.Doc} */
045        @ACheckOption(name = "Use Single-Quoted String Literals", description = "Use Single-Quoted String Literals in Javascript code")
046        private boolean useSingleQuotedLiterals = true;
047
048        /** The default literal quotes for Javascript. */
049        private String quotationCharacter = "'";
050
051        /** The default finding message when checking for single-quoted literals */
052        private String findingMessage = "Use Single-Quoted String Literals";
053
054        /** {@inheritDoc} */
055        @Override
056        protected String getXPathSelectionString() {
057                return "//STATEMENT";
058        }
059
060        /** {@inheritDoc} */
061        @Override
062        public void initialize() {
063                if (!useSingleQuotedLiterals) {
064                        quotationCharacter = "\"";
065                        findingMessage = "Use Double-Quoted String Literals";
066                }
067        }
068
069        /** {@inheritDoc} */
070        @Override
071        protected void processEntity(ShallowEntity entity) throws CheckException {
072                List<IToken> tokens = entity.ownStartTokens();
073                List<Integer> indices = TokenStreamUtils.findAll(tokens, EnumSet.of(ETokenType.STRING_LITERAL));
074                for (int index : indices) {
075                        IToken token = tokens.get(index);
076                        if (!token.getText().startsWith(quotationCharacter)) {
077                                createFinding(findingMessage, token);
078                        }
079                }
080        }
081
082}
