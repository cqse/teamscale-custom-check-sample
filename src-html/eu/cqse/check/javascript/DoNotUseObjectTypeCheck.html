008package eu.cqse.check.javascript;
009
010import static org.conqat.lib.commons.string.StringUtils.countLines;
011
012import java.util.regex.Matcher;
013import java.util.regex.Pattern;
014
015import eu.cqse.check.base.CommentCheckBase;
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.core.CheckGroupNames;
019import eu.cqse.check.framework.core.ECheckParameter;
020import eu.cqse.check.framework.core.EFindingEnablement;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.scanner.IToken;
023
024/**
025 * {@ConQAT.Doc}
026 */
027@ACheck(name = "Avoid using the 'Object' type", description = "The `Object` type in annotations is very generic and should be replaced with either a custom `@typedef` definition or the `{name1 : valueType1, name2 : valueType2}` syntax.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
028                ELanguage.JAVASCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class DoNotUseObjectTypeCheck extends CommentCheckBase {
030
031        /**
032         * Regex matching 'Object' annotations inside @param and @type tags,
033         * e.g. @param{Object}, @type{string|Object}, but not @type{Object<string,
034         * string}.
035         */
036        private static final Pattern OBJECT_TYPE_REGEX = Pattern
037                        .compile("@\\s*(param|type)\\s*\\{(.*[^\\w])?Object([^\\.<](.*))?\\}");
038
039        /** {@inheritDoc} */
040        @Override
041        protected void processComment(IToken token, int startLine, int endLine) throws CheckException {
042                String comment = token.getText();
043                Matcher commentMatcher = OBJECT_TYPE_REGEX.matcher(comment);
044
045                while (commentMatcher.find()) {
046                        int offsetWithinComment = commentMatcher.start();
047                        createFinding("Avoid the `Object` type in type annotations",
048                                        startLine + countLines(comment.substring(0, offsetWithinComment)) - 1);
049                }
050        }
051}
