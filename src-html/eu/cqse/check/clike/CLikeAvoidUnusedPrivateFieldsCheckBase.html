008package eu.cqse.check.clike;
009
010import java.util.List;
011
012import eu.cqse.check.framework.scanner.IToken;
013import eu.cqse.check.framework.shallowparser.SubTypeNames;
014import eu.cqse.check.framework.util.CLikeLanguageFeatureParserBase;
015
016/**
017 * Base class for checks detecting unused private fields in clike languages.
018 * 
019 * @author $Author: heinemann $
020 * @version $Rev: 55643 $
021 * @ConQAT.Rating YELLOW Hash: 0909166B1FBD15E4A838B1FBC4A644AC
022 */
023public abstract class CLikeAvoidUnusedPrivateFieldsCheckBase
024                extends CLikeFieldUsageCheckBase {
025
026        /** Description of this check. */
027        public static final String DESCRIPTION = "A private field is declared but never read.";
028
029        /** Constructor. */
030        public CLikeAvoidUnusedPrivateFieldsCheckBase(
031                        CLikeLanguageFeatureParserBase languageFeatureParser) {
032                super(languageFeatureParser);
033        }
034
035        /** {@inheritDoc} */
036        @Override
037        protected String getFieldSelectionXPath() {
038                return "//ATTRIBUTE[subtype('" + SubTypeNames.ATTRIBUTE
039                                + "') and modifiers('private')]";
040        }
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getFindingMessage(String fieldName) {
045                return "Private field '" + fieldName + "' is never read.";
046        }
047
048        /** {@inheritDoc} */
049        @Override
050        protected boolean analyzeTokens(List<IToken> tokens, String fieldName,
051                        boolean isShadowed) {
052                return !languageFeatureParser
053                                .getVariableReadsFromTokens(tokens, fieldName, true, isShadowed)
054                                .isEmpty();
055        }
056}
