003import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.CASE_LABEL_CLAUSE_LIST;
004import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.ERROR;
005import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.EXIT_STATEMENT;
006import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT;
007import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.FINAL_DEFAULT_CLAUSE_LIST;
008import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.FINAL_SWITCH_CLAUSE;
009import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.FIRST_CASE_LABEL_CLAUSE_LIST;
010import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.SWITCH_CLAUSE;
011import static eu.cqse.check.cpp.misra.switches.WellFormedCheckVisitor.ESwitchNodeTypes.SWITCH_STATEMENT;
012import static eu.cqse.check.framework.shallowparser.SubTypeNames.ANONYMOUS_BLOCK;
013import static eu.cqse.check.framework.shallowparser.SubTypeNames.CASE;
014import static eu.cqse.check.framework.shallowparser.SubTypeNames.DEFAULT;
015import static eu.cqse.check.framework.shallowparser.SubTypeNames.LOOP_SUB_TYPE_NAMES;
016import static eu.cqse.check.framework.shallowparser.SubTypeNames.SWITCH;
017import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.META;
018import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.STATEMENT;

020import java.util.EnumSet;

022import org.conqat.lib.commons.assertion.CCSMAssert;

024import eu.cqse.check.framework.shallowparser.SubTypeNames;
025import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
026import eu.cqse.check.framework.shallowparser.framework.IShallowEntityVisitor;
027import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
028import eu.cqse.check.framework.util.CppLanguageFeatureParser;

030/**
031 * Checks whether a switch statement is well formed. For detailed syntax rules
032 * please look for rule 16.1 in the MISRA-C guideline. Does not check for
033 * missing default clause, since this is the responsibility of
034 * {@link eu.cqse.check.clike.SwitchMustHaveDefaultCheck} check<br>
035 * <br>
036 * SWITCH_STATEMENT -> FIRST_CASE_LABEL_CLAUSE_LIST -> SWITCH_CLAUSE ->
037 * EXIT_STATEMENT -> (CASE_LABEL_CLAUSE_LIST -> SWITCH_CLAUSE ->
038 * EXIT_STATEMENT)* -> FINAL_DEFAULT_CLAUSE_LIST -> FINAL_SWITCH_CLAUSE ->
039 * FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT <br>
040 * <br>
041 * SWITCH_STATEMENT: 'switch' entity <br>
042 * FIRST_CASE_LABEL_CLAUSE_LIST and CASE_LABEL_CLAUSE_LIST: at least one 'case'
043 * label <br>
044 * SWITCH_CLAUSE and FINAL_SWITCH_CLAUSE: optional sequence of Statements <br>
045 * EXIT_STATEMENT and FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT:
046 * {@link eu.cqse.check.framework.util.CppLanguageFeatureParser#isCaseOrDefaultExitStatement(ShallowEntity)
047 * Exit Statement} <br>
048 * FINAL_DEFAULT_CLAUSE_LIST: optional list of leading 'case' labels and one
049 * 'default' label
050 */
051public class WellFormedCheckVisitor implements IShallowEntityVisitor {
052
053        /**
054         * Different types of nodes for keeping track while traversing and recognizing
055         * errors by non valid transitions
056         */
057        enum ESwitchNodeTypes {
058                SWITCH_STATEMENT,
059                FIRST_CASE_LABEL_CLAUSE_LIST,
060                SWITCH_CLAUSE,
061                EXIT_STATEMENT,
062                CASE_LABEL_CLAUSE_LIST,
063                FINAL_DEFAULT_CLAUSE_LIST,
064                FINAL_SWITCH_CLAUSE,
065                FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT,
066                ERROR;
067        }
068
069        /** @see ESwitchNodeTypes */
070        ESwitchNodeTypes actualNodeType;
071
072        /**
073         * Error message, which is used for findings creation. Default text is used if
074         * no error occurred, but some part is missing.
075         */
076        private String errorMessage = "`switch` statement is incomplete";
077
078        /**
079         * The current entity, which is handled. If an entity is not handled (loops)
080         * currentEntity is not changed. If an error occurred currentEntity is not
081         * changed any longer, because not any following entity is handled and hence
082         * could be used as findings location.
083         */
084        private ShallowEntity currentEntity;
085
086        /**
087         * Stores whether or not the last statement encountered in a case or default
088         * block is a return statement
089         */
090        private boolean lastStatementHitInCaseOrDefaultBlockIsReturn = false;
091
092        /** {@inheritDoc} */
093        @Override
094        public boolean visit(ShallowEntity entity) {
095                if (hasError() || LOOP_SUB_TYPE_NAMES.contains(entity.getSubtype())) {
096                        return false;
097                }
098                currentEntity = entity;
099
100                if (isSwitch(entity)) {
101                        actualNodeType = SWITCH_STATEMENT;
102                } else if (isCase(entity)) {
103                        handleCase();
104                        lastStatementHitInCaseOrDefaultBlockIsReturn = false;
105                } else if (isDefault(entity)) {
106                        handleDefault();
107                        lastStatementHitInCaseOrDefaultBlockIsReturn = false;
108                } else if (CppLanguageFeatureParser.isCaseOrDefaultExitStatement(entity) && !isInLambda(entity)
109                                && isSwitch(entity.getParent())) {
110                        handleExitStatement();
111                } else if (!entity.getSubtype().equals(ANONYMOUS_BLOCK)) {
112                        String shallowEntityName = entity.getName();
113                        lastStatementHitInCaseOrDefaultBlockIsReturn = shallowEntityName != null
114                                        && shallowEntityName.equals(SubTypeNames.RETURN.toString());
115                        handleStatements();
116                }
117                return entity.hasChildren();
118        }
119
120        /** {@inheritDoc} */
121        @Override
122        public void endVisit(ShallowEntity entity) {
123                // We expect that there is a last default statement and so we check if there is
124                // a last return statement at end of traversing a switch statement
125                if (isSwitch(entity) && actualNodeType == FINAL_SWITCH_CLAUSE && lastStatementHitInCaseOrDefaultBlockIsReturn) {
126                        actualNodeType = FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT;
127                }
128        }
129
130        /** Handles appearing case entity */
131        private void handleCase() {
132                String findingMessage = "Expecting an exit statement before `case` clause";
133                switch (actualNodeType) {
134                case SWITCH_STATEMENT:
135                        actualNodeType = FIRST_CASE_LABEL_CLAUSE_LIST;
136                        break;
137                case EXIT_STATEMENT:
138                        actualNodeType = CASE_LABEL_CLAUSE_LIST;
139                        break;
140                case FIRST_CASE_LABEL_CLAUSE_LIST:
141                case CASE_LABEL_CLAUSE_LIST:
142                        break;
143                case SWITCH_CLAUSE:
144                        if (lastStatementHitInCaseOrDefaultBlockIsReturn) {
145                                lastStatementHitInCaseOrDefaultBlockIsReturn = false;
146                                actualNodeType = CASE_LABEL_CLAUSE_LIST;
147                                break;
148                        }
149
150                        error(findingMessage);
151                        break;
152                case FINAL_DEFAULT_CLAUSE_LIST:
153                case FINAL_SWITCH_CLAUSE:
154                case FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT:
155                default:
156                        error(findingMessage);
157                        break;
158                }
159        }
160
161        /** Handles appearing default entity */
162        private void handleDefault() {
163                String findingMessage = "Expecting an exit statement before `default` clause";
164                switch (actualNodeType) {
165                case CASE_LABEL_CLAUSE_LIST:
166                case EXIT_STATEMENT:
167                        actualNodeType = FINAL_DEFAULT_CLAUSE_LIST;
168                        break;
169
170                case SWITCH_CLAUSE:
171                        if (lastStatementHitInCaseOrDefaultBlockIsReturn) {
172                                lastStatementHitInCaseOrDefaultBlockIsReturn = false;
173                                actualNodeType = CASE_LABEL_CLAUSE_LIST;
174                                break;
175                        }
176
177                        error(findingMessage);
178                        break;
179                case SWITCH_STATEMENT:
180                case FIRST_CASE_LABEL_CLAUSE_LIST:
181                case FINAL_DEFAULT_CLAUSE_LIST:
182                case FINAL_SWITCH_CLAUSE:
183                case FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT:
184                default:
185                        error(findingMessage);
186                        break;
187                }
188        }
189
190        /** Handles appearing exit statement */
191        private void handleExitStatement() {
192                switch (actualNodeType) {
193                case FIRST_CASE_LABEL_CLAUSE_LIST:
194                case CASE_LABEL_CLAUSE_LIST:
195                case SWITCH_CLAUSE:
196                        actualNodeType = EXIT_STATEMENT;
197                        break;
198                case FINAL_SWITCH_CLAUSE:
199                case FINAL_DEFAULT_CLAUSE_LIST:
200                        actualNodeType = FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT;
201                        break;
202                case EXIT_STATEMENT:
203                case SWITCH_STATEMENT:
204                case FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT:
205                default:
206                        error("Unnecessary `exit` statement");// Redundant exit statement
207                        break;
208                }
209        }
210
211        /** Handles appearing non exit statement */
212        private void handleStatements() {
213                switch (actualNodeType) {
214                case FIRST_CASE_LABEL_CLAUSE_LIST:
215                case CASE_LABEL_CLAUSE_LIST:
216                        actualNodeType = SWITCH_CLAUSE;
217                        break;
218                case FINAL_DEFAULT_CLAUSE_LIST:
219                        actualNodeType = FINAL_SWITCH_CLAUSE;
220                        break;
221                case SWITCH_CLAUSE:
222                case FINAL_SWITCH_CLAUSE:
223                        break;
224                case EXIT_STATEMENT:
225                case SWITCH_STATEMENT:
226                case FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT:
227                default:
228                        error("Expecting an exit statement at end of `case` or `default` clause"); // unexpected statement
229                        break;
230                }
231        }
232
233        /** Sets {@link ESwitchNodeTypes#ERROR} status */
234        private void error(String message) {
235                CCSMAssert.isNotEmpty(message, "The error message must be defined");
236                this.errorMessage = message;
237                this.actualNodeType = ERROR;
238        }
239
240        private static boolean isSwitch(ShallowEntity entity) {
241                return entity.getType() == STATEMENT && entity.getSubtype().equals(SWITCH);
242        }
243
244        private static boolean isDefault(ShallowEntity entity) {
245                return entity.getType() == META && entity.getSubtype().equals(DEFAULT);
246        }
247
248        private static boolean isCase(ShallowEntity entity) {
249                return entity.getType() == META && entity.getSubtype().equals(CASE);
250        }
251
252        /** Checks if given shallow entity is inside lambda definition. */
253        private static boolean isInLambda(ShallowEntity entity) {
254                ShallowEntity parent = entity.getParent();
255                return parent.getType().equals(EShallowEntityType.METHOD)
256                                && entity.getParent().getSubtype().equals(SubTypeNames.LAMBDA_EXPRESSION);
257
258        }
259
260        /** Returns whether the visitor is in a valid end status. */
261        boolean isInExitState() {
262                return EnumSet.of(FINAL_DEFAULT_CLAUSE_EXIT_STATEMENT, EXIT_STATEMENT).contains(actualNodeType);
263        }
264
265        /** Returns whether the visitor is in an error state */
266        private boolean hasError() {
267                return actualNodeType == ERROR;
268        }
269
270        /** @see #errorMessage */
271        protected String getErrorMessage() {
272                return errorMessage;
273        }
274
275        /** @see #currentEntity */
276        protected ShallowEntity getCurrentEntity() {
277                return currentEntity;
278        }
