033/**
034 * Precondition for test completeness : Input is the fully preprocessed
035 * compilation unit. Vocabulary describing the different lexemes encountered
036 * stems from {@linkplain <a href=
037 * "https://en.cppreference.com/w/cpp/language/declarations">cpp reference</a>}
038 */
039@ACheck(name = "Consistent Storage Class Modifiers", description = "If at its first declaration an object is declared with internal linkage (using the storage class modifier `static`), this takes precedence over any later declarations with external linkage. To avoid false assumptions, all declarations should specify the same linkage.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
040                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
041@ACheckRuleset(ruleId = "Rule 8.8", ruleset = MISRA)
042public class ConsistentStorageClassModifierCheck extends EntityCheckBase {
043
044        /**
045         * mapping of object names to the Entity where they first occur, and a Boolean
046         * specifying whether they have internal linkage
047         */
048        private Map<String, Pair<ShallowEntity, Boolean>> nameToFirstOccurenceXLinkage;
049
050        @Override
051        protected ECodeViewOption getCodeViewOption() {
052                return ECodeViewOption.FILTERED_PREPROCESSED;
053        }
054
055        /** {@inheritDoc} */
056        @Override
057        public void execute() throws CheckException {
058                nameToFirstOccurenceXLinkage = new HashMap<>();
059                super.execute();
060        }
061
062        /** {@inheritDoc} */
063        @Override
064        protected String getXPathSelectionString() {
065                // the use of one '/'' here is intentional - linkage in C is explicitly
066                // specified
067                // only for objects at global scope - that is, immediately below the root entity
068                return "/ATTRIBUTE|/METHOD";
069        }
070
071        /** {@inheritDoc} */
072        @Override
073        protected void processEntity(ShallowEntity entity) throws CheckException {
074                // need to distinguish between simple declarations and function definitions
075                if (entity.getType() == EShallowEntityType.ATTRIBUTE || (entity.getType() == EShallowEntityType.METHOD
076                                && entity.getSubtype().equals(SubTypeNames.FUNCTION_DECLARATION))) {
077                        processSimpleDeclaration(entity);
078                        // The shallow parser parses an entire simple declaration as either an
079                        // ATTRIBUTE[global variable]
080                        // or METHOD[method declaration], depending on what occurs (first)
081                        // in the init-declarator-list
082                } else if (entity.getSubtype().equals(SubTypeNames.FUNCTION)) {
083                        processFunctionDefinition(entity);
084                }
085
086        }
087
088        /**
089         * Function definitions are at the same time declarations
090         */
091        private void processFunctionDefinition(ShallowEntity entity) throws CheckException {
092                checkConsistentLinkage(entity.getName(), specifyInternalLinkage(entity.ownStartTokens()), entity);
093        }
094
095        /**
096         * whether the given tokens (which should belong to a
097         * declarator-specifier-sequence) specify internal linkage
098         */
099        private static boolean specifyInternalLinkage(List<IToken> tokens) {
100                return TokenStreamUtils.contains(tokens, ETokenType.STATIC);
101        }
102
103        /**
104         * Simple declarations are of the form S D1, D2, â€¦, Dn; where S is the
105         * declarator-specifier-sequence (decl-specifier-seq), Di are all
106         * initializer/declarators (init-declarator)s.
107         */
108        private void processSimpleDeclaration(ShallowEntity entity) throws CheckException {
109                List<List<IToken>> initializerDeclaratorList = TokenStreamUtils.splitWithNesting(entity.ownStartTokens(),
110                                ETokenType.COMMA, Arrays.asList(ETokenType.LPAREN, ETokenType.LBRACE),
111                                Arrays.asList(ETokenType.RPAREN, ETokenType.RBRACE));
112                // The first entry in our list will contain the decl-specifier-seq as well as
113                // the first init-declarator,
114                List<IToken> declaratorSpecifierSequence = initializerDeclaratorList.get(0);
115                // not using the static keyword assumes a default extern linkage. In cpp
116                // redeclaration without a specifier
117                // means keeping any prior explicit specification, so this is only true for C
118                boolean internalLinkage = specifyInternalLinkage(declaratorSpecifierSequence); // the decl-specifier-seq
119                                                                                                                                                                                // properties apply
120                // to all declarators.
121                // the ShallowEntity name provided is that of the first object declared
122                checkConsistentLinkage(entity.getName(), internalLinkage, entity);
123                // Process the rest of the init-declarator-list
124                for (List<IToken> iTokenList : initializerDeclaratorList.subList(1, initializerDeclaratorList.size())) {
125                        // We use the heuristic that in an init-declarator, the leftmost 'identifier'
126                        // token
127                        // encountered is the name of the object being declared. This holds for C:
128                        // http://www.unixwiz.net/techtips/reading-cdecl.html, it wouldn't work in cpp
129                        // because it would cause issues with attributes or member pointer access.
130                        String name = iTokenList.get(TokenStreamUtils.firstTokenOfType(iTokenList, ETokenType.IDENTIFIER))
131                                        .getText();
132                        checkConsistentLinkage(name, internalLinkage, entity);
133                }
134        }
135
136        /**
137         * If name already exists in {@link #nameToFirstOccurenceXLinkage} with a
138         * different linkage, create a finding. Else add (name , (entity,linkage)) to
139         * the map.
140         */
141        private void checkConsistentLinkage(String name, boolean internalLinkage, ShallowEntity entity)
142                        throws CheckException {
143                if (nameToFirstOccurenceXLinkage.containsKey(name)) {
144                        Pair<ShallowEntity, Boolean> entityIsInternalPair = nameToFirstOccurenceXLinkage.get(name);
145                        if (!entityIsInternalPair.getSecond().equals(internalLinkage)) {
146                                createFindingForSiblingEntities("`" + name + "` has conflicting linkage declarations",
147                                                Arrays.asList(entityIsInternalPair.getFirst(), entity));
148                        }
149                } else {
150                        nameToFirstOccurenceXLinkage.put(name, new Pair<>(entity, internalLinkage));
151                }
152        }
153
