008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.List;
011
012import org.conqat.lib.commons.string.StringUtils;
013
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.CheckImplementationBase;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.core.option.ACheckOption;
021import eu.cqse.check.framework.core.phase.ECodeViewOption;
022import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
023import eu.cqse.check.framework.scanner.ELanguage;
024import eu.cqse.check.framework.scanner.ETokenType;
025import eu.cqse.check.framework.scanner.IToken;
026import eu.cqse.check.framework.shallowparser.SubTypeNames;
027import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
028import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
029import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
030import eu.cqse.check.framework.util.LanguageFeatureParser;
031import eu.cqse.check.framework.util.tokens.TokenPattern;
032import eu.cqse.check.framework.util.tokens.TokenPatternMatch;
033
034@ACheck(name = "Function types should be in prototype form", description = ""
035                + "All parameters in a function declaration should have a name.", groupName = CheckGroupNames.BAD_PRACTICE, //
036                defaultEnablement = EFindingEnablement.YELLOW, languages = { ELanguage.CPP }, parameters = {
037                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
038@ACheckRuleset(ruleId = "8.2", ruleset = MISRA)
039public class FunctionsInPrototypeFormCheck extends CheckImplementationBase {
040
041        /** {@ConQAT.Doc} */
042        @ACheckOption(name = "Force to explicitly specify void for functions without parameters", description = ""
043                        + "Enable this to check Misra conformance. In C a function with the following signature "
044                        + "'int foo()' takes an unspecified number of arguments. "
045                        + "To explicitly specify that the function takes no arguments void should be used eg. 'int foo(void)'. "
046                        + "In C++ this is not the case and enabling this check leads to false positives.")
047        private boolean checkVoidForEmptyParameters = false;
048
049        @Override
050        protected ECodeViewOption getCodeViewOption() {
051                return ECodeViewOption.FILTERED_PREPROCESSED;
052        }
053
054        /**
055         * Matches function pointers eg. int16_t ( *pf2_t ) ( int16_t ); Here it matches
056         * ( *pf2_t ) ( int16_t )
057         *
058         * The captured group 0 contains the parameters eg. (int16_t)
059         */
060        private static final TokenPattern FUNCTION_POINTER_PATTERN = new TokenPattern()
061                        .sequence(ETokenType.LPAREN, ETokenType.MULT, ETokenType.IDENTIFIER, ETokenType.RPAREN)
062                        .skipNested(ETokenType.LPAREN, ETokenType.RPAREN, false).group(0);
063
064        /** {@inheritDoc} */
065        @Override
066        public void execute() throws CheckException {
067                for (ShallowEntity entity : select(createSelectionPattern(EShallowEntityType.METHOD, SubTypeNames.FUNCTION,
068                                SubTypeNames.FUNCTION_DECLARATION, SubTypeNames.FUNCTION_POINTER_DECLARATION))) {
069                        processMethods(entity);
070                }
071                for (ShallowEntity entity : select(createSelectionPattern(EShallowEntityType.TYPE, SubTypeNames.TYPEDEF))) {
072                        processTypedefs(entity);
073                }
074        }
075
076        /**
077         * Handles type definitions of function pointers eg.
078         *
079         * typedef int16_t ( *pf2_t ) ( int16_t );
080         */
081        private void processTypedefs(ShallowEntity entity) throws CheckException {
082                TokenPatternMatch match = FUNCTION_POINTER_PATTERN.findFirstMatch(entity.ownStartTokens());
083                if (match == null) {
084                        return;
085                }
086                List<IToken> parameterTokens = TokenStreamUtils.tokensBetweenWithNesting(match.groupTokens(0),
087                                ETokenType.LPAREN, ETokenType.RPAREN);
088                processParameters(entity, LanguageFeatureParser.CPP.splitVariableTokens(parameterTokens));
089        }
090
091        private void processMethods(ShallowEntity entity) throws CheckException {
092                List<List<IToken>> parameters = LanguageFeatureParser.CPP.getSplitParameterTokens(entity);
093                processParameters(entity, parameters);
094        }
095
096        private void processParameters(ShallowEntity entity, List<List<IToken>> parameters) throws CheckException {
097                if (parameters.isEmpty() && checkVoidForEmptyParameters) {
098                        createFinding("Not in prototype form, missing ``void`` in parameter list", entity.ownStartTokens());
099                        return;
100                }
101                for (List<IToken> parameter : parameters) {
102                        if (parameter.isEmpty()) {
103                                continue;
104                        }
105                        ETokenType typeOfFirstToken = parameter.get(0).getType();
106                        if (typeOfFirstToken == ETokenType.VOID || typeOfFirstToken == ETokenType.ELLIPSIS) {
107                                return;
108                        }
109                        String type = LanguageFeatureParser.CPP.getVariableTypeFromTokens(parameter);
110                        IToken variableName = LanguageFeatureParser.CPP.getVariableNameFromTokens(parameter);
111                        if (variableName == null || StringUtils.isEmpty(type)) {
112                                createFinding("Not all parameter names are specified", entity.ownStartTokens());
113                                return;
114                        }
115                }
116        }
117
118}
