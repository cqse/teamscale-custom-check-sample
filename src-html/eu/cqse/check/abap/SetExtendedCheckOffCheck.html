006package eu.cqse.check.abap;
007
008import java.util.List;
009
010import eu.cqse.check.base.EntityCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.core.option.ACheckOption;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.SubTypeNames;
021import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023
024/**
025 * Checks whether SET EXTENDED CHECK [ON|OFF] is used.
026 */
027@ACheck(name = SetExtendedCheckOffCheck.CHECK_NAME, description = "The obsolote statement SET EXTENDED CHECK [OFF|ON] "
028                + "must not be used to deactivate the Extended Program Check for code regions. "
029                + "To ignore invalid findings, these should be put on the blacklist in Teamscale instead.", groupName = CheckGroupNames.FINDINGS_SUPPRESSION, languages = ELanguage.ABAP, defaultEnablement = EFindingEnablement.RED, parameters = {
030                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
031public class SetExtendedCheckOffCheck extends EntityCheckBase {
032
033        /** Name of this check */
034        public static final String CHECK_NAME = "SET EXTENDED CHECK OFF must not be used";
035
036        /** {@ConQAT.Doc} */
037        @ACheckOption(name = CHECK_NAME
038                        + " - also report SET EXTENDED CHECK ON", description = "If enabled, also any usage of SET EXTENDED CHECK ON is reported as finding, "
039                                        + "otherwise only SET EXTENDED CHECK OFF is reported.")
040        private boolean includeExtendedCheckOn = false;
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getXPathSelectionString() {
045                return "//STATEMENT[subtype('" + SubTypeNames.SET + "')]";
046
047        }
048
049        /** {@inheritDoc} */
050        @Override
051        protected void processEntity(ShallowEntity entity) throws CheckException {
052                List<IToken> tokens = entity.ownStartTokens();
053                if (TokenStreamUtils.startsWith(tokens, ETokenType.SET, ETokenType.EXTENDED, ETokenType.CHECK)) {
054                        if (tokens.size() < 4) {
055                                throw new CheckException("Incomplete SET EXETENDED CHECK [ON|OFF] statement found at " + entity
056                                                + ". Might be caused by a parser error.");
057                        }
058                        ETokenType switchValue = tokens.get(3).getType();
059                        if (switchValue == ETokenType.OFF) {
060                                createFinding("Usage of SET EXTENDED CHECK OFF", entity);
061                        }
062                        if (switchValue == ETokenType.ON && includeExtendedCheckOn) {
063                                createFinding("Usage of SET EXTENDED CHECK ON", entity);
064                        }
065                }
066        }
067}
