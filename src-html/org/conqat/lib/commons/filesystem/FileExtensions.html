018package org.conqat.lib.commons.filesystem;
019
020/**
021 * String constants for different file extensions.
022 */
023public class FileExtensions {
024
025        /** Extension for HTML files. */
026        public static final String HTML = ".html";
027
028        /** Extension for JavaScript files. */
029        public static final String JAVASCRIPT = ".js";
030
031        /** Extension for Soy/Closure template files. */
032        public static final String SOY = ".soy";
033
034        /** Extension for comma seperated value files. */
035        public static final String CSV = ".csv";
036
037        /** Extension for simple text files. */
038        public static final String TXT = ".txt";
039
040        /** Extension for LaTeX document class files. */
041        public static final String CLS = ".cls";
042
043        /** Extension for AUTOSAR ECU extract files. */
044        public static final String ECUEXTRACT = ".ecuextract";
045
046        /** Extension for AUTOSAR ECU extract variant files. */
047        public static final String ECUEXTRACT_VARIANT = ".ecuextract.variant";
048
049        /** Extension for AUTOSAR configuration XML files. */
050        public static final String ARXML = ".arxml";
051
052        /** Extension for AUTOSAR ECU configuration files. */
053        public static final String ECUCONFIG = ".ecuconfig";
054
055        /** Extension for source map files. */
056        public static final String MAP = ".map";
057
058        /** Extension for JSON files. */
059        public static final String JSON = ".json";
060}
