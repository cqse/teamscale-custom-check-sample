017package org.conqat.lib.commons.collections;
018
019import java.io.Serializable;
020import java.util.ArrayList;
021import java.util.Arrays;
022import java.util.Collection;
023import java.util.List;
024import java.util.regex.Pattern;
025
026import org.conqat.lib.commons.assertion.CCSMAssert;
027import org.conqat.lib.commons.clone.IDeepCloneable;
028
029/**
030 * A list of RegEx pattern. the reason for this name ("Basic") is that it
031 * contains slightly less functions that the version found in the ConQAT engine.
032 */
033public class BasicPatternList extends ArrayList<Pattern> implements IDeepCloneable, Serializable {
034
035        /** Version used for serialization. */
036        private static final long serialVersionUID = 1;
037
038        /** Creates an empty {@link BasicPatternList} */
039        public BasicPatternList() {
040                // do nothing
041        }
042
043        /** Creates a {@link BasicPatternList} with the collection of patterns */
044        public BasicPatternList(Collection<? extends Pattern> patterns) {
045                super(patterns);
046        }
047
048        /** Creates a pattern list for the specified patterns. */
049        public BasicPatternList(Pattern... patterns) {
050                this(Arrays.asList(patterns));
051        }
052
053        /**
054         * Returns true, if the pattern list is empty or {@link #matchesAny(String)}
055         * returns true.
056         */
057        public boolean emptyOrMatchesAny(String s) {
058                return isEmpty() || matchesAny(s);
059        }
060
061        /**
062         * Returns whether the given string matches at least one of the contained
063         * pattern. For this the <code>Matcher.matches()</code> method is used.
064         */
065        public boolean matchesAny(String s) {
066                for (Pattern p : this) {
067                        CCSMAssert.isNotNull(p, "Encountered null pattern!");
068
069                        if (p.matcher(s).matches()) {
070                                return true;
071                        }
072                }
073                return false;
074        }
075
076        /**
077         * Returns whether in the given string at least one of the contained pattern
078         * is found. For this the <code>Matcher.find()</code> method is used.
079         */
080        public boolean findsAnyIn(String s) {
081                for (Pattern p : this) {
082                        CCSMAssert.isNotNull(p, "Encountered null pattern!");
083
084                        if (p.matcher(s).find()) {
085                                return true;
086                        }
087                }
088                return false;
089        }
090
091        /** {@inheritDoc} */
092        @Override
093        public BasicPatternList deepClone() {
094                return new BasicPatternList(this);
095        }
096
097        /**
098         * Returns a list with the regular expressions from which the patters were
099         * compiled
100         */
101        public List<String> asStringList() {
102                List<String> patterns = new ArrayList<String>();
103                for (Pattern pattern : this) {
104                        patterns.add(pattern.pattern());
105                }
106                return patterns;
107        }
108}
