017package org.conqat.lib.commons.assertion;
018
019import java.util.Optional;
020
021import org.conqat.lib.commons.string.StringUtils;
022
023/**
024 * This class provides simple methods to implement assertions. Please refer to
025 * the {@linkplain org.conqat.lib.commons.assertion package documentation} for a
026 * discussion of assertions vs preconditions.
027 */
028public class CCSMAssert {
029
030        /**
031         * Checks if a condition is <code>true</code>.
032         * 
033         * @param condition
034         *            condition to check
035         * @param message
036         *            exception message
037         * @throws AssertionError
038         *             if the condition if <code>false</code>
039         */
040        public static void isTrue(boolean condition, String message) throws AssertionError {
041                if (!condition) {
042                        throw new AssertionError(message);
043                }
044        }
045
046        /**
047         * Checks if a condition is <code>false</code>.
048         * 
049         * @param condition
050         *            condition to check
051         * @param message
052         *            exception message
053         * @throws AssertionError
054         *             if the condition if <code>true</code>
055         */
056        public static void isFalse(boolean condition, String message) throws AssertionError {
057                if (condition) {
058                        throw new AssertionError(message);
059                }
060        }
061
062        /** Checks that the object is a instance of the class */
063        public static void isInstanceOf(Object object, Class<?> clazz) {
064                CCSMAssert.isTrue(clazz.isInstance(object), object + " must be instance of " + clazz);
065        }
066
067        /**
068         * This calls {@link #isInstanceOf(Object, Class)} and, if this doesn't fail
069         * returns the casted object.
070         */
071        @SuppressWarnings("unchecked")
072        public static <T> T checkedCast(Object object, Class<T> clazz) {
073                isInstanceOf(object, clazz);
074                return (T) object;
075        }
076
077        /**
078         * @throws AssertionError
079         *             with message
080         */
081        public static void fail(String message) throws AssertionError {
082                throw new AssertionError(message);
083        }
084
085        /**
086         * Checks whether a reference is <code>null</code>.
087         * 
088         * @param reference
089         *            reference to check
090         * @throws AssertionError
091         *             if the reference is <code>null</code>
092         */
093        public static void isNotNull(Object reference) throws AssertionError {
094                isNotNull(reference, "Reference must not be null");
095        }
096
097        /**
098         * Checks whether a reference is <code>null</code>.
099         * 
100         * @param reference
101         *            reference to check
102         * @param message
103         *            exception message
104         * @throws AssertionError
105         *             if the reference is <code>null</code>
106         */
107        public static void isNotNull(Object reference, String message) throws AssertionError {
108                if (reference == null) {
109                        throw new AssertionError(message);
110                }
111        }
112
113        /**
114         * Checks whether a String is <code>null</code> or empty.
115         * 
116         * @param s
117         *            String to check
118         * @param message
119         *            exception message
120         * @throws AssertionError
121         *             if the String is empty
122         */
123        public static void isNotEmpty(String s, String message) throws AssertionError {
124                if (StringUtils.isEmpty(s)) {
125                        throw new AssertionError(message);
126                }
127        }
128
129        /**
130         * Checks whether an Optional is present.
131         * 
132         * @param optional
133         *            Optional to check
134         * @param message
135         *            exception message
136         * @throws AssertionError
137         *             if the Optional is not present
138         */
139        public static void isPresent(Optional<?> optional, String message) {
140                if (!optional.isPresent()) {
141                        throw new AssertionError(message);
142                }
143        }
144
145}
