017package org.conqat.lib.commons.treemap;
018
019import java.awt.Color;
020import java.awt.geom.Rectangle2D;
021import java.util.List;
022
023/**
024 * Interface for nodes used for building the tree map node hierarchy which is
025 * then rendered as a tree map.
026 * 
027 * @author Benjamin Hummel
028 * 
029 * @param <T>
030 *            the type the user data has.
031 */
032public interface ITreeMapNode<T> {
033
034        /** Returns the text of the tree map node */
035        String getText();
036
037        /**
038         * Returns the list of children of this node. This usually is a readonly
039         * list.
040         */
041        List<ITreeMapNode<T>> getChildren();
042
043        /** Returns the area of this node including all subnodes. */
044        double getArea();
045
046        /** Returns the base color used for drawing this node. */
047        Color getColor();
048
049        /** Returns the color used for drawing the pattern (if any) of this node. */
050        Color getPatternColor();
051
052        /**
053         * Returns the pattern used for drawing the node (may be <code>null</code>
054         * to use no pattern).
055         */
056        IDrawingPattern getDrawingPattern();
057
058        /**
059         * Returns the rectangle this node was layouted into. If the tree was not
060         * yet layouted, this may be null, otherwise it should be the value set by
061         * {@link #setLayoutRectangle(Rectangle2D)}.
062         */
063        Rectangle2D getLayoutRectangle();
064
065        /** Sets the rectangle this node should be layouted into. */
066        void setLayoutRectangle(Rectangle2D rect);
067
068        /** Get displayable name of the node. */
069        String getTooltipId();
070
071        /** Returns keys for structured displayable data. */
072        List<String> getTooltipKeys();
073
074        /** Returns the value to be displayed for a single key. */
075        Object getTooltipValue(String key);
076
077        /**
078         * Re-calculates the aggregated area for this node and for all of its
079         * children. This is required after filtering treemap nodes.
080         * 
081         * @return the new area for this node.
082         */
083        double recalculateAreaAggregates();
084
085}
