017package org.conqat.lib.commons.color;
018
019import java.awt.Color;
020import java.util.function.Function;
021import java.util.regex.Pattern;
022
023import org.conqat.lib.commons.assertion.CCSMAssert;
024import org.conqat.lib.commons.enums.EnumUtils;
025
026/**
027 * Methods for handling colors.
028 */
029public class ColorUtils {
030
031        /** Pattern used for finding */
032        public static Pattern HTML_COLOR_PATTERN = Pattern.compile("#[0-9a-f]{6}", Pattern.CASE_INSENSITIVE);
033
034        /** Converts a color to a HTML color in the format #RRGGBB. */
035        public static String toHtmlString(Color color) {
036                return String.format("#%06X", color.getRGB() & 0xffffff);
037        }
038
039        /**
040         * Returns a color from a string. If the string could not be decoded, null is
041         * returned. This methods supports the html color format (i.e. #RRGGBB) and some
042         * predefined names (e.g. red, green, etc.).
043         */
044        public static Color fromString(String s) {
045                if (HTML_COLOR_PATTERN.matcher(s).matches()) {
046                        return Color.decode("0x" + s.substring(1));
047                }
048
049                if (s.startsWith("ccsm-")) {
050                        ECCSMColor color = EnumUtils.valueOfIgnoreCase(ECCSMColor.class, s.substring(5));
051                        if (color != null) {
052                                return color.getColor();
053                        }
054                }
055
056                EAWTColors color = EnumUtils.valueOfIgnoreCase(EAWTColors.class, s);
057                if (color != null) {
058                        return color.getColor();
059                }
060
061                return null;
062        }
063
064        /** List of colors defined in AWT used as a lookup table. */
065        private enum EAWTColors {
066
067                /** Red */
068                RED(Color.RED),
069
070                /** Green */
071                GREEN(Color.GREEN),
072
073                /** Blue */
074                BLUE(Color.BLUE),
075
076                /** Yellow */
077                YELLOW(Color.YELLOW),
078
079                /** Orange */
080                ORANGE(Color.ORANGE),
081
082                /** White */
083                WHITE(Color.WHITE),
084
085                /** Black */
086                BLACK(Color.BLACK),
087
088                /** Gray */
089                GRAY(Color.GRAY),
090
091                /** Cyan */
092                CYAN(Color.CYAN),
093
094                /** Magenta */
095                MAGENTA(Color.MAGENTA);
096
097                /** The color actual color. */
098                private final Color color;
099
100                /** Constructor. */
101                private EAWTColors(Color color) {
102                        this.color = color;
103                }
104
105                /** Returns the color for a enum constant. */
106                public Color getColor() {
107                        return color;
108                }
109        }
110
111        /**
112         * Blend together two colors, using the specified factor to indicate the weight
113         * given to the first color.
114         */
115        public static Color blend(double factor, Color color1, Color color2) {
116                CCSMAssert.isTrue(factor >= 0 && factor <= 1, "Factor must be between 0 and 1 but is: " + factor);
117                int r = getColorComponent(factor, color1, color2, Color::getRed);
118                int g = getColorComponent(factor, color1, color2, Color::getGreen);
119                int b = getColorComponent(factor, color1, color2, Color::getBlue);
120                return new Color(r, g, b);
121        }
122
123        /** Get the color component. */
124        protected static int getColorComponent(double factor, Color color1, Color color2,
125                        Function<Color, Integer> function) {
126                return (int) Math.round(factor * function.apply(color1) + (1.0 - factor) * function.apply(color2));
127        }
128}
