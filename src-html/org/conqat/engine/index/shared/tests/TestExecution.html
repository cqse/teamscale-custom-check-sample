046/**
047 * Representation of a single test (method) execution.
048 */
049@ExportToJavaScript
050public class TestExecution implements Serializable {
051
052        private static final long serialVersionUID = 1L;
053
054        /**
055         * The uniform path of the test (method) that was executed. This is an absolute
056         * (i.e. hierarchical) reference which identifies the test uniquely in the scope
057         * of the Teamscale project. It may (but is not required to) correspond to the
058         * path of some automated test case source code known to Teamscale. If the test
059         * was parameterized, this path is expected to reflect the parameter in some
060         * manner.
061         */
062        @JsonProperty("uniformPath")
063        private final String uniformPath;
064
065        /**
066         * Duration of the execution in seconds.
067         */
068        @JsonProperty("durationSeconds")
069        private final double durationSeconds;
070
071        /**
072         * The actual execution result state.
073         */
074        @JsonProperty("result")
075        private final ETestExecutionResult result;
076
077        /**
078         * Optional message given for test failures (normally contains a stack trace).
079         * For skipped and ignored tests the message will hold the message of skipped
080         * and ignored tests. e.g. "Flickers see TS-12345". May be {@code null}.
081         */
082        @JsonProperty("message")
083        @Nullable
084        private final String message;
085
086        private TestExecution(RelativeUniformPath uniformPath, double durationSeconds, ETestExecutionResult result,
087                        String message) {
088                this(uniformPath.toString(), durationSeconds, result, message);
089        }
090
091        @VisibleForTesting
092        TestExecution(String uniformPath, double durationSeconds, ETestExecutionResult result, String message) {
093                Preconditions.checkArgument(uniformPath != null, "Uniform path can't be null for test execution");
094                Preconditions.checkArgument(result != null, "Result can't be null for test execution");
095                Preconditions.checkArgument(durationSeconds >= 0, "Test duration can't be negative");
096                this.uniformPath = uniformPath;
097                this.durationSeconds = durationSeconds;
098                this.result = result;
099                this.message = message;
100        }
101
102        /**
103         * @see #uniformPath
104         */
105        public String getUniformPath() {
106                return uniformPath;
107        }
108
109        /**
110         * @see #durationSeconds
111         */
112        public double getDurationMillis() {
113                return durationSeconds * 1000.0;
114        }
115
116        /**
117         * @see #durationSeconds
118         */
119        public double getDurationSeconds() {
120                return durationSeconds;
121        }
122
123        /**
124         * @see #result
125         */
126        public ETestExecutionResult getResult() {
127                return result;
128        }
129
130        /**
131         * @see #message
132         */
133        public Optional<String> getMessage() {
134                return Optional.ofNullable(message);
135        }
136
137        @Override
138        public boolean equals(Object o) {
139                if (this == o) {
140                        return true;
141                }
142                if (o == null || getClass() != o.getClass()) {
143                        return false;
144                }
145                TestExecution that = (TestExecution) o;
146                return Double.compare(that.durationSeconds, durationSeconds) == 0
147                                && Objects.equals(uniformPath, that.uniformPath) && result == that.result
148                                && Objects.equals(message, that.message);
149        }
150
151        @Override
152        public int hashCode() {
153                return Objects.hash(uniformPath, durationSeconds, result, message);
154        }
155
156        @Override
157        public String toString() {
158                return ReflectionToStringBuilder.toString(this);
159        }
160
161        /**
162         * Returns whether something went wrong during execution of this test (either
163         * direct test failure or error during execution).
164         */
165        public boolean isFailure() {
166                return result == ETestExecutionResult.ERROR || result == ETestExecutionResult.FAILURE;
167        }
168
169        /**
170         * Create an assessment for this test result.
171         */
172        public Assessment createAssessment() {
173                switch (result) {
174                case PASSED:
175                        return new Assessment(ETrafficLightColor.GREEN);
176                case IGNORED:
177                        return new Assessment(ETrafficLightColor.YELLOW);
178                case SKIPPED:
179                        return new Assessment(ETrafficLightColor.YELLOW);
180                case ERROR:
181                        return new Assessment(ETrafficLightColor.RED);
182                case FAILURE:
183                        return new Assessment(ETrafficLightColor.RED);
184                default:
185                        return new Assessment(ETrafficLightColor.UNKNOWN);
186                }
187        }
188
189        /**
190         * Compute the test execution path.
191         */
192        public UniformPath toUniformPath() {
193                return TestUniformPathUtils.convertToUniformPath(uniformPath);
194        }
195
196        /**
197         * Merges the {@link TestExecution}s. The result is the worst
198         * {@link ETestExecutionResult} with the maximum encountered
199         * {@link TestExecution#durationSeconds}.
200         */
201        public static TestExecution merge(Collection<TestExecution> testExecutions) {
202                Preconditions.checkArgument(!testExecutions.isEmpty(), "Can't merge emtpy collection of test executions.");
203
204                if (testExecutions.size() == 1) {
205                        return Iterables.getOnlyElement(testExecutions);
206                }
207
208                Iterator<TestExecution> it = testExecutions.iterator();
209                TestExecution worstExecution = it.next();
210                while (it.hasNext()) {
211                        TestExecution nextExecution = it.next();
212                        Preconditions.checkArgument(nextExecution.uniformPath.equals(worstExecution.uniformPath),
213                                        "Can't merge test executions from separate uniform paths.");
214                        if (ETestExecutionResult.worst(nextExecution.result, worstExecution.result) != worstExecution.result) {
215                                worstExecution = nextExecution;
216                        }
217                }
218
219                double maxDurationSeconds = testExecutions.stream().mapToDouble(TestExecution::getDurationSeconds).max()
220                                .getAsDouble();
221
222                return new TestExecution(worstExecution.uniformPath, maxDurationSeconds, worstExecution.result,
223                                worstExecution.message);
224        }
225
226        /**
227         * Builder for creating {@link TestExecution} instances.
228         */
229        public static class Builder {
230
231                private static final Pattern UNESCAPED_SLASH = Pattern.compile("(?<!\\\\)/");
232
233                private final RelativeUniformPath uniformPath;
234
235                private double durationInSeconds = 0;
236
237                private ETestExecutionResult result;
238
239                private String failureMessage;
240
241                private Builder(RelativeUniformPath uniformPath) {
242                        this.uniformPath = uniformPath;
243                }
244
245                /**
246                 * Creates a builder for a fully qualified test class name and test method name.
247                 * This also supports (unescaped) slashes as separator.
248                 */
249                public static Builder fromClassAndName(String fullyQualifiedClassName, String testName) {
250                        // replace all slashes by a dot, so both dots and slashes build hierarchy
251                        fullyQualifiedClassName = UNESCAPED_SLASH.matcher(fullyQualifiedClassName).replaceAll(".");
252                        return fromPathAndName(Arrays.asList(fullyQualifiedClassName.split("\\.")), testName);
253                }
254
255                /**
256                 * Creates a builder for a relative uniform path.
257                 */
258                public static Builder fromPath(RelativeUniformPath path) {
259                        return new Builder(path);
260                }
261
262                /**
263                 * Creates a builder for a path and test name.
264                 */
265                private static Builder fromPathAndName(List<String> pathSegments, String testName) {
266                        return fromPath(RelativeUniformPath.of(pathSegments).addSuffix(UniformPath.escapeSegment(testName)));
267                }
268
269                /**
270                 * Sets the failure message. Trims leading whitespace on each line.
271                 */
272                public Builder setFailureMessage(String failureMessage) {
273                        if (failureMessage != null) {
274                                failureMessage = StringUtils.removeWhitespaceAtBeginningOfLine(failureMessage.trim());
275                                if (!failureMessage.isEmpty()) {
276                                        this.failureMessage = failureMessage;
277                                }
278                        }
279                        return this;
280                }
281
282                /**
283                 * Sets the test duration.
284                 */
285                public Builder setDuration(Duration duration) {
286                        this.durationInSeconds = duration.toMillis() / 1000d;
287                        return this;
288                }
289
290                /**
291                 * Sets the duration.
292                 */
293                public Builder setDurationInSeconds(double durationInSeconds) {
294                        this.durationInSeconds = durationInSeconds;
295                        return this;
296                }
297
298                /**
299                 * Sets the execution result.
300                 */
301                public Builder setResult(ETestExecutionResult result) {
302                        this.result = result;
303                        return this;
304                }
305
306                /**
307                 * Creates a new {@link TestExecution} instance from this builder.
308                 */
309                public TestExecution build() {
310                        return new TestExecution(uniformPath, durationInSeconds, result, failureMessage);
311                }
312
313                /**
314                 * Returns the absolute uniform path of this test execution including test
315                 * namespace/path and test name.
316                 */
317                public RelativeUniformPath getAbsolutePath() {
318                        return uniformPath;
319                }
320
321        }
