018package org.conqat.engine.index.shared;
019
020import java.io.Serializable;
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.List;
024
025import org.conqat.lib.commons.js_export.ExportToJavaScript;
026
027import com.fasterxml.jackson.annotation.JsonProperty;
028
029/** Result object that wraps a list of added and removed findings */
030@ExportToJavaScript
031public class FindingDelta implements Serializable {
032
033        /** Version used for serialization. */
034        private static final long serialVersionUID = 1;
035
036        /** The added findings. */
037        @JsonProperty("addedFindings")
038        private final List<TrackedFinding> addedFindings;
039
040        /** The findings in changed code. */
041        @JsonProperty("findingsInChangedCode")
042        private final List<TrackedFinding> findingsInChangedCode;
043
044        /** The removed findings. */
045        @JsonProperty("removedFindings")
046        private final List<TrackedFinding> removedFindings;
047
048        /** The summary of the finding churn. */
049        @JsonProperty("findingSummary")
050        private final List<FindingsSummaryCategoryInfo> findingSummary;
051
052        /**
053         * The number of added findings in the delta, regardless of any applied filters.
054         */
055        @JsonProperty("numberOfAddedFindings")
056        private final int numberOfAddedFindings;
057
058        /** The number of findings in code changed since the delta, */
059        @JsonProperty("numberOfFindingsInChangedCode")
060        private final int numberOfFindingsInChangedCode;
061
062        /**
063         * The number of removed findings in the delta, regardless of any applied
064         * filters.
065         */
066        @JsonProperty("numberOfRemovedFindings")
067        private final int numberOfRemovedFindings;
068
069        /** Constructor. */
070        public FindingDelta(Collection<TrackedFinding> addedFindings, Collection<TrackedFinding> findingsInChangedCode,
071                        Collection<TrackedFinding> removedFindings) {
072                this(addedFindings, findingsInChangedCode, removedFindings, addedFindings.size(), findingsInChangedCode.size(),
073                                removedFindings.size(), null);
074        }
075
076        /** Constructor. */
077        public FindingDelta(Collection<TrackedFinding> addedFindings, Collection<TrackedFinding> findingsInChangedCode,
078                        Collection<TrackedFinding> removedFindings, int numberOfAddedFindings, int numberOfFindingsInChangedCode,
079                        int numberOfRemovedFindings, List<FindingsSummaryCategoryInfo> findingSummary) {
080                this.numberOfAddedFindings = numberOfAddedFindings;
081                this.numberOfFindingsInChangedCode = numberOfFindingsInChangedCode;
082                this.numberOfRemovedFindings = numberOfRemovedFindings;
083                this.addedFindings = new ArrayList<>(addedFindings);
084                this.findingsInChangedCode = new ArrayList<>(findingsInChangedCode);
085                this.removedFindings = new ArrayList<>(removedFindings);
086                this.findingSummary = findingSummary;
087        }
088
089        /** Returns addedFindings. */
090        public List<TrackedFinding> getAddedFindings() {
091                return addedFindings;
092        }
093
094        /** Returns findingsInChangedCode. */
095        public List<TrackedFinding> getFindingsInChangedCode() {
096                return findingsInChangedCode;
097        }
098
099        /** Returns removedFindings. */
100        public List<TrackedFinding> getRemovedFindings() {
101                return removedFindings;
102        }
103
104        /** Returns findingSummary. */
105        public List<FindingsSummaryCategoryInfo> getFindingSummary() {
106                return findingSummary;
107        }
108
109        /** Returns numberOfAddedFindings. */
110        public int getNumberOfAddedFindings() {
111                return numberOfAddedFindings;
112        }
113
114        /** Returns numberOfFindingsInChangedCode. */
115        public int getNumberOfFindingsInChangedCode() {
116                return numberOfFindingsInChangedCode;
117        }
118
119        /** Returns numberOfRemovedFindings. */
120        public int getNumberOfRemovedFindings() {
121                return numberOfRemovedFindings;
122        }
123}
