017package org.conqat.engine.index.shared;
018
019import java.io.Serializable;
020import java.util.Arrays;
021import java.util.Comparator;
022import java.util.Objects;
023
024import org.conqat.engine.commons.util.JsonUtils;
025import org.conqat.engine.core.core.ConQATException;
026import org.conqat.engine.service.shared.ServiceUtils;
027import org.conqat.lib.commons.assertion.CCSMAssert;
028import org.conqat.lib.commons.io.ByteArrayUtils;
029import org.conqat.lib.commons.js_export.ExportToJavaScript;
030import org.conqat.lib.commons.string.StringUtils;
031
032import com.fasterxml.jackson.annotation.JsonCreator;
033import com.fasterxml.jackson.annotation.JsonProperty;
034
035/**
036 * Immutable class describing a single commit by its branch name and a
037 * timestamp. The timestamp must be unique within the branch. They are
038 * comparable by timestamp, where equal timestamps are resolved alphabetically
039 * by the branch name.
040 *
041 * <strong>This class is used for communication with IDE clients (via the
042 * {@link org.conqat.engine.service.shared.client.IdeServiceClient}), so special
043 * care has to be taken when changing its signature!</strong>
044 */
045@ExportToJavaScript
046public class CommitDescriptor implements Serializable, Comparable<CommitDescriptor> {
047
048        /**
049         * The separator used between branch name and timestamp in
050         * {@link #toBranchTimestampKeyWithSeparator()}.
051         */
052        public static final byte[] SEPARATOR = { 1, 2, 1 };
053
054        /**
055         * Name of the branch if no branch is specified.
056         * 
057         * @see org.conqat.engine.persistence.store.hist.HistoryAccessOption#NO_BRANCH_NAME
058         */
059        // HistoryAccessOption is not accessible in all project where this class is
060        // used as external source file
061        @SuppressWarnings("javadoc")
062        private static final String NO_BRANCH_NAME = "##no-branch##";
063
064        /** Version for serialization. */
065        private static final long serialVersionUID = 1L;
066
067        /** Special value to indicate the HEAD of a branch. */
068        public static final String HEAD_TIMESTAMP = "HEAD";
069
070        /**
071         * A comparator for comparison by timestamps. Null commits are ordered to the
072         * front.
073         * 
074         * If there are commits with identical timestamps, we order them by comparing
075         * the branch names to guarantee a stable order. This should only happen in
076         * special cases since teamscale ensures that no timestamp is given to two
077         * commits.
078         */
079        public static final Comparator<CommitDescriptor> BY_TIMESTAMP_COMPARATOR = Comparator.nullsFirst(
080                        Comparator.comparingLong(CommitDescriptor::getTimestamp).thenComparing(CommitDescriptor::getBranchName));
081
082        /** The name of the JSON property name for {@link #branchName}. */
083        private static final String BRANCH_NAME_PROPERTY = "branchName";
084
085        /** The name of the JSON property name for {@link #timestamp}. */
086        private static final String TIMESTAMP_PROPERTY = "timestamp";
087
088        /** The name of the branch. */
089        @JsonProperty(BRANCH_NAME_PROPERTY)
090        private final String branchName;
091
092        /** The timestamp on the branch. */
093        @JsonProperty(TIMESTAMP_PROPERTY)
094        private final long timestamp;
095
096        /**
097         * Constructor. <br/>
098         * use {@link CommitDescriptor#createUnbranchedDescriptor(long)} to create a
099         * unbranched commit descriptor.
100         */
101        @JsonCreator
102        public CommitDescriptor(@JsonProperty(BRANCH_NAME_PROPERTY) String branchName,
103                        @JsonProperty(TIMESTAMP_PROPERTY) long timestamp) {
104                CCSMAssert.isTrue(timestamp >= 0, "Timestamp must be >= 0 but is " + timestamp);
105                CCSMAssert.isNotNull(branchName);
106
107                this.branchName = branchName;
108                this.timestamp = timestamp;
109        }
110
111        /**
112         * Create a {@link CommitDescriptor} without branch specification. <br/>
113         * Should be used with caution.
114         */
115        public static CommitDescriptor createUnbranchedDescriptor(long timestamp) {
116                return new CommitDescriptor(NO_BRANCH_NAME, timestamp);
117        }
118
119        /**
120         * Create a copy of this commit descriptor with <code>timestamp - 1</code>.
121         * 
122         * Using this CommitDescriptor to store a new commit might overwrite an existing
123         * commit if the timestamp is already used.
124         */
125        public CommitDescriptor cloneWithDecrementedTimestamp() {
126                return new CommitDescriptor(this.branchName, this.timestamp - 1);
127        }
128
129        /**
130         * Create a copy of this commit descriptor with <code>timestamp + 1</code>.
131         * 
132         * Using this CommitDescriptor to store a new commit might overwrite an existing
133         * commit if the timestamp is already used.
134         */
135        public CommitDescriptor cloneWithIncrementedTimestamp() {
136                return new CommitDescriptor(this.branchName, this.timestamp + 1);
137        }
138
139        /**
140         * @see #branchName
141         */
142        public String getBranchName() {
143                return branchName;
144        }
145
146        /** Return true if no branch is specified. */
147        public boolean isUnbranched() {
148                return NO_BRANCH_NAME.equals(this.branchName);
149        }
150
151        /** Returns whether this commit is a head timestamp */
152        public boolean isHeadCommit() {
153                return timestamp == Long.MAX_VALUE;
154        }
155
156        /**
157         * @see #timestamp
158         */
159        public long getTimestamp() {
160                return timestamp;
161        }
162
163        /** {@inheritDoc} */
164        @Override
165        public boolean equals(Object obj) {
166                if (obj instanceof CommitDescriptor) {
167                        CommitDescriptor other = (CommitDescriptor) obj;
168                        return other.timestamp == timestamp && Objects.equals(this.branchName, other.branchName);
169                }
170                return false;
171        }
172
173        /** {@inheritDoc} */
174        @Override
175        public int hashCode() {
176                return Objects.hashCode(branchName) ^ Long.hashCode(timestamp);
177        }
178
179        /** {@inheritDoc} */
180        @Override
181        public int compareTo(CommitDescriptor other) {
182                if (timestamp == other.timestamp) {
183                        return branchName.compareTo(other.branchName);
184                }
185                return Long.compare(timestamp, other.timestamp);
186        }
187
188        /** Converts this to a JSON representation. */
189        public String toJson() {
190                return JsonUtils.serializeToJSON(this);
191        }
192
193        /** Reads a {@link CommitDescriptor} from a JSON string. */
194        public static CommitDescriptor fromJson(String json) throws ConQATException {
195                return JsonUtils.deserializeFromJsonWithNullCheck(json, CommitDescriptor.class);
196        }
197
198        /** {@inheritDoc} */
199        @Override
200        public String toString() {
201                return branchName + "@" + timestamp;
202        }
203
204        /** Returns a timestamp+branchName key/byte[] representation. */
205        public byte[] toTimestampBranchKey() {
206                return ByteArrayUtils.concat(ByteArrayUtils.longToByteArray(getTimestamp()),
207                                StringUtils.stringToBytes(getBranchName()));
208        }
209
210        /** Parses a commit descriptor from its toString() representation. */
211        public static CommitDescriptor fromStringRepresentation(String representation) {
212                int separatorPosition = representation.lastIndexOf("@");
213                CCSMAssert.isTrue(separatorPosition >= 0,
214                                () -> "Invalid string representation of commit descriptor: " + representation);
215                String branch = representation.substring(0, separatorPosition);
216                String timestamp = representation.substring(separatorPosition + 1);
217                return new CommitDescriptor(branch, Long.parseLong(timestamp));
218        }
219
220        /** Parses a timestamp+branchName key/byte[] representation. */
221        public static CommitDescriptor fromTimestampBranchKey(byte[] key) {
222                long timestamp = ByteArrayUtils.byteArrayToLong(Arrays.copyOf(key, Long.BYTES));
223                String branchName = StringUtils.bytesToString(Arrays.copyOfRange(key, Long.BYTES, key.length));
224                return new CommitDescriptor(branchName, timestamp);
225        }
226
227        /**
228         * Returns a branchName+timestamp key/byte[] representation. NOTE: This is
229         * *DANGEROUS* since keys generated with this function may cause unwanted branch
230         * names (prefixes of wanted branch names) to be returned from store scans
231         * (TS-16367). To protect against this, use
232         * {@link #toBranchTimestampKeyWithSeparator()}.
233         */
234        public byte[] toBranchTimestampKey() {
235                return ByteArrayUtils.concat(StringUtils.stringToBytes(getBranchName()),
236                                ByteArrayUtils.longToByteArray(getTimestamp()));
237        }
238
239        /**
240         * Returns a branchName+timestamp key/byte[] representation with
241         * {@link #SEPARATOR} in between.
242         */
243        public byte[] toBranchTimestampKeyWithSeparator() {
244                return ByteArrayUtils.concat(StringUtils.stringToBytes(getBranchName()), SEPARATOR,
245                                ByteArrayUtils.longToByteArray(getTimestamp()));
246        }
247
248        /** Parses a branchN+timestamp key/byte[] representation. */
249        public static CommitDescriptor fromBranchTimestampKey(byte[] key) {
250                String branchName = StringUtils.bytesToString(Arrays.copyOf(key, key.length - Long.BYTES));
251                long timestamp = ByteArrayUtils.byteArrayToLong(Arrays.copyOfRange(key, key.length - Long.BYTES, key.length));
252                return new CommitDescriptor(branchName, timestamp);
253        }
254
255        /** Creates a commit descriptor for the latest revision on a branch */
256        public static CommitDescriptor latestOnBranch(String branchName) {
257                return new CommitDescriptor(branchName, Long.MAX_VALUE);
258        }
259
260        /**
261         * Returns a format used for service calls ("branch:timestamp", or "timestamp"
262         * if no branch is specified).
263         * 
264         * @see #toEncodedPathParam()
265         * @see #toEncodedPathParam()
266         */
267        public String toServiceCallFormat() {
268                String result = branchName + ":";
269                if (NO_BRANCH_NAME.contentEquals(branchName)) {
270                        result = StringUtils.EMPTY_STRING;
271                }
272
273                if (timestamp == Long.MAX_VALUE) {
274                        return result + HEAD_TIMESTAMP;
275                }
276                return result + timestamp;
277        }
278
279        /**
280         * Returns a url-encoded format used for service calls ("branch:timestamp", or
281         * "timestamp" if no branch is specified). The format is suitable for calls that
282         * include the commit as query parameter or in the arguments of the navigation
283         * hash.
284         */
285        public String toEncodedQueryParam() {
286                return ServiceUtils.encodeQueryParameter(toServiceCallFormat());
287        }
288
289        /**
290         * Returns a url-encoded format used for service calls ("branch:timestamp", or
291         * "timestamp" if no branch is specified). The format is suitable for calls that
292         * include the commit as path parameter.
293         */
294        public String toEncodedPathParam() {
295                return ServiceUtils.encodePathSegment(toServiceCallFormat());
296        }
297}
