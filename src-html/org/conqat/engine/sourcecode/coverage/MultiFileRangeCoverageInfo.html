017package org.conqat.engine.sourcecode.coverage;
018
019import java.io.Serializable;
020import java.util.Comparator;
021import java.util.HashMap;
022import java.util.Map;
023import java.util.Map.Entry;
024import java.util.Set;
025import java.util.stream.Collectors;
026
027import org.conqat.lib.commons.collections.IntList;
028
029/**
030 * Holds line range coverage information for a test covering multiple files.
031 */
032public class MultiFileRangeCoverageInfo implements Serializable {
033
034        private static final long serialVersionUID = 1L;
035
036        /**
037         * Mapping from normalized uniform path to a list of line numbers. The list may
038         * contain duplicates and may be unsorted.
039         */
040        private final Map<String, IntList> uniformPathToRanges = new HashMap<>();
041
042        /**
043         * The timestamp of the code this coverage data refers to. May be -1 if unknown.
044         */
045        private long timestamp;
046
047        /** Constructor with a default timestamp. */
048        public MultiFileRangeCoverageInfo() {
049                this(-1);
050        }
051
052        public MultiFileRangeCoverageInfo(long timestamp) {
053                this.timestamp = timestamp;
054        }
055
056        /**
057         * Merges all coverage contained in newCoverage into this coverage. Coverage
058         * from the same test case replaces existing coverage.
059         */
060        public void addLineCoverage(MultiFileRangeCoverageInfo newCoverage) {
061                Set<Entry<String, IntList>> newTestCoverage = newCoverage.uniformPathToRanges.entrySet();
062                for (Entry<String, IntList> testCoverage : newTestCoverage) {
063                        uniformPathToRanges.put(testCoverage.getKey(), testCoverage.getValue());
064                }
065        }
066
067        /** Adds the coverage information for the given lines. */
068        public void addLineCoverage(String uniformPath, IntList lines) {
069                uniformPathToRanges.computeIfAbsent(uniformPath, k -> new IntList()).addAll(lines);
070        }
071
072        /** @see #timestamp */
073        public long getTimestamp() {
074                return timestamp;
075        }
076
077        /** @see #timestamp */
078        public void setTimestamp(long timestamp) {
079                this.timestamp = timestamp;
080        }
081
082        /** Returns the coverage represented by this object in raw form. */
083        public Map<String, IntList> getCoverage() {
084                return uniformPathToRanges;
085        }
086
087        /**
088         * {@inheritDoc}
089         */
090        @Override
091        public String toString() {
092                return "Covered: "
093                                + uniformPathToRanges.entrySet().stream().sorted(Comparator.comparing(Entry::getKey))
094                                                .map((pair) -> pair.getKey() + " " + pair.getValue()).collect(Collectors.joining(", "))
095                                + "; timestamp: " + timestamp;
096        }
097
098        /**
099         * If the test contains coverage from the given file, the uniform path is
100         * adjusted to match the path used inside Teamscale.
101         */
102        public void changeFilePath(String reportPath, String pathInTeamscale) {
103                if (uniformPathToRanges.containsKey(reportPath) && !reportPath.equals(pathInTeamscale)) {
104                        IntList ranges = uniformPathToRanges.remove(reportPath);
105                        uniformPathToRanges.put(pathInTeamscale, ranges);
106                }
107        }
108}
