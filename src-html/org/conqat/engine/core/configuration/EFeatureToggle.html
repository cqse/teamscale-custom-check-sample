017package org.conqat.engine.core.configuration;
018
019import org.conqat.lib.commons.js_export.ExportToJavaScript;
020
021/**
022 * Enum containing all available feature toggles in Teamscale. Feature toggles
023 * are enabled via System properties.
024 */
025@ExportToJavaScript
026public enum EFeatureToggle {
027
028        /**
029         * Enables the code usage widget that shows execution counts per method. This is
030         * currently only usable for ABAP projects with direct coverage report upload.
031         * The toggle will be removed once code usage analysis supports more languages.
032         */
033        CODE_USAGE_WIDGET("com.teamscale.feature-toggle.code-usage-widget"),
034
035        /**
036         * Feature configuration for coverage computations (filter coverable volumes).
037         */
038        BMW_COVERABLE_VOLUME_FILTER("bmw.teamscale.feature-toggle.coverable-volume-filter"),
039
040        /**
041         * Disable the architecture regex overlap check. Use this if you are using super
042         * huge architectures that slow down the architecture analysis significantly.
043         */
044        DISABLE_ARCHITECTURE_OVERLAP_CHECK("com.teamscale.feature-toggle.disable-architecture-overlap-check"),
045
046        /** Enables the landing page that is used for our own demo instance. */
047        ENABLE_DEMO_LANDING_PAGE("com.teamscale.landing.enable"),
048
049        /** Enables debug log level for the eclipse plugin. */
050        ECLIPSE_PLUGIN_DEBUG_LOG("com.teamscale.eclipse.debug"),
051
052        /**
053         * Disable the check for sufficient memory. Doesn't make sense in some scenarios
054         * (e.g. cloud)
055         */
056        DISABLE_MEMORY_CHECK("com.teamscale.disable-memory-check"),
057
058        /**
059         * Disables the commit chart widget that shows the commits per user.
060         */
061        DISABLE_COMMIT_CHART_WIDGET("com.teamscale.feature-toggle.disable-commit-chart-widget"),
062
063        /** Disables the automatic creation of webhooks for repository connectors. */
064        DISABLE_AUTOMATIC_WEBHOOK_CREATION("com.teamscale.feature-toggle.disable-automatic-webhook-creation"),
065
066        /**
067         * Overwrite coverable lines calculated by Teamscale with information from
068         * coverage reports.
069         */
070        USE_COVERABLE_LINES_FROM_COVERAGE_REPORTS("com.teamscale.feature-toggle.use-coverable-lines-from-coverage-reports"),
071
072        /**
073         * Enables support for reading legacy ConQAT in-file review ratings.
074         */
075        ENABLE_CONQAT_INFILE_RATING("com.teamscale.feature-toggle.conqat-infile-rating"),
076
077        /** Enables the Reports perspective (work in progress) */
078        ENABLE_REPORTS_PERSPECTIVE("com.teamscale.ui.reports-enabled");
079
080        /**
081         * The feature toggle's id. Also used as key for the corresponding System
082         * property.
083         */
084        private String id;
085
086        /** Returns whether the given feature toggle is enabled or not. */
087        public boolean isEnabled() {
088                return Boolean.getBoolean(id);
089        }
090
091        /** Constructor */
092        private EFeatureToggle(String id) {
093                this.id = id;
094        }
095
096        /** @see #id */
097        public String getId() {
098                return id;
099        }
100
101        /** Allows to enable/disable the feature toggle. */
102        public void setFeatureToggle(boolean value) {
103                System.setProperty(id, Boolean.toString(value));
104
105        }
106}
