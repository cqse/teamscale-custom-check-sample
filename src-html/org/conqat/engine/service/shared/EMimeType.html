017package org.conqat.engine.service.shared;
018
019import java.util.HashMap;
020import java.util.Map;
021import java.util.Set;
022
023/**
024 * Enumeration of MIME types used/supported. This is by no means meant to cover
025 * all existing MIME types, but only those used in this application.
026 */
027public enum EMimeType {
028
029        /** Plain text. */
030        PLAIN("text/plain"),
031
032        /** HTML text. */
033        HTML("text/html", "application/xhtml+xml"),
034
035        /** XML text. */
036        XML("text/xml", "application/xml"),
037
038        /** JavaScript code. */
039        JAVASCRIPT("text/javascript", "application/x-javascript"),
040
041        /** CSS code. */
042        CSS("text/css"),
043
044        /** CSV, i.e. comma-separated values. */
045        CSV("text/csv", "application/vnd.ms-excel"),
046
047        /** JSON code. */
048        JSON("application/json"),
049
050        /** multipart/form-data */
051        MULTIPART_FORM_DATA("multipart/form-data"),
052
053        /** Url encoded form data. */
054        URL_ENCODED_FORM_DATA("application/x-www-form-urlencoded"),
055
056        /** Icon format. */
057        ICON("image/x-icon"),
058
059        /** GIF image format. */
060        GIF("image/gif"),
061
062        /** JPEG image format. */
063        JPEG("image/jpeg"),
064
065        /** PNG image format. */
066        PNG("image/png", "image/x-png"),
067
068        /** SVG vector image format. */
069        SVG("image/svg+xml"),
070
071        /** Fonts (general format). */
072        FONT("application/x-font"),
073
074        FONT_TTF("font/ttf"),
075
076        FONT_WOFF("font/woff"),
077
078        FONT_WOFF_2("font/woff2"),
079
080        /** MP3 files. */
081        MP3("audio/mpeg3"),
082
083        /** ZIP. */
084        ZIP("application/zip"),
085
086        /** Octet stream (binary data). */
087        OCTET_STREAM("application/octet-stream"),
088
089        /** SOAP request (sent e.g. by the TFS build server). */
090        SOAP("application/soap+xml"),
091
092        /** Portable document format. */
093        PDF("application/pdf"),
094
095        /** Markdown */
096        MD("text/markdown");
097
098        /** Map allowing lookup of type string to enum literal. */
099        private static final Map<String, EMimeType> TYPE_MAP = new HashMap<>();
100
101        static {
102                /** Initializes the {@link #TYPE_MAP}. */
103                for (EMimeType value : values()) {
104                        TYPE_MAP.put(value.type, value);
105                        for (String additionalType : value.additionalTypes) {
106                                TYPE_MAP.put(additionalType, value);
107                        }
108                }
109        }
110
111        /** The type. */
112        private final String type;
113
114        /** Additional types to register for. */
115        private final String[] additionalTypes;
116
117        /** Constructor. */
118        EMimeType(String type, String... additionalTypes) {
119                this.type = type;
120                this.additionalTypes = additionalTypes;
121        }
122
123        /** Returns the type string. */
124        public String getType() {
125                return type;
126        }
127
128        /** Returns the enum literal for a given type string (or null). */
129        public static EMimeType fromType(String type) {
130                if (type == null) {
131                        return XML;
132                }
133                return TYPE_MAP.get(removeCharsetFromMimeType(type));
134        }
135
136        /** Returns all supported mime types as a Set of Strings. */
137        public static Set<String> getAllSupportedMimeTypes() {
138                return TYPE_MAP.keySet();
139        }
140
141        /**
142         * remove charset portion, browser may append additional information after a
143         * semicolon example: application/xml;q=0.9
144         */
145        private static String removeCharsetFromMimeType(String type) {
146                return type.replaceFirst(";.*$", "");
147        }
148
149}
