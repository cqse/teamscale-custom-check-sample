017package org.conqat.engine.service.shared;
018
019import java.util.HashMap;
020import java.util.Map;
021import java.util.Set;
022
023/**
024 * Enumeration of MIME types used/supported. This is by no means meant to cover
025 * all existing MIME types, but only those used in this application.
026 */
027public enum EMimeType {
028
029        /** Plain text. */
030        PLAIN("text/plain"),
031
032        /** HTML text. */
033        HTML("text/html", "application/xhtml+xml"),
034
035        /** XML text. */
036        XML("text/xml", "application/xml"),
037
038        /** JavaScript code. */
039        JAVASCRIPT("text/javascript", "application/x-javascript"),
040
041        /** CSS code. */
042        CSS("text/css"),
043
044        /** CSV, i.e. comma-separated values. */
045        CSV("text/csv", "application/vnd.ms-excel"),
046
047        /** JSON code. */
048        JSON("application/json"),
049
050        /** multipart/form-data */
051        MULTIPART_FORM_DATA("multipart/form-data"),
052
053        /** Icon format. */
054        ICON("image/x-icon"),
055
056        /** GIF image format. */
057        GIF("image/gif"),
058
059        /** JPEG image format. */
060        JPEG("image/jpeg"),
061
062        /** PNG image format. */
063        PNG("image/png", "image/x-png"),
064
065        /** SVG vector image format. */
066        SVG("image/svg+xml"),
067
068        /** Fonts (general format). */
069        FONT("application/x-font"),
070
071        FONT_TTF("font/ttf"),
072
073        FONT_WOFF("font/woff"),
074
075        FONT_WOFF_2("font/woff2"),
076
077        /** MP3 files. */
078        MP3("audio/mpeg3"),
079
080        /** ZIP. */
081        ZIP("application/zip"),
082
083        /** Octet stream (binary data). */
084        OCTET_STREAM("application/octet-stream"),
085
086        /** SOAP request (sent e.g. by the TFS build server). */
087        SOAP("application/soap+xml"),
088
089        /** Portable document format. */
090        PDF("application/pdf"),
091
092        /** Markdown */
093        MD("text/markdown");
094
095        /** Map allowing lookup of type string to enum literal. */
096        private static final Map<String, EMimeType> TYPE_MAP = new HashMap<>();
097
098        static {
099                /** Initializes the {@link #TYPE_MAP}. */
100                for (EMimeType value : values()) {
101                        TYPE_MAP.put(value.type, value);
102                        for (String additionalType : value.additionalTypes) {
103                                TYPE_MAP.put(additionalType, value);
104                        }
105                }
106        }
107
108        /** The type. */
109        private final String type;
110
111        /** Additional types to register for. */
112        private final String[] additionalTypes;
113
114        /** Constructor. */
115        EMimeType(String type, String... additionalTypes) {
116                this.type = type;
117                this.additionalTypes = additionalTypes;
118        }
119
120        /** Returns the type string. */
121        public String getType() {
122                return type;
123        }
124
125        /** Returns the enum literal for a given type string (or null). */
126        public static EMimeType fromType(String type) {
127                if (type == null) {
128                        return XML;
129                }
130                return TYPE_MAP.get(removeCharsetFromMimeType(type));
131        }
132
133        /** Returns all supported mime types as a Set of Strings. */
134        public static Set<String> getAllSupportedMimeTypes() {
135                return TYPE_MAP.keySet();
136        }
137
138        /**
139         * remove charset portion, browser may append additional information after a
140         * semicolon example: application/xml;q=0.9
141         */
142        private static String removeCharsetFromMimeType(String type) {
143                return type.replaceFirst(";.*$", "");
144        }
145
146}
