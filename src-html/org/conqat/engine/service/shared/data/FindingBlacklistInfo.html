008package org.conqat.engine.service.shared.data;
009
010import java.io.Serializable;
011import java.util.Objects;
012
013import javax.annotation.Nullable;
014
015import org.conqat.lib.commons.js_export.ExportToJavaScript;
016import org.conqat.lib.commons.string.StringUtils;
017
018import com.fasterxml.jackson.annotation.JsonCreator;
019import com.fasterxml.jackson.annotation.JsonProperty;
020
021/**
022 * Information about a blacklisted finding.
023 *
024 * <strong>This class is used for communication with IDE clients (via the @link
025 * {@link org.conqat.engine.service.shared.client.IdeServiceClient}, special
026 * care has to be taken when changing its signature!</strong>
027 */
028@ExportToJavaScript
029public class FindingBlacklistInfo implements Serializable {
030
031        /** Serial version UID. */
032        private static final long serialVersionUID = 1;
033
034        /** The name of the JSON property name for {@link #findingId}. */
035        private static final String FINDING_ID_PROPERTY = "findingId";
036
037        /** The name of the JSON property name for {@link #timestamp}. */
038        private static final String TIMESTAMP_PROPERTY = "timestamp";
039
040        /** The name of the JSON property name for {@link #user}. */
041        private static final String USER_PROPERTY = "user";
042
043        /** The name of the JSON property name for {@link #rationale}. */
044        private static final String RATIONALE_PROPERTY = "rationale";
045
046        /** The name of the JSON property name for {@link #type}. */
047        private static final String TYPE_PROPERTY = "type";
048
049        /** The ID of the blacklisted finding */
050        @JsonProperty(FINDING_ID_PROPERTY)
051        private final String findingId;
052
053        /** The time when the finding was blacklisted */
054        @JsonProperty(TIMESTAMP_PROPERTY)
055        private final long timestamp;
056
057        /** The name of the user who blacklisted the finding */
058        @JsonProperty(USER_PROPERTY)
059        private final String user;
060
061        /** A rationale why the finding was blacklisted */
062        @JsonProperty(RATIONALE_PROPERTY)
063        private final String rationale;
064
065        @JsonProperty(TYPE_PROPERTY)
066        @Nullable
067        private final EFindingBlacklistType type;
068
069        /** Copy constructor. */
070        public FindingBlacklistInfo(FindingBlacklistInfo info) {
071                this.findingId = info.findingId;
072                this.timestamp = info.timestamp;
073                this.user = info.user;
074                this.rationale = info.rationale;
075                this.type = info.type;
076        }
077
078        /** Constructor. */
079        @JsonCreator
080        public FindingBlacklistInfo(@JsonProperty(FINDING_ID_PROPERTY) String findingId,
081                        @JsonProperty(TIMESTAMP_PROPERTY) long timestamp, @JsonProperty(USER_PROPERTY) String user,
082                        @JsonProperty(RATIONALE_PROPERTY) String rationale,
083                        @JsonProperty(TYPE_PROPERTY) EFindingBlacklistType type) {
084                this.findingId = findingId;
085                this.timestamp = timestamp;
086                this.user = user;
087                this.rationale = rationale;
088                this.type = type;
089        }
090
091        /** Clones this FindingBlacklistInfo but uses the given findingId. */
092        public FindingBlacklistInfo cloneWithNewFindingId(String findingId) {
093                return new FindingBlacklistInfo(findingId, timestamp, user, rationale, type);
094        }
095
096        /** Clones this FindingBlacklistInfo but uses the given findingId. */
097        public FindingBlacklistInfo cloneWithNewType(EFindingBlacklistType newType) {
098                return new FindingBlacklistInfo(findingId, timestamp, user, rationale, newType);
099        }
100
101        /**
102         * Ensures that the type of blacklisting is set. If it is null it is tried to
103         * heuristically derive the type from the rationale. The default is
104         * {@link EFindingBlacklistType#TOLERATION}.
105         */
106        public static FindingBlacklistInfo ensureBlacklistTypeSet(FindingBlacklistInfo blacklistInfo) {
107                if (blacklistInfo.getType() != null) {
108                        return blacklistInfo;
109                }
110                String rationale = blacklistInfo.getRationale();
111                if (!StringUtils.isEmpty(rationale) && rationale.toLowerCase().contains("false")
112                                && rationale.toLowerCase().contains("positive")) {
113                        return blacklistInfo.cloneWithNewType(EFindingBlacklistType.FALSE_POSITIVE);
114                }
115                return blacklistInfo.cloneWithNewType(EFindingBlacklistType.TOLERATION);
116        }
117
118        /** Returns the finding id. */
119        public String getFindingId() {
120                return findingId;
121        }
122
123        /** Returns the timestamp. */
124        public long getTimestamp() {
125                return timestamp;
126        }
127
128        /** Returns the user who blacklisted the finding. */
129        public String getUser() {
130                return user;
131        }
132
133        /** Returns the rationale. */
134        public String getRationale() {
135                return rationale;
136        }
137
138        /**
139         * Returns the blacklist type. This can be <code>null</code> for backwards
140         * compatibility
141         */
142        public EFindingBlacklistType getType() {
143                return type;
144        }
145
146        /** {@inheritDoc} */
147        @Override
148        public final boolean equals(Object obj) {
149                if (obj instanceof FindingBlacklistInfo) {
150                        FindingBlacklistInfo other = (FindingBlacklistInfo) obj;
151                        return Objects.equals(findingId, other.findingId) && Objects.equals(rationale, other.rationale)
152                                        && timestamp == other.timestamp && Objects.equals(user, other.user)
153                                        && Objects.equals(type, other.type);
154                }
155                return false;
156        }
157
158        /** {@inheritDoc} */
159        @Override
160        public final int hashCode() {
161                return Objects.hash(findingId, rationale, timestamp, user, type);
162        }
163
164}
