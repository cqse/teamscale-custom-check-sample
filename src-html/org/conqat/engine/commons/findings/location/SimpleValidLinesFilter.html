001package org.conqat.engine.commons.findings.location;
002
003import java.util.Set;
004
005import org.conqat.lib.commons.region.LineBasedRegion;
006import org.conqat.lib.commons.string.StringUtils;
007
008/**
009 * Simple line adjuster is just a specialized form of {@link LocationAdjuster}
010 * that performs way faster when only line adjustments are needed and the
011 * content to be adjusted between is exactly the same. It basically only does
012 * discard lines that are outside of the file content.
013 */
014public class SimpleValidLinesFilter implements ILineAdjuster {
015
016        private final int totalLineCount;
017
018        public SimpleValidLinesFilter(String content) {
019                totalLineCount = StringUtils.countLines(content);
020        }
021
022        /** Returns whether the given line is valid for the underlying content. */
023        private boolean isValidLine(int line) {
024                return line > 0 && line <= totalLineCount;
025        }
026
027        @Override
028        public LineBasedRegion adjustLine(int line, Set<Integer> invalidLines) {
029                if (!isValidLine(line)) {
030                        invalidLines.add(line);
031                        return null;
032                }
033                return new LineBasedRegion(line, line);
034        }
035
036        /** Returns the line count of the original text */
037        @Override
038        public int getOriginalLineCount() {
039                return totalLineCount;
040        }
041}
